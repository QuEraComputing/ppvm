11/// Output format for `parse`.
12#[derive(Clone, Debug, clap::ValueEnum)]
13pub enum Format {
14    Pretty,
15    Debug,
16    Json,
17}
18
19/// Output format for the measurement record from `run`.
20#[derive(Clone, Debug, clap::ValueEnum)]
21pub enum MeasurementFormat {
22    /// One flat bit string per shot: `0`/`1`, lost qubit = `2`.
23    Bits,
24    /// Raw debug representation of all shots.
25    Debug,
26}
27
28pub fn run(
29    file: &str,
30    shots: usize,
31    seed: Option<u64>,
32    output: Option<&str>,
33    quiet: bool,
34    format: MeasurementFormat,
35) -> Result<()> {
36    // Compile once, then run every shot against the shared module. The thread
37    // pool is sized once in `main` via the top-level `--threads` flag.
38    let module =
39        ppvm_vihaco::load_module_file(file).wrap_err_with(|| format!("failed to load {file}"))?;
40    let records = ppvm_vihaco::shots::run_shots(&module, shots, seed)
41        .wrap_err_with(|| format!("failed to run {file}"))?;
42    if quiet {
43        return Ok(());
44    }
45
46    let text = match format {
47        MeasurementFormat::Bits => format_shot_records(&records),
48        MeasurementFormat::Debug => format!("{records:?}"),
49    };
50
51    match output {
52        Some(path) => {
53            std::fs::write(path, format!("{text}\n"))
54                .wrap_err_with(|| format!("failed to write {path}"))?;
55            eprintln!("Results written to {path}");
56        }
57        None => println!("{text}"),
58    }
59    Ok(())
60}

62/// Render one shot per line.
63fn format_shot_records(records: &[ShotRecord]) -> String {
64    records
65        .iter()
66        .map(format_shot_record)
67        .collect::<Vec<_>>()
68        .join("\n")
69}
70
71/// Render one shot. Layout:
72/// - Tableau-style (only measurements): `bits` — same as the pre-trace format.
73/// - PauliSum-style (only traces): comma-separated floats.
74/// - Both present: `bits | t0,t1,...`.
75/// - Both empty: empty string.
76fn format_shot_record(record: &ShotRecord) -> String {
77    let bits = format_shot(&record.measurements);
78    let traces = format_traces(&record.traces);
79    match (bits.is_empty(), traces.is_empty()) {
80        (false, false) => format!("{bits} | {traces}"),
81        (false, true) => bits,
82        (true, false) => traces,
83        (true, true) => String::new(),
84    }
85}
86
87/// Render a shot's full measurement record as one flat bit string, all events
88/// and qubits concatenated: `Zero` → `0`, `One` → `1`, `Lost` → `2` (the
89/// outcome's own enum value). An empty record renders as the empty string.
90fn format_shot(record: &[MeasurementResult]) -> String {
91    record
92        .iter()
93        .flatten()
94        .map(|outcome| char::from(b'0' + *outcome as u8))
95        .collect()
96}
97
98/// Render trace values as comma-separated floats with default `f64` formatting
99/// (`1.0` → `"1"`, `1.5` → `"1.5"`, etc.). Empty record renders as empty string.
100fn format_traces(traces: &[f64]) -> String {
101    traces
102        .iter()
103        .map(|t| format!("{t}"))
104        .collect::<Vec<_>>()
105        .join(",")
106}

108pub fn parse(file: &str, format: Format) -> Result<()> {
109    let source =
110        std::fs::read_to_string(file).wrap_err_with(|| format!("failed to read {file}"))?;
111    let parsed = ppvm_vihaco::parse_program(&source)?;
112
113    match format {
114        Format::Json => {
115            eprintln!("Warning: JSON format not yet supported for AST, using debug format");
116            println!("{:#?}", parsed);
117        }
118        Format::Debug => {
119            println!("{:#?}", parsed);
120        }
121        Format::Pretty => {
122            println!("Module:");
123            println!("  Headers: {}", parsed.headers.len());
124            for (i, header) in parsed.headers.iter().enumerate() {
125                println!("    [{}] {:?}", i, header);
126            }
127            println!("  Functions: {}", parsed.functions.len());
128            for (i, func) in parsed.functions.iter().enumerate() {
129                println!(
130                    "    [{}] {}({} params, {} body items)",
131                    i,
132                    func.name,
133                    func.params.len(),
134                    func.body.len()
135                );
136            }
137        }
138    }
139
140    Ok(())
141}

143pub fn dump(file: &str, output: Option<&str>, force: bool) -> Result<()> {
144    let output_file = match output {
145        Some(output_file_name) => output_file_name.to_string(),
146        None => Path::new(file)
147            .with_extension("ssb")
148            .to_string_lossy()
149            .into_owned(),
150    };
151
152    // Don't clobber an existing file unless asked to.
153    if !force && Path::new(&output_file).exists() {
154        return Err(eyre::eyre!(
155            "{output_file} already exists; pass --force to overwrite"
156        ));
157    }
158
159    ppvm_vihaco::dump_file(file, &output_file)
160        .wrap_err_with(|| format!("failed to dump {file}"))?;
161    eprintln!("Bytecode written to {output_file}");
162    Ok(())
163}

165/// A command entered at the debugger prompt.
166enum DebugCommand {
167    Step,
168    Continue,
169    Quit,
170}

172/// Step through a program interactively, pausing at `breakpoint` instructions.
173/// With `break_at_start`, also pauses before the first instruction so any
174/// program can be stepped from the beginning.
175pub fn debug(file: &str, break_at_start: bool) -> Result<()> {
176    let stdin = std::io::stdin();
177    let mut input = stdin.lock();
178    let mut output = std::io::stdout();
179    debug_loop(file, break_at_start, &mut input, &mut output)
180}
181
182/// Core debugger loop, generic over its IO so it can be driven by tests.
183fn debug_loop(
184    file: &str,
185    break_at_start: bool,
186    input: &mut impl BufRead,
187    output: &mut impl Write,
188) -> Result<()> {
189    let mut machine = PPVM::default();
190    machine
191        .load_file(file)
192        .wrap_err_with(|| format!("failed to load {file}"))?;
193    machine.init()?;
194
195    let mut paused = break_at_start;
196    let mut ever_paused = paused;
197
198    loop {
199        // Safety net: stop if execution has run off the end of the code.
200        if machine.current_instruction().is_none() {
201            writeln!(output, "Program counter past end of code.")?;
202            break;
203        }
204
205        if paused {
206            print_location(&machine, output)?;
207            match read_command(input, output)? {
208                DebugCommand::Quit => {
209                    writeln!(output, "Quit.")?;
210                    return Ok(());
211                }
212                DebugCommand::Continue => paused = false,
213                DebugCommand::Step => {}
214            }
215        }
216
217        match machine.step_once()? {
218            StepOutcome::Continue => {}
219            StepOutcome::Breakpoint => {
220                paused = true;
221                ever_paused = true;
222                writeln!(output, "-- breakpoint hit --")?;
223            }
224            StepOutcome::Return | StepOutcome::Halt => {
225                writeln!(output, "Program finished.")?;
226                break;
227            }
228        }
229    }
230
231    writeln!(
232        output,
233        "Measurements: {}",
234        format_shot(&machine.measurement_record())
235    )?;
236    let traces = machine.trace_record();
237    if !traces.is_empty() {
238        writeln!(output, "Traces: {}", format_traces(&traces))?;
239    }
240    if !ever_paused {
241        writeln!(
242            output,
243            "(no breakpoint was hit; pass --break-at-start to step from the beginning)"
244        )?;
245    }
246    Ok(())
247}

249/// Print the program counter, the next instruction, and measurements so far.
250fn print_location(machine: &PPVM, output: &mut impl Write) -> Result<()> {
251    let pc = machine.current_pc();
252    match machine.current_instruction() {
253        Some(inst) => writeln!(output, "pc={pc}  next: {inst}")?,
254        None => writeln!(output, "pc={pc}  (end of code)")?,
255    }
256    writeln!(
257        output,
258        "measurements: {}",
259        format_shot(&machine.measurement_record())
260    )?;
261    let traces = machine.trace_record();
262    if !traces.is_empty() {
263        writeln!(output, "traces: {}", format_traces(&traces))?;
264    }
265    Ok(())
266}

268/// Prompt for and read a debugger command. A bare Enter steps; EOF quits.
269fn read_command(input: &mut impl BufRead, output: &mut impl Write) -> Result<DebugCommand> {
270    loop {
271        write!(output, "> s step | c continue | q quit: ")?;
272        output.flush()?;
273
274        let mut line = String::new();
275        if input.read_line(&mut line)? == 0 {
276            // EOF (e.g. stdin closed): treat as quit so we never spin.
277            return Ok(DebugCommand::Quit);
278        }
279        match line.trim() {
280            "" | "s" | "step" => return Ok(DebugCommand::Step),
281            "c" | "continue" => return Ok(DebugCommand::Continue),
282            "q" | "quit" => return Ok(DebugCommand::Quit),
283            other => writeln!(output, "unknown command: {other:?}")?,
284        }
285    }
286}
