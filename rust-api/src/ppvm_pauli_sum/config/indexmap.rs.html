4use std::marker::PhantomData;
5
6use ppvm_pauli_word::word::PauliWord;
7use ppvm_traits::config::Config;
8use ppvm_traits::traits::{Coefficient, NoStrategy, PauliWordTrait, Strategy};
9
10/// `IndexMap`-backed [`Config`] with `[u8; N]` storage and `FxHasher`.
11#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord, Hash)]
12pub struct ByteFxHash<
13    const N: usize,
14    C: Coefficient,
15    St: Strategy = NoStrategy,
16    W: PauliWordTrait = PauliWord<[u8; N], fxhash::FxBuildHasher>,
17>(PhantomData<(C, St, W)>);
18
19impl<const N: usize, C: Coefficient, St: Strategy, W: PauliWordTrait> Config
20    for ByteFxHash<N, C, St, W>
21{
22    type Storage = [u8; N];
23    type Coeff = C;
24    type BuildHasher = fxhash::FxBuildHasher;
25    type PauliWordType = W;
26    type Map = indexmap::IndexMap<Self::PauliWordType, Self::Coeff, Self::BuildHasher>;
27    type Strategy = St;
28}
29
30/// `IndexMap`-backed [`Config`] with `[u8; N]` storage and `gxhash`.
31///
32/// `gxhash` is AES-based and native-only, so this config is unavailable on
33/// `wasm32`; use [`ByteFxHash`] there.
34#[cfg(all(feature = "gxhash", not(target_arch = "wasm32")))]
35#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord, Hash)]
36pub struct ByteGxHash<
37    const N: usize,
38    C: Coefficient,
39    St: Strategy = NoStrategy,
40    W: PauliWordTrait = PauliWord<[u8; N], gxhash::GxBuildHasher>,
41>(PhantomData<(C, St, W)>);
42
43#[cfg(all(feature = "gxhash", not(target_arch = "wasm32")))]
44impl<const N: usize, C: Coefficient, St: Strategy, W: PauliWordTrait> Config
45    for ByteGxHash<N, C, St, W>
46{
47    type Storage = [u8; N];
48    type Coeff = C;
49    type BuildHasher = gxhash::GxBuildHasher;
50    type PauliWordType = W;
51    type Map = indexmap::IndexMap<W, Self::Coeff, Self::BuildHasher>;
52    type Strategy = St;
53}

55/// [`ByteFxHash`] specialised to `f64` coefficients.
56pub type ByteFxHashF64<
57    const N: usize,
58    St = NoStrategy,
59    Wd = PauliWord<[u8; N], fxhash::FxBuildHasher>,

63pub type ByteGxHashF64<
64    const N: usize,
65    St = NoStrategy,
66    Wd = PauliWord<[u8; N], gxhash::GxBuildHasher>,
67> = ByteGxHash<N, f64, St, Wd>;
