4use pyo3::exceptions::{PyIOError, PyValueError};
5use pyo3::prelude::*;
6use std::ops::Deref;
7
8use ppvm_stim::{ExtendedProgram, parse_extended, validate};
9
10/// Python-facing wrapper around a validated extended Stim program.
11#[pyclass(name = "StimProgram", module = "ppvm_python_native")]
12pub struct PyStimProgram(pub ExtendedProgram);
13
14#[pymethods]
15impl PyStimProgram {
16    /// Parse and validate a Stim circuit string.
17    #[staticmethod]
18    pub fn parse(src: &str) -> PyResult<Self> {
19        let program = parse_extended(src).map_err(stim_to_pyerr)?;
20        validate(&program).map_err(stim_to_pyerr_exec)?;
21        Ok(Self(program))
22    }
23
24    /// Read a `.stim` file and parse it.
25    #[staticmethod]
26    pub fn from_file(path: &str) -> PyResult<Self> {
27        let src = std::fs::read_to_string(path)
28            .map_err(|e| PyIOError::new_err(format!("failed to read {path}: {e}")))?;
29        Self::parse(&src)
30    }
31
32    fn __repr__(&self) -> String {
33        format!(
34            "<StimProgram instructions={} measurements={}>",
35            self.0.instructions.len(),
36            self.0.measurement_count()
37        )
38    }
39}
40
41impl Deref for PyStimProgram {
42    type Target = ExtendedProgram;
43    fn deref(&self) -> &ExtendedProgram {
44        &self.0
45    }
46}
47
48fn stim_to_pyerr(e: ppvm_stim::ExtendedParseError) -> PyErr {
49    PyValueError::new_err(format!("{e}"))
50}
51
52fn stim_to_pyerr_exec(e: ppvm_stim::ExecError) -> PyErr {
53    PyValueError::new_err(format!("{e}"))
54}
