4use std::hash::BuildHasher;
5
6use crate::traits::{ACMap, Coefficient, PauliStorage, PauliWordTrait, Strategy};
7
8/// Compile-time configuration bundle for a [`PauliSum`](crate::sum::PauliSum).
9///
10/// `Config` is a zero-cost generic dispatch mechanism: it pins down the
11/// storage backing (`Storage`), coefficient type (`Coeff`), truncation
12/// strategy (`Strategy`), hasher (`BuildHasher`), Pauli-word representation
13/// (`PauliWordType`), and concrete map (`Map`) used by a single
14/// `PauliSum` instantiation. Pre-built bundles live in the submodules
15/// ([`fxhash`], [`indexmap`], …); user code may define its own.
16pub trait Config: Clone {
17    /// Backing storage for an individual Pauli word — typically a `[u8; N]`
18    /// where `N` is the number of bytes packed.
19    type Storage: PauliStorage;
20    /// Numeric coefficient type (e.g. `f64`, `Complex<f64>`).
21    type Coeff: Coefficient;
22    /// Truncation strategy applied when
23    /// [`PauliSum::truncate`](crate::sum::PauliSum::truncate) runs.
24    type Strategy: Strategy;
25    /// Hasher used by the underlying map.
26    type BuildHasher: BuildHasher + Clone + Default;
27    /// Concrete [`PauliWordTrait`] implementation used as map keys.
28    type PauliWordType: PauliWordTrait;
29    /// Concrete map type satisfying [`ACMap`] over the chosen
30    /// storage / coefficient / hasher / word combination.
31    type Map: ACMap<Self::Storage, Self::Coeff, Self::BuildHasher, Self::PauliWordType>;
32}
33
34/// Pre-built configs using rustc-hash's `FxHasher` and `f64` coefficients.
35pub mod fx64hash;
36/// Pre-built configs using `FxHasher`.
37pub mod fxhash;
38
39/// Pre-built configs backed by `dashmap::DashMap` for concurrent access.
40#[cfg(feature = "dashmap")]
41pub mod dashmap;
42
43/// Pre-built configs backed by `indexmap::IndexMap`, preserving
44/// insertion order — useful for deterministic iteration and snapshot
45/// testing.
46#[cfg(feature = "indexmap")]
47pub mod indexmap;
48
49/// Pre-built configs using `gxhash` — fast on platforms with AES
50/// hardware acceleration. Requires the `gxhash` feature.
51#[cfg(all(
52    feature = "gxhash",
53    any(target_arch = "x86", target_arch = "x86_64", target_arch = "aarch64")
54))]
55pub mod gxhash;
