26impl<'a, S, V, Hasher, W> ACMapAddAssign<S, V, Hasher, W> for DashMap<W, V, Hasher>
27where
28    S: PauliStorage + 'a,
29    V: Coefficient + Sync + Send + 'a,
30    Hasher: Default + Clone + BuildHasher + Sync + Send + 'a,
31    W: PauliWordTrait + Sync + Send + 'a,
32{
33    fn add_assign(&mut self, key: W, value: V) {
34        self.entry(key)
35            .and_modify(|v| *v += value.clone())
36            .or_insert(value);
37    }
38
39    fn map_add_assign<F>(&self, dest: &mut Self, f: F)
40    where
41        F: Fn(&W, &V) -> (W, V) + Sync + Send,
42    {
43        self.par_iter().for_each(|entry| {
44            let (new_k, new_v) = f(entry.key(), entry.value());
45            dest.entry(new_k)
46                .and_modify(|v| *v += new_v.clone())
47                .or_insert(new_v);
48        });
49    }
50}

93impl<'a, C, Hasher, W> Trace<'a, W> for DashMap<W, C, Hasher>
94where
95    C: Coefficient + Send + Sync + 'a,
96    Hasher: Clone + BuildHasher + Default + Send + Sync + 'a,
97    W: std::hash::Hash + std::cmp::Eq + Sync + Send + 'a,
98    for<'b> W: Trace<'b, W, Output = bool>,
99{
100    type Output = C;
101    fn trace(&'a self, value: &'a W) -> Self::Output {
102        self.par_iter()
103            .filter(|entry| value.trace(entry.key()))
104            .map(|entry| entry.value().clone())
105            .sum()
106    }
107}

125impl<'a, C, H, W> ACMapConsume for DashMap<W, C, H>
126where
127    C: Coefficient + Send + Sync + 'a,
128    H: Clone + BuildHasher + 'a + Send + Sync,
129    W: std::hash::Hash + std::cmp::Eq + Clone + Sync + Send + 'a,
130{
131    fn consume(&mut self, dest: &mut Self) {
132        dest.par_iter().for_each(|entry| {
133            self.entry(entry.key().clone())
134                .and_modify(|v| *v += entry.value().clone())
135                .or_insert_with(|| entry.value().clone());
136        });
137

142        // );
143        dest.clear();
144    }

147impl<'a, S, C, H, W> ACMapInsert<S, C, H, W> for DashMap<W, C, H>
148where
149    S: PauliStorage + 'a,
150    C: Coefficient + Send + Sync + 'a,
151    H: Default + Clone + BuildHasher + 'a + Send + Sync,
152    W: PauliWordTrait + Send + Sync + 'a,
153{
154    fn map_insert<F>(&mut self, dest: &mut Self, f: F)
155    where
156        F: Fn(&W, &mut C) -> Option<(W, C)> + Sync + Send,
157    {
158        self.par_iter_mut().for_each(|mut entry| {
159            let (k, v) = entry.pair_mut();
160            if let Some((new_k, new_v)) = f(k, v) {
161                dest.entry(new_k)
162                    .and_modify(|v| *v += new_v.clone())
163                    .or_insert(new_v);
164            }
165        })
166    }
167
168    fn map_insert_multiple<F>(&mut self, dest: &mut Self, f: F)
169    where
170        F: Fn(&W, &mut C) -> Option<Vec<(W, C)>> + Sync + Send,
171    {
172        self.par_iter_mut().for_each(|mut entry| {
173            let (k, v) = entry.pair_mut();
174            if let Some(new_entries) = f(k, v) {
175                for (new_k, new_v) in new_entries {
176                    dest.insert(new_k, new_v);
177                }
178            }
179        })
180    }
181}
