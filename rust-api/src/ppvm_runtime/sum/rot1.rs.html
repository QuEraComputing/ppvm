4pub(crate) fn rotate_1_map_insert_closure<T: Config>(
5    k: &T::PauliWordType,
6    v: &mut T::Coeff,
7    axis: Pauli,
8    addr0: usize,
9    sin: &T::Coeff,
10    cos: &T::Coeff,
11) -> Option<(T::PauliWordType, T::Coeff)> {
12    let p_g = k.get(addr0);
13    if axis == Pauli::L {
14        panic!("Rotation axis cannot be L");
15    }
16    if p_g == Pauli::L {
17        // Rotation does nothing on lost qubits
18        return None;
19    }
20    let (eps, p_q) = levi_civita(p_g as u8, axis as u8);
21    if eps == 0 {
22        return None;
23    } else {
24        let mut coeff = v.clone();
25        *v *= cos.clone();
26        let mut new_word = k.clone();
27        new_word.set_xbit(addr0, p_q & 0b01 != 0);
28        new_word.set_zbit(addr0, p_q & 0b10 != 0);
29        new_word.rehash();
30
31        coeff *= sin.mul_sign(eps);
32        return Some((new_word, coeff));
33    }
34}

36impl<T, S, H> RotationOne<T> for PauliSum<T>
37where
38    S: PauliStorage,
39    H: std::hash::BuildHasher + Clone + Default,
40    T: Config<Storage = S, BuildHasher = H>,
41    T::Coeff: std::ops::MulAssign,
42    T::Map: ACMapInsert<S, T::Coeff, H, T::PauliWordType> + ACMapConsume,
43{
44    fn rotate_1(&mut self, axis: Pauli, addr0: usize, theta: <T as Config>::Coeff) {
45        let (sin, cos) = theta.sin_cos();
46        self.map_insert(|k, v| rotate_1_map_insert_closure::<T>(k, v, axis, addr0, &sin, &cos));
47    }
48}

50/// 2-bit Pauli code: 00 I, 01 X, 10 Z, 11 Y
51/// Returns \(ε, k\) so that  –i \[P_i, P_j\]/2 = ε · P_k.
52/// For every commuting pair it yields (0, 0).
53#[inline]
54pub fn levi_civita(i: u8, j: u8) -> (i8, u8) {
55    // --------------------------------------------------- third Pauli by XOR
56    let k = i ^ j; // 0 when i == j
57
58    // ----------- commute ⇔ i==0  OR  j==0  OR  k==0  (no false positives)
59    let commute = ((i == 0) | (j == 0) | (k == 0)) as u8; // 1 = commute
60
61    // ------------------------------------------------------ sign ε_{ijk}
62    #[inline]
63    fn rank(p: u8) -> u8 {
64        let b1 = p >> 1; // MSB
65        (b1 << 1).wrapping_sub(b1 & (p & 1)) // 0,1,2 for X,Y,Z
66    }
67
68    let ri = rank(i);
69    let rj = rank(j);
70
71    // diff = (rj - ri) mod 3   without an actual modulus
72    let mut diff = rj.wrapping_sub(ri).wrapping_add(3); // 0…5
73    diff -= 3 & (0u8.wrapping_sub(diff >> 2)); // if ≥3 subtract 3
74
75    // +1 when diff == 1,  –1 when diff == 2
76    let eps_raw = 1i8 - 2 * ((diff >> 1) as i8);
77
78    // --------------------------------------------------- zero when commute
79    let eps = eps_raw * (1 - commute as i8); // 0 if commute
80    let k = k * (1 - commute); // 0 if commute
81
82    (eps, k)
83}
