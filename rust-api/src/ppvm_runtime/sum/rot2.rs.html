5const PAULIS: [Pauli; 4] = [Pauli::I, Pauli::X, Pauli::Z, Pauli::Y];
6
7impl<T, S, H> RotationTwo<T> for PauliSum<T>
8where
9    S: PauliStorage,
10    H: std::hash::BuildHasher + Clone + Default,
11    T: Config<Storage = S, BuildHasher = H>,
12    T::Coeff: std::ops::MulAssign,
13    T::Map: ACMapInsert<T::Storage, T::Coeff, T::BuildHasher, T::PauliWordType> + ACMapConsume,
14{
15    fn rotate_2(
16        &mut self,
17        axis_a_x: u8,
18        axis_a_z: u8,
19        axis_b_x: u8,
20        axis_b_z: u8,
21        a: usize,
22        b: usize,
23        theta: T::Coeff,
24    ) {
25        if axis_a_x > 3 || axis_a_z > 3 || axis_b_x > 3 || axis_b_z > 3 {
26            panic!("Rotation axis cannot be L");
27        }
28        let (sin, cos) = theta.sin_cos();
29        let axis_a = PAULIS[(axis_a_z << 1 | axis_a_x) as usize];
30        let axis_b = PAULIS[(axis_b_z << 1 | axis_b_x) as usize];
31        self.map_insert(|k, v| {
32            // NOTE: case of both qubits being lost is handled by single-qubit rotation logic
33            if k.get_lbit(a) {
34                // fall back to single-qubit rotation on qubit b
35                return rotate_1_map_insert_closure::<T>(k, v, axis_b, b, &sin, &cos);
36            }
37            if k.get_lbit(b) {
38                // fall back to single-qubit rotation on qubit a
39                return rotate_1_map_insert_closure::<T>(k, v, axis_a, a, &sin, &cos);
40            }
41            let (eps, x_a, z_a, x_b, z_b) = comm_2(
42                axis_a_x,
43                axis_a_z,
44                axis_b_x,
45                axis_b_z,
46                k.get_xbit(a) as u8,
47                k.get_zbit(a) as u8,
48                k.get_xbit(b) as u8,
49                k.get_zbit(b) as u8,
50            );
51
52            if eps == 0 {
53                return None;
54            } else {
55                let mut coeff = v.clone();
56                *v *= cos.clone();
57
58                let mut new_word = k.clone();
59                new_word.set_xbit(a, x_a == 1);
60                new_word.set_xbit(b, x_b == 1);
61                new_word.set_zbit(a, z_a == 1);
62                new_word.set_zbit(b, z_b == 1);
63                new_word.rehash();
64
65                coeff *= sin.mul_sign(-eps);
66                return Some((new_word, coeff));
67            }
68        });
69    }
70}

72// R_{G}[P] = cos(theta) P - sin(theta) [G, P]/2i
73// Encoding for one qubit
74//   x = 0/1  (X flag)   z = 0/1  (Z flag)
75//   (x,z) = 00 → I , 10 → X , 01 → Z , 11 → Y   (little-endian)
76//
77// Two–qubit word Q = q0 ⊗ q1  is given as four bits
78//   (x_a , z_a)  … qubit-0 of Q
79//   (x_b , z_b)  … qubit-1 of Q
80// P = p0 ⊗ p1   likewise
81//   (x_c , z_c)  … qubit-0 of P
82//   (x_d , z_d)  … qubit-1 of P
83//
84// The function returns
85//   (coeff , xx_out , zz_out)
86// where  coeff ∈ {-1,0,+1}  and  (xx_out, zz_out) are the usual packed masks
87// (bit-0 = qubit-0, bit-1 = qubit-1).
88/// Branch-free commutator  [ Q , P ] / (2 i )
89///
90///   Inputs are eight *bool-as-u8* flags (0 or 1):
91///       (x_a,z_a)  Q qubit-0       (x_b,z_b)  Q qubit-1
92///       (x_c,z_c)  P qubit-0       (x_d,z_d)  P qubit-1
93///
94///   Returns (coeff , x_out0 , z_out0 , x_out1 , z_out1)
95///           coeff ∈ { -1 , 0 , +1 }
96#[inline(always)]
97pub fn comm_2(
98    x_a: u8,
99    z_a: u8, // Q₀
100    x_b: u8,
101    z_b: u8, // Q₁
102    x_c: u8,
103    z_c: u8, // P₀
104    x_d: u8,
105    z_d: u8, // P₁
106) -> (i8, u8, u8, u8, u8) {
107    // ── 1.  per-qubit anticommutation bits  a₀ , a₁  ───────────────────────
108    let a0 = (x_a & z_c) ^ (z_a & x_c); // qubit-0 anticommutes?
109    let a1 = (x_b & z_d) ^ (z_b & x_d); // qubit-1 anticommutes?
110
111    // overall commutator present when exactly one qubit anticommutes
112    let present = a0 ^ a1; // 0 ↔ commute, 1 ↔ anticommute
113
114    // ── 2.  sign of the coefficient (+1 / −1)  ─────────────────────────────
115    // 16-entry bit-mask: 1 → “negative orientation” ( −1 ), 0 → “positive” (+1)
116    const SIGN_NEG: u16 = 0x2840; // pre-computed once
117
118    let idx0 = (z_a << 3) | (x_a << 2) | (z_c << 1) | x_c; // qubit-0 pair index 0-15
119    let idx1 = (z_b << 3) | (x_b << 2) | (z_d << 1) | x_d; // qubit-1 pair index
120
121    let neg0 = (((SIGN_NEG >> idx0) as u8) & 1) & a0; // only meaningful if a₀ = 1
122    let neg1 = (((SIGN_NEG >> idx1) as u8) & 1) & a1;
123
124    // coeff = ( +1 or −1 ) from the unique acted-on qubit; 0 if they commute
125    let coeff = (((1 - ((neg0 as i8) << 1)) * (a0 as i8))
126        + ((1 - ((neg1 as i8) << 1)) * (a1 as i8)))
127        * (present as i8); // ensures 0 when commuting
128
129    // ── 3.  output flags  (product of Q and P)  masked to zero if commuting ─
130    let x_out0 = (x_a ^ x_c) & present;
131    let z_out0 = (z_a ^ z_c) & present;
132    let x_out1 = (x_b ^ x_d) & present;
133    let z_out1 = (z_b ^ z_d) & present;
134
135    (coeff, x_out0, z_out0, x_out1, z_out1)
136}
