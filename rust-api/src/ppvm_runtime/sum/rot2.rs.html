4use super::rot1::rotate_1_map_insert_closure;
5use crate::traits::*;
6use crate::{char::Pauli, config::Config, sum::PauliSum};

8const PAULIS: [Pauli; 4] = [Pauli::I, Pauli::X, Pauli::Z, Pauli::Y];
9
10impl<T: Config> RotationTwo<T> for PauliSum<T>
11where
12    T::Coeff: std::ops::MulAssign,
13    T::Map: ACMapInsert<T::Storage, T::Coeff, T::BuildHasher, T::PauliWordType> + ACMapConsume,
14{
15    fn rotate_2(&mut self, axis_a: [u8; 2], axis_b: [u8; 2], a: usize, b: usize, theta: T::Coeff) {
16        let [axis_a_x, axis_a_z] = axis_a;
17        let [axis_b_x, axis_b_z] = axis_b;
18        if axis_a_x > 3 || axis_a_z > 3 || axis_b_x > 3 || axis_b_z > 3 {
19            panic!("Rotation axis cannot be L");
20        }
21        let (sin, cos) = theta.sin_cos();
22        let pauli_a = PAULIS[(axis_a_z << 1 | axis_a_x) as usize];
23        let pauli_b = PAULIS[(axis_b_z << 1 | axis_b_x) as usize];
24        self.map_insert(|k, v| {
25            // NOTE: case of both qubits being lost is handled by single-qubit rotation logic
26            if k.get_lbit(a) {
27                // fall back to single-qubit rotation on qubit b
28                return rotate_1_map_insert_closure::<T>(k, v, pauli_b, b, &sin, &cos);
29            }
30            if k.get_lbit(b) {
31                // fall back to single-qubit rotation on qubit a
32                return rotate_1_map_insert_closure::<T>(k, v, pauli_a, a, &sin, &cos);
33            }
34            let (eps, x_a, z_a, x_b, z_b) = comm_2(
35                axis_a,
36                axis_b,
37                [k.get_xbit(a) as u8, k.get_zbit(a) as u8],
38                [k.get_xbit(b) as u8, k.get_zbit(b) as u8],
39            );
40
41            if eps == 0 {
42                None
43            } else {
44                let mut coeff = v.clone();
45                *v *= cos.clone();
46
47                let mut new_word = k.clone();
48                new_word.set_xbit(a, x_a == 1);
49                new_word.set_xbit(b, x_b == 1);
50                new_word.set_zbit(a, z_a == 1);
51                new_word.set_zbit(b, z_b == 1);
52                new_word.rehash();
53
54                coeff *= sin.mul_sign(-eps);
55                Some((new_word, coeff))
56            }
57        });
58    }
59}

61// R_{G}[P] = cos(theta) P - sin(theta) [G, P]/2i
62// Encoding for one qubit
63//   x = 0/1  (X flag)   z = 0/1  (Z flag)
64//   (x,z) = 00 → I , 10 → X , 01 → Z , 11 → Y   (little-endian)
65//
66// Two–qubit word Q = q0 ⊗ q1  is given as four bits
67//   (x_a , z_a)  … qubit-0 of Q
68//   (x_b , z_b)  … qubit-1 of Q
69// P = p0 ⊗ p1   likewise
70//   (x_c , z_c)  … qubit-0 of P
71//   (x_d , z_d)  … qubit-1 of P
72//
73// The function returns
74//   (coeff , xx_out , zz_out)
75// where  coeff ∈ {-1,0,+1}  and  (xx_out, zz_out) are the usual packed masks
76// (bit-0 = qubit-0, bit-1 = qubit-1).
77/// Branch-free commutator  [ Q , P ] / (2 i )
78///
79///   Each qubit is encoded as `[x, z]` bits (0 or 1):
80///       `q0 = [x_a, z_a]`  Q qubit-0       `q1 = [x_b, z_b]`  Q qubit-1
81///       `p0 = [x_c, z_c]`  P qubit-0       `p1 = [x_d, z_d]`  P qubit-1
82///
83///   Returns (coeff , x_out0 , z_out0 , x_out1 , z_out1)
84///           coeff ∈ { -1 , 0 , +1 }
85#[inline(always)]
86pub fn comm_2(q0: [u8; 2], q1: [u8; 2], p0: [u8; 2], p1: [u8; 2]) -> (i8, u8, u8, u8, u8) {
87    let [x_a, z_a] = q0;
88    let [x_b, z_b] = q1;
89    let [x_c, z_c] = p0;
90    let [x_d, z_d] = p1;
91    // ── 1.  per-qubit anticommutation bits  a₀ , a₁  ───────────────────────
92    let a0 = (x_a & z_c) ^ (z_a & x_c); // qubit-0 anticommutes?
93    let a1 = (x_b & z_d) ^ (z_b & x_d); // qubit-1 anticommutes?
94
95    // overall commutator present when exactly one qubit anticommutes
96    let present = a0 ^ a1; // 0 ↔ commute, 1 ↔ anticommute
97
98    // ── 2.  sign of the coefficient (+1 / −1)  ─────────────────────────────
99    // 16-entry bit-mask: 1 → “negative orientation” ( −1 ), 0 → “positive” (+1)
100    const SIGN_NEG: u16 = 0x2840; // pre-computed once
101
102    let idx0 = (z_a << 3) | (x_a << 2) | (z_c << 1) | x_c; // qubit-0 pair index 0-15
103    let idx1 = (z_b << 3) | (x_b << 2) | (z_d << 1) | x_d; // qubit-1 pair index
104
105    let neg0 = (((SIGN_NEG >> idx0) as u8) & 1) & a0; // only meaningful if a₀ = 1
106    let neg1 = (((SIGN_NEG >> idx1) as u8) & 1) & a1;
107
108    // coeff = ( +1 or −1 ) from the unique acted-on qubit; 0 if they commute
109    let coeff = (((1 - ((neg0 as i8) << 1)) * (a0 as i8))
110        + ((1 - ((neg1 as i8) << 1)) * (a1 as i8)))
111        * (present as i8); // ensures 0 when commuting
112
113    // ── 3.  output flags  (product of Q and P)  masked to zero if commuting ─
114    let x_out0 = (x_a ^ x_c) & present;
115    let z_out0 = (z_a ^ z_c) & present;
116    let x_out1 = (x_b ^ x_d) & present;
117    let z_out1 = (z_b ^ z_d) & present;
118
119    (coeff, x_out0, z_out0, x_out1, z_out1)
120}
