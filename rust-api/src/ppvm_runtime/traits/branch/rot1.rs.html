4use crate::char::Pauli;
5use crate::config::Config;
6
7/// Single-qubit Pauli rotations `exp(-i θ/2 · P)`.
8pub trait RotationOne<T: Config> {
9    /// Rotate about `axis` (one of `X`, `Y`, `Z`) by angle `theta`.
10    fn rotate_1(&mut self, axis: Pauli, addr0: usize, theta: T::Coeff);
11    /// `RX(θ)` on qubit `addr0`.
12    fn rx(&mut self, addr0: usize, theta: impl Into<T::Coeff>) {
13        self.rotate_1(Pauli::X, addr0, theta.into())
14    }
15    /// `RY(θ)` on qubit `addr0`.
16    fn ry(&mut self, addr0: usize, theta: impl Into<T::Coeff>) {
17        self.rotate_1(Pauli::Y, addr0, theta.into())
18    }
19    /// `RZ(θ)` on qubit `addr0`.
20    fn rz(&mut self, addr0: usize, theta: impl Into<T::Coeff>) {
21        self.rotate_1(Pauli::Z, addr0, theta.into())
22    }
23}
