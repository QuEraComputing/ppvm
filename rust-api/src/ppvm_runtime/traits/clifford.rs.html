6/// The minimal Clifford gate set: the single-qubit Paulis (`X`, `Y`, `Z`),
7/// Hadamard (`H`), phase gate (`S`), and the two entangling Cliffords
8/// `CNOT` and `CZ`.
9///
10/// Implemented by `PauliSum`, by every tableau type, and — via the
11/// blanket impl in this module — by every [`PauliWordTrait`]
12/// implementor.
13///
14/// # Examples
15///
16/// Build the GHZ-preparation circuit on a `PauliSum` (Heisenberg picture,
17/// so gates are applied in reverse):
18///
19/// ```
20/// use ppvm_runtime::prelude::*;
21///
22/// let mut state: PauliSum<config::indexmap::ByteFxHashF64<1>> =
23///     PauliSum::builder().n_qubits(2).build();
24/// state += ("ZZ", 1.0);
25/// state.cnot(0, 1);
26/// state.h(0);
27/// assert_eq!(state.len(), 1);
28/// ```
29pub trait Clifford {
30    /// Apply Pauli `X` to qubit `index`.
31    fn x(&mut self, index: usize);
32    /// Apply Pauli `Y` to qubit `index`.
33    fn y(&mut self, index: usize);
34    /// Apply Pauli `Z` to qubit `index`.
35    fn z(&mut self, index: usize);
36    /// Apply Hadamard `H` to qubit `index`.
37    fn h(&mut self, index: usize);
38    /// Apply phase gate `S` to qubit `index`.
39    fn s(&mut self, index: usize);
40    /// Apply `CNOT(control, target)`.
41    fn cnot(&mut self, control: usize, target: usize);
42    /// Apply `CZ(control, target)`.
43    fn cz(&mut self, control: usize, target: usize);
44}
45
46/// Additional Clifford gates beyond the minimal set: `S†`, `√X`, `√X†`,
47/// `√Y`, `√Y†`, and `CY`.
48pub trait CliffordExtensions: Clifford {
49    /// Apply `S†` (the adjoint of `S`) to qubit `addr0`.
50    fn s_adj(&mut self, addr0: usize);
51    /// Apply `√X` to qubit `addr0`.
52    fn sqrt_x(&mut self, addr0: usize);
53    /// Apply `(√X)†` to qubit `addr0`.
54    fn sqrt_x_adj(&mut self, addr0: usize);
55    /// Apply `√Y` to qubit `addr0`.
56    fn sqrt_y(&mut self, addr0: usize);
57    /// Apply `(√Y)†` to qubit `addr0`.
58    fn sqrt_y_adj(&mut self, addr0: usize);
59
60    /// Apply `CY(addr0, addr1)`.
61    fn cy(&mut self, addr0: usize, addr1: usize);
62}

64// === Blanket Clifford impl for PauliWordTrait ===
65//
66// Any type implementing [`PauliWordTrait`] automatically gets word-level
67// Clifford gate behavior. X/Y/Z are no-ops at the word level (they only
68// affect phase, tracked separately in `PhasedPauliWord`). H, S, CNOT, CZ
69// transform the bit representation. All gates honor loss bits via
70// `get_lbit`, which returns `false` for non-lossy types.
71//
72// New PauliWordTrait implementors get this behavior for free.
73//
74// Breaking change for downstream crates: this blanket impl conflicts with
75// any external `impl Clifford for MyWord where MyWord: PauliWordTrait`.
76// Downstreams that need custom Clifford semantics must not implement
77// `PauliWordTrait` on that type — see `PauliWordTrait`'s docs and the
78// `PhasedPauliWord` impl in `crate::phase::clifford` for the pattern.
79
80impl<T: PauliWordTrait> Clifford for T {
81    #[inline]
82    fn x(&mut self, _index: usize) {
83        // X * I * X = I    00 -> 00, 0
84        // X * X * X = X    10 -> 10, 0
85        // X * Z * X = -Z   01 -> 01, 1
86        // X * Y * X = -Y   11 -> 11, 1
87        // word-level no-op: phase tracked at PhasedPauliWord level
88    }
89
90    #[inline]
91    fn y(&mut self, _index: usize) {
92        // word-level no-op
93    }
94
95    #[inline]
96    fn z(&mut self, _index: usize) {
97        // word-level no-op
98    }
99
100    #[inline]
101    fn h(&mut self, index: usize) {
102        // H * I * H = I, H * X * H = Z, H * Z * H = X, H * Y * H = -Y
103        if self.get_lbit(index) {
104            return;
105        }
106        let ix = self.get_xbit(index);
107        let iz = self.get_zbit(index);
108        self.set_xbit(index, iz);
109        self.set_zbit(index, ix);
110        self.rehash();
111    }
112
113    #[inline]
114    fn s(&mut self, index: usize) {
115        // S * I * S = I, S * X * S = Y, S * Z * S = Z, S * Y * S = -X
116        if self.get_lbit(index) {
117            return;
118        }
119        let z = self.get_xbit(index) ^ self.get_zbit(index);
120        self.set_zbit(index, z);
121        self.rehash();
122    }
123
124    #[inline]
125    fn cnot(&mut self, control: usize, target: usize) {
126        if self.get_lbit(control) || self.get_lbit(target) {
127            return;
128        }
129        let control_z = self.get_zbit(target) ^ self.get_zbit(control);
130        let target_x = self.get_xbit(control) ^ self.get_xbit(target);
131        self.set_zbit(control, control_z);
132        self.set_xbit(target, target_x);
133        self.rehash();
134    }
135
136    #[inline]
137    fn cz(&mut self, control: usize, target: usize) {
138        if self.get_lbit(control) || self.get_lbit(target) {
139            return;
140        }
141        // flip the control z if target x is 1
142        let control_z = self.get_zbit(control) ^ self.get_xbit(target);
143        self.set_zbit(control, control_z);
144        // flip the target z if control x is 1
145        let target_z = self.get_zbit(target) ^ self.get_xbit(control);
146        self.set_zbit(target, target_z);
147        self.rehash();
148    }
149}

151/// Batched Clifford gates: apply the same gate to many qubits in one call.
152///
153/// Default implementations loop over the corresponding single-qubit
154/// (or single-pair) method on [`Clifford`]. Types like the stabilizer
155/// `Tableau` override these methods with a fused inner-loop or bitmask
156/// implementation. Types that don't need specialization can implement
157/// this trait with an empty `impl` to use the defaults.
158pub trait CliffordBatch: Clifford {
159    /// Apply Pauli `X` to every qubit in `indices`.
160    fn x_batch(&mut self, indices: &[usize]) {
161        for &q in indices {
162            self.x(q);
163        }
164    }
165    /// Apply Pauli `Y` to every qubit in `indices`.
166    fn y_batch(&mut self, indices: &[usize]) {
167        for &q in indices {
168            self.y(q);
169        }
170    }
171    /// Apply Pauli `Z` to every qubit in `indices`.
172    fn z_batch(&mut self, indices: &[usize]) {
173        for &q in indices {
174            self.z(q);
175        }
176    }
177    /// Apply Hadamard `H` to every qubit in `indices`.
178    fn h_batch(&mut self, indices: &[usize]) {
179        for &q in indices {
180            self.h(q);
181        }
182    }
183    /// Apply phase gate `S` to every qubit in `indices`.
184    fn s_batch(&mut self, indices: &[usize]) {
185        for &q in indices {
186            self.s(q);
187        }
188    }
189    /// Apply `CNOT` to every `(control, target)` pair.
190    fn cnot_batch(&mut self, pairs: &[(usize, usize)]) {
191        for &(c, t) in pairs {
192            self.cnot(c, t);
193        }
194    }
195    /// Apply `CZ` to every `(control, target)` pair.
196    fn cz_batch(&mut self, pairs: &[(usize, usize)]) {
197        for &(c, t) in pairs {
198            self.cz(c, t);
199        }
200    }
201}

243impl<T: PauliWordTrait> CliffordExtensions for T {
244    // |    Gate    |  X  |  Y  |  Z  |
245    // |:----------:|:---:|:---:|:---:|
246    // |     s      | -Y  |  X  |  Z  |
247    // |   s_adj    |  Y  | -X  |  Z  |
248    // |   sqrt_x   |  X  | -Z  |  Y  |
249    // | sqrt*x*adj |  X  |  Z  | -Y  |
250    // |   sqrt_y   |  Z  |  Y  | -X  |
251    // | sqrt*y*adj | -Z  |  Y  |  X  |
252
253    #[inline]
254    fn s_adj(&mut self, addr0: usize) {
255        // s_adj has the same bit mapping as s (only phases differ)
256        self.s(addr0);
257    }
258
259    #[inline]
260    fn sqrt_x(&mut self, addr0: usize) {
261        if self.get_lbit(addr0) {
262            return;
263        }
264        let x = self.get_xbit(addr0);
265        let z = self.get_zbit(addr0);
266        self.set_xbit(addr0, x ^ z);
267        self.rehash();
268    }
269
270    #[inline]
271    fn sqrt_x_adj(&mut self, addr0: usize) {
272        if self.get_lbit(addr0) {
273            return;
274        }
275        let x = self.get_xbit(addr0);
276        let z = self.get_zbit(addr0);
277        self.set_xbit(addr0, x ^ z);
278        self.rehash();
279    }
280
281    #[inline]
282    fn sqrt_y(&mut self, addr0: usize) {
283        if self.get_lbit(addr0) {
284            return;
285        }
286        let x = self.get_xbit(addr0);
287        let z = self.get_zbit(addr0);
288        self.set_xbit(addr0, z);
289        self.set_zbit(addr0, x);
290        self.rehash();
291    }
292
293    #[inline]
294    fn sqrt_y_adj(&mut self, addr0: usize) {
295        if self.get_lbit(addr0) {
296            return;
297        }
298        let x = self.get_xbit(addr0);
299        let z = self.get_zbit(addr0);
300        self.set_xbit(addr0, z);
301        self.set_zbit(addr0, x);
302        self.rehash();
303    }
304
305    // | CY  |  I  |  X  |  Y  |  Z  |
306    // |:---:|:---:|:---:|:---:|:---:|
307    // |  I  | II  | ZX  | IY  | ZZ  |
308    // |  X  | XY  | -YZ | XI  | YX  |
309    // |  Y  | YY  | XZ  | YI  | -XX |
310    // |  Z  | ZI  | IX  | ZY  | IZ  |
311    #[inline]
312    fn cy(&mut self, addr0: usize, addr1: usize) {
313        if self.get_lbit(addr0) || self.get_lbit(addr1) {
314            return;
315        }
316        let xc = self.get_xbit(addr0);
317        let zc = self.get_zbit(addr0);
318        let xt = self.get_xbit(addr1);
319        let zt = self.get_zbit(addr1);
320        self.set_zbit(addr0, zc ^ xt ^ zt);
321        self.set_xbit(addr1, xt ^ xc);
322        self.set_zbit(addr1, zt ^ xc);
323        self.rehash();
324    }
