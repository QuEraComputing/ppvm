35pub trait CorrelatedLossChannel<T: Config> {
36    /// Apply a correlated loss channel to qubits at `addr0` and `addr1`.
37    ///
38    /// The three probabilities are:
39    /// * `p[0]`: The probability of losing both qubits simultaneously when
40    ///   both of them are in the qubit subspace.
41    /// * `p[1]`: The probability of losing either one qubit when both of them are
42    ///   in the qubit subspace.
43    /// * `p[2]`: The probability of losing one qubit when the other one has already
44    ///   been lost prior to the channel.
45    fn correlated_loss_channel(&mut self, addr0: usize, addr1: usize, p: [T::Coeff; 3]);
46}
