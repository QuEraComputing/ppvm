8/// Single-qubit Pauli error channel — apply `X`, `Y`, or `Z` with the
9/// three given probabilities.
10pub trait PauliError<T: Config> {
11    /// Apply a Pauli-error channel to qubit `addr0` with the probability
12    /// triple `p = [p_x, p_y, p_z]`.
13    fn pauli_error(&mut self, addr0: usize, p: [T::Coeff; 3]);
14}
15
16/// Apply the same single-qubit Pauli error channel uniformly to every
17/// qubit in the system.
18pub trait PauliErrorAll<T: Config> {
19    /// Apply `Pauli error` with probabilities `p = [p_x, p_y, p_z]` to
20    /// every qubit.
21    fn pauli_error_all(&mut self, p: [T::Coeff; 3]);
22}
23
24/// Two-qubit Pauli error channel.
25pub trait TwoQubitPauliError<T: Config> {
26    /// Apply a two-qubit Pauli-error channel to `(addr0, addr1)`.
27    /// Probabilities are given in the order:
28    /// `{IX, IY, IZ, XI, XX, XY, XZ, YI, YX, YY, YZ, ZI, ZX, ZY, ZZ}`.
29    fn two_qubit_pauli_error(&mut self, addr0: usize, addr1: usize, p: [T::Coeff; 15]);
30}
31
32/// Single-qubit depolarizing channel.
33pub trait Depolarizing<T: Config> {
34    /// Depolarize qubit `addr0` with probability `p`.
35    fn depolarize(&mut self, addr0: usize, p: T::Coeff);
36}
37
38/// Two-qubit depolarizing channel.
39pub trait Depolarizing2<T: Config> {
40    /// Depolarize the pair `(addr0, addr1)` with probability `p`.
41    fn depolarize2(&mut self, addr0: usize, addr1: usize, p: T::Coeff);
42}
43
44/// Amplitude-damping channel (single qubit).
45pub trait AmplitudeDamping<T: Config> {
46    /// Apply amplitude damping with damping parameter `gamma`.
47    fn amplitude_damping(&mut self, addr0: usize, gamma: T::Coeff);
48}
49
50/// Single-qubit loss channel — with probability `p`, mark the qubit as
51/// lost (`Pauli::L`).
52pub trait LossChannel<T: Config> {
53    /// Apply a loss channel to qubit `addr0` with loss probability `p`.
54    fn loss_channel(&mut self, addr0: usize, p: T::Coeff);
55}
56
57/// Correlated two-qubit loss channel.
58pub trait CorrelatedLossChannel<T: Config> {
59    /// Apply a correlated loss channel to qubits at `addr0` and `addr1`.
60    ///
61    /// The three probabilities are:
62    /// * `p[0]`: The probability of losing both qubits simultaneously when
63    ///   both of them are in the qubit subspace.
64    /// * `p[1]`: The probability of losing either one qubit when both of them are
65    ///   in the qubit subspace.
66    /// * `p[2]`: The probability of losing one qubit when the other one has already
67    ///   been lost prior to the channel.
68    fn correlated_loss_channel(&mut self, addr0: usize, addr1: usize, p: [T::Coeff; 3]);
69}
70
71/// Reset the loss bit on a qubit — used to model a re-cooling /
72/// re-loading event that brings a previously-lost atom back.
73pub trait ResetLossChannel<T: Config> {
74    /// Clear the loss bit at `addr0`.
75    fn reset_loss_channel(&mut self, addr0: usize);
76}
