10pub trait PauliWordTrait:
11    Clone + Hash + Eq + PauliIter + From<String> + Clifford + Display
12{
13    fn new(nqubits: usize) -> Self;
14
15    fn n_qubits(&self) -> usize;
16
17    // getter methods
18    fn get_xbit(&self, index: usize) -> bool;
19    fn get_zbit(&self, index: usize) -> bool;
20    fn get_lbit(&self, index: usize) -> bool;
21
22    // setter methods
23    fn set_xbit(&mut self, index: usize, value: bool);
24    fn set_zbit(&mut self, index: usize, value: bool);
25
26    fn weight(&self) -> usize;
27
28    fn loss_weight(&self) -> usize;
29
30    fn rehash(&mut self);
31
32    fn get(&self, index: usize) -> Pauli;
33
34    fn get_multiple<const Q: usize>(&self, indices: [usize; Q]) -> Self {
35        let mut result = Self::new(Q);
36        for (i, &idx) in indices.iter().enumerate() {
37            result.set(i, self.get(idx));
38        }
39        result
40    }
41
42    fn set_multiple<const Q: usize, B: PauliStorage>(&mut self, indices: [usize; Q], values: &Self);
43
44    fn get_slice(&self, slice: std::ops::Range<usize>) -> Self;
45
46    fn is(&self, index: usize, pauli: Pauli) -> bool;
47
48    fn set(&mut self, index: usize, pauli: Pauli) -> &mut Self;
49
50    #[inline(always)]
51    fn set_new(&self, index: usize, pauli: Pauli) -> Self {
52        if index >= self.n_qubits() {
53            panic!("Index out of bounds");
54        }
55        let mut new = self.clone();
56        new.set(index, pauli);
57        new
58    }
59
60    #[inline(always)]
61    fn set_new_2(&self, index_0: usize, pauli_0: Pauli, index_1: usize, pauli_1: Pauli) -> Self {
62        let mut new = self.clone();
63        new.set(index_0, pauli_0);
64        new.set(index_1, pauli_1);
65        new
66    }
67
68    /// Check if this word anticommutes with a single-qubit Pauli at `addr0`,
69    /// where `pauli = (xbit, zbit)`.
70    #[inline(always)]
71    fn anticommutes_at(&self, addr0: usize, pauli: (bool, bool)) -> bool {
72        (self.get_xbit(addr0) & pauli.1) ^ (self.get_zbit(addr0) & pauli.0)
73    }
