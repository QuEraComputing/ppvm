4use crate::char::Pauli;
5use crate::traits::PauliStorage;
6use std::fmt::Display;
7use std::hash::Hash;

9/// Iterate over a Pauli word slot-by-slot.
10pub trait PauliIter {
11    /// Yield the [`Pauli`] symbol at each qubit position, in order.
12    fn iter(&self) -> impl Iterator<Item = Pauli>;
13}
14
15/// Word-level Pauli operations. Types implementing this trait automatically
16/// gain [`crate::traits::Clifford`] and [`crate::traits::CliffordExtensions`]
17/// behavior via blanket impls in the `crate::traits::clifford` module, using
18/// `get_lbit` / `get_xbit` / `set_xbit` / `set_zbit` / `rehash` to transform
19/// Pauli words.
20///
21/// # Word-level vs. state-level gate semantics
22///
23/// The blanket `Clifford` impl applies gates at the *bit* level of a single
24/// Pauli word. X / Y / Z are bit-level no-ops on a Pauli word — they affect
25/// phase, not the X/Z bits — so `word.x(i)`, `word.y(i)`, `word.z(i)` are
26/// deliberately silent. Phase is tracked separately by
27/// [`crate::phase::PhasedPauliWord`], which implements `Clifford` manually.
28///
29/// If you need a word representation whose gate behavior is *not* pure
30/// bit manipulation (phase tracking, fused multi-qubit updates, alternative
31/// loss semantics), do **not** implement `PauliWordTrait` on it — define a
32/// specialized `Clifford` impl instead, the way `PhasedPauliWord` does.
33/// Implementing both `PauliWordTrait` and a custom `impl Clifford for ...`
34/// for the same type will not compile: the blanket impl overlaps with your
35/// custom impl (coherence error), not silently shadows it.
36pub trait PauliWordTrait: Clone + Hash + Eq + PauliIter + From<String> + Display {
37    /// Construct an identity word over `nqubits` qubits.
38    fn new(nqubits: usize) -> Self;
39
40    /// Number of qubits.
41    fn n_qubits(&self) -> usize;
42
43    /// X bit at `index`.
44    fn get_xbit(&self, index: usize) -> bool;
45    /// Z bit at `index`.
46    fn get_zbit(&self, index: usize) -> bool;
47    /// Loss bit at `index` (always `false` for non-lossy implementations).
48    fn get_lbit(&self, index: usize) -> bool;
49
50    /// Set the X bit at `index`.
51    fn set_xbit(&mut self, index: usize, value: bool);
52    /// Set the Z bit at `index`.
53    fn set_zbit(&mut self, index: usize, value: bool);
54
55    /// Number of non-identity slots (counts `X`, `Y`, `Z`, and — for
56    /// lossy variants — `L`).
57    fn weight(&self) -> usize;
58
59    /// Number of slots marked as lost; always `0` for non-lossy variants.
60    fn loss_weight(&self) -> usize;
61
62    /// Recompute the cached hash. Call after batch mutations.
63    fn rehash(&mut self);
64
65    /// [`Pauli`] symbol at `index`.
66    fn get(&self, index: usize) -> Pauli;
67
68    /// Build a new word containing only the slots at `indices`.
69    fn get_multiple<const Q: usize>(&self, indices: [usize; Q]) -> Self {
70        let mut result = Self::new(Q);
71        for (i, &idx) in indices.iter().enumerate() {
72            result.set(i, self.get(idx));
73        }
74        result
75    }
76
77    /// Overwrite the slots at `indices` with the slots from `values`.
78    fn set_multiple<const Q: usize, B: PauliStorage>(&mut self, indices: [usize; Q], values: &Self);
79
80    /// Build a new word from the contiguous slice `slice` of this one.
81    fn get_slice(&self, slice: std::ops::Range<usize>) -> Self;
82
83    /// Quick check: is the slot at `index` equal to `pauli`?
84    fn is(&self, index: usize, pauli: Pauli) -> bool;
85
86    /// Set the slot at `index` to `pauli`, in place. Returns `&mut self`
87    /// for chaining.
88    fn set(&mut self, index: usize, pauli: Pauli) -> &mut Self;
89
90    /// Return a clone with the slot at `index` set to `pauli`.
91    #[inline(always)]
92    fn set_new(&self, index: usize, pauli: Pauli) -> Self {
93        if index >= self.n_qubits() {
94            panic!("Index out of bounds");
95        }
96        let mut new = self.clone();
97        new.set(index, pauli);
98        new
99    }
100
101    /// Return a clone with two slots overwritten.
102    #[inline(always)]
103    fn set_new_2(&self, index_0: usize, pauli_0: Pauli, index_1: usize, pauli_1: Pauli) -> Self {
104        let mut new = self.clone();
105        new.set(index_0, pauli_0);
106        new.set(index_1, pauli_1);
107        new
108    }
109
110    /// Check if this word anticommutes with a single-qubit Pauli at `addr0`,
111    /// where `pauli = (xbit, zbit)`.
112    #[inline(always)]
113    fn anticommutes_at(&self, addr0: usize, pauli: (bool, bool)) -> bool {
114        (self.get_xbit(addr0) & pauli.1) ^ (self.get_zbit(addr0) & pauli.0)
115    }
116}
