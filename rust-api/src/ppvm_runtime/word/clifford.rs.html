1use std::hash::BuildHasher;
2
3use super::data::PauliWord;
4use crate::traits::{Clifford, PauliStorage, PauliWordTrait};
5
6impl<A, H, const REHASH: bool> Clifford for PauliWord<A, H, REHASH>
7where
8    A: PauliStorage,
9    H: BuildHasher + Clone + Default,
10{
11    #[inline]
12    fn x(&mut self, _index: usize) {
13        // X * I * X = I    00 -> 00, 0
14        // X * X * X = X    10 -> 10, 0
15        // X * Z * X = -Z   01 -> 01, 1
16        // X * Y * X = -Y.  11 -> 11, 1
17    }
18
19    #[inline]
20    fn y(&mut self, _index: usize) {
21        // Y * I * Y = I    00 -> 00, 0
22        // Y * X * Y = -X   10 -> 10, 1
23        // Y * Z * Y = -Z   01 -> 01, 1
24        // Y * Y * Y = Y    11 -> 11, 0
25    }
26
27    #[inline]
28    fn z(&mut self, _index: usize) {
29        // Z * I * Z = I    00 -> 00, 0
30        // Z * X * Z = -X   10 -> 10, 1
31        // Z * Z * Z = Z    01 -> 01, 0
32        // Z * Y * Z = -Y   11 -> 11, 1
33    }
34
35    #[inline]
36    fn h(&mut self, index: usize) {
37        // H * I * H = I    00 -> 00, 0
38        // H * X * H = Z    10 -> 01, 0
39        // H * Z * H = X    01 -> 10, 0
40        // H * Y * H = -Y   11 -> 11, 1
41        let index_x = self.xbits[index];
42        let index_z = self.zbits[index];
43        self.xbits.set(index, index_z);
44        self.zbits.set(index, index_x);
45        self.rehash();
46    }
47
48    #[inline]
49    fn s(&mut self, index: usize) {
50        // S * I * S = I    00 -> 00, 0
51        // S * X * S = Y    10 -> 11, 0
52        // S * Z * S = Z    01 -> 01, 0
53        // S * Y * S = -X   11 -> 10, 1
54        let z = self.xbits[index] ^ self.zbits[index];
55        self.zbits.set(index, z);
56        self.rehash();
57    }
58
59    #[inline]
60    fn cnot(&mut self, control: usize, target: usize) {
61        //                          xx zz    xx zz  phase
62        // CNOT * II * CNOT == II,  00 00 -> 00 00, 0
63        // CNOT * IX * CNOT == IX,  01 00 -> 01 00, 0
64        // CNOT * IZ * CNOT == ZZ,  00 01 -> 00 11, 0
65        // CNOT * IY * CNOT == ZY,  01 01 -> 01 11, 0
66
67        // CNOT * XI * CNOT == XX,  10 00 -> 11 00, 0
68        // CNOT * XX * CNOT == XI,  11 00 -> 10 00, 0
69        // CNOT * XY * CNOT == YZ,  11 01 -> 10 11, 0
70        // CNOT * XZ * CNOT == -YY, 10 01 -> 11 11, 1
71
72        // CNOT * ZI * CNOT == ZI,  00 10 -> 00 10, 0
73        // CNOT * ZX * CNOT == ZX,  01 10 -> 01 10, 0
74        // CNOT * ZY * CNOT == IY,  01 11 -> 01 01, 0
75        // CNOT * ZZ * CNOT == IZ,  00 11 -> 00 01, 0
76
77        // CNOT * YI * CNOT == YX,  10 10 -> 11 10, 0
78        // CNOT * YX * CNOT == YI,  11 10 -> 10 10, 0
79        // CNOT * YY * CNOT == -XZ, 11 11 -> 10 01, 1
80        // CNOT * YZ * CNOT == XY,  10 11 -> 11 01, 0
81        let control_z = self.zbits[target] ^ self.zbits[control];
82        let target_x = self.xbits[control] ^ self.xbits[target];
83        self.zbits.set(control, control_z);
84        self.xbits.set(target, target_x);
85        self.rehash();
86    }
87
88    #[inline]
89    fn cz(&mut self, control: usize, target: usize) {
90        // CZ = |0><0| I + |1><1| Z
91        // CZ * II * CZ = II,   00 00 -> 00 00, 0
92        // CZ * IX * CZ = ZX,   01 00 -> 01 10, 0
93        // CZ * IY * CZ = ZY,   01 01 -> 01 11, 0
94        // CZ * IZ * CZ = ZZ,   00 01 -> 00 01, 0
95
96        // CZ * XI * CZ = XZ,   10 00 -> 10 01, 0
97        // CZ * XX * CZ = YY,   11 00 -> 11 11, 0
98        // CZ * XY * CZ = -YX,  11 01 -> 11 10, 1
99        // CZ * XZ * CZ = XI,   10 01 -> 10 00, 0
100
101        // CZ * ZI * CZ == ZI,  00 10 -> 00 10, 0
102        // CZ * ZX * CZ == IX,  01 10 -> 01 00, 0
103        // CZ * ZY * CZ == IY,  01 11 -> 01 01, 0
104        // CZ * ZZ * CZ == ZZ,  00 11 -> 00 11, 0
105
106        // CZ * YI * CZ == YZ,  10 10 -> 10 11, 0
107        // CZ * YX * CZ == -XY, 11 10 -> 11 01, 1
108        // CZ * YY * CZ == XX,  11 11 -> 11 00, 0
109        // CZ * YZ * CZ == YI,  10 11 -> 10 10, 0
110
111        // xx: identity
112        // zz:
113        // xx: 00, identity
114        // xx: 01, 00 -> 10, 01 -> 11, 10 -> 00, 11 -> 01
115        // xx: 10, 00 -> 01, 01 -> 00, 10 -> 11, 11 -> 10
116        // xx: 11, 00 -> 11, 01 -> 10, 10 -> 01, 11 -> 00
117
118        // flip the control z if target x is 1
119        let control_z = self.zbits[control] ^ self.xbits[target];
120        self.zbits.set(control, control_z);
121        // flip the target z if control x is 1
122        let target_z = self.zbits[target] ^ self.xbits[control];
123        self.zbits.set(target, target_z);
124        self.rehash();
125    }
126}

162            ("XZ", "YY"),

164            ("ZX", "ZX"),
165            ("ZY", "IY"),
166            ("ZZ", "IZ"),
167            ("YI", "YX"),
168            ("YX", "YI"),
169            ("YY", "XZ"),
170            ("YZ", "XY"),
171        ] {
172            let mut output: PauliWord<u64> = PauliWord::from(input);
