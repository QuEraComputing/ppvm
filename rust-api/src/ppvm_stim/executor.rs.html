22/// Unwrap a plain qubit target. The `validate` pass guarantees that only the
23/// control slot of a controlled Pauli may be a `rec[...]`, so every target
24/// reached here through a non-control path is a qubit.
25#[inline]
26fn qubit(t: Target) -> usize {
27    t.as_qubit()
28        .expect("non-control gate targets are validated as qubits by validate")
29}
30
31// Inline capacity for the gate-target buffers below. Sized so typical narrow
32// and moderate-width gate instructions stay on the stack; wider broadcasts
33// spill to the heap (same as a plain `Vec`), so this is never worse.
34const TARGETS_INLINE: usize = 16;
35
36/// Collect plain qubit targets, for the batched single-qubit gate paths.
37/// Stack-allocates (no per-instruction heap alloc) for up to [`TARGETS_INLINE`]
38/// targets, since `*_many` takes a `&[usize]` and the source is `&[Target]`.
39fn qubits(targets: &[Target]) -> SmallVec<[usize; TARGETS_INLINE]> {
40    targets.iter().map(|&t| qubit(t)).collect()
41}
42
43/// Collect (control, target) qubit pairs, for the batched two-qubit gate paths.
44/// Only valid when no target is a measurement record (see [`has_record_control`]).
45fn qubit_pairs(targets: &[Target]) -> SmallVec<[(usize, usize); TARGETS_INLINE / 2]> {
46    targets
47        .chunks_exact(2)
48        .map(|p| (qubit(p[0]), qubit(p[1])))
49        .collect()
50}

52/// Whether any target is a measurement-record control `rec[-k]`. Such a gate
53/// takes the per-pair feed-forward path rather than the batched fast path.
54fn has_record_control(targets: &[Target]) -> bool {
55    targets.iter().any(|t| matches!(t, Target::Rec(_)))
56}
57
58/// Resolve a measurement-record lookback `rec[-k]` against the running record.
59/// `k == 1` is the most recent measurement. A lost-qubit measurement (`None`)
60/// resolves to `false` (no feed-forward applied). An out-of-range lookback also
61/// resolves to `false`, but `validate` rejects those up front, so on a validated
62/// program the only `false`-from-missing case is a lost-qubit measurement.
63#[inline]
64fn record_bit(record: &[Option<bool>], k: usize) -> bool {
65    record
66        .len()
67        .checked_sub(k)
68        .and_then(|i| record.get(i).copied().flatten())
69        .unwrap_or(false)
70}

72/// Measure qubit `q` in the Z basis, reset it to |0>, and return the reported
73/// (readout-noise-flipped) outcome — the building block for `MR`/`MRX`/`MRY`.
74///
75/// The reset must act on the *true* (pre-flip) outcome, so this cannot delegate
76/// to `measure_noisy`; instead it applies the same `flip_with_prob` draw and
77/// overwrites the record entry with the reported value, so the measurement
78/// record matches the returned result exactly (consistent with `M`).
79fn measure_reset_z<T, I, C>(
80    tab: &mut GeneralizedTableau<T, I, C>,
81    q: usize,
82    noise: f64,
83) -> Option<bool>
84where
85    T: Config,
86    <<T as Config>::Storage as BitView>::Store: PrimInt,
87    C: SparseVector<Complex<T::Coeff>, I> + std::fmt::Debug,
88    T::Coeff: One
89        + Zero
90        + Clone
91        + num::Num
92        + ToPrimitive
93        + std::fmt::Debug
94        + std::ops::Mul<f64>
95        + PartialOrd<f64>
96        + PartialOrd
97        + Send
98        + Sync,
99    Complex<T::Coeff>: std::ops::Mul<Output = Complex<T::Coeff>>
100        + From<Complex64>
101        + std::ops::MulAssign
102        + std::ops::AddAssign
103        + One
104        + ComplexFloat
105        + Copy,
106    I: TableauIndex + Debug + Send + Sync,
107{
108    let true_outcome = tab.measure(q);
109    if true_outcome == Some(true) {
110        tab.x(q);
111    }
112    let recorded = true_outcome.map(|b| tab.flip_with_prob(b, noise));
113    tab.overwrite_last_measurement_record(recorded);
114    recorded
115}

117/// Validate and execute a parsed extended Stim program against a tableau,
118/// returning the per-measurement results in circuit order.
119pub fn execute<T, I, C>(
120    program: &ExtendedProgram,
121    tab: &mut GeneralizedTableau<T, I, C>,
122) -> Result<Vec<Option<bool>>, ExecError>
123where
124    T: Config,
125    <<T as Config>::Storage as BitView>::Store: PrimInt,
126    C: SparseVector<Complex<T::Coeff>, I> + std::fmt::Debug,
127    T::Coeff: One
128        + Zero
129        + Clone
130        + num::Num
131        + ToPrimitive
132        + std::fmt::Debug
133        + std::ops::Mul<f64>
134        + PartialOrd<f64>
135        + PartialOrd
136        + Send
137        + Sync,
138    Complex<T::Coeff>: std::ops::Mul<Output = Complex<T::Coeff>>
139        + From<Complex64>
140        + std::ops::MulAssign
141        + std::ops::AddAssign
142        + One
143        + ComplexFloat
144        + Copy,
145    I: TableauIndex + Debug + Send + Sync,
146{
147    validate(program)?;
148    let count = program.measurement_count();
149    let mut results = Vec::with_capacity(count);
150    execute_validated(&program.instructions, tab, &mut results);
151    Ok(results)
152}

289/// Like [`sample`] but skips validation — call only when the program has
290/// already been validated (e.g. via [`validate`](fn@validate)).
291///
292/// When the `rayon` feature is enabled this dispatches to
293/// [`sample_parallel_validated`] for large batches, falling back to
294/// [`sample_serial_validated`] for small ones.
295#[cfg(feature = "rayon")]
296pub fn sample_validated<T, I, C, F>(
297    instructions: &[ExtendedInstruction],
298    measurement_count: usize,
299    num_shots: usize,
300    make_tableau: F,
301) -> Vec<Vec<Option<bool>>>
302where
303    T: Config,
304    <<T as Config>::Storage as BitView>::Store: PrimInt,
305    C: SparseVector<Complex<T::Coeff>, I> + std::fmt::Debug,
306    T::Coeff: One
307        + Zero
308        + Clone
309        + num::Num
310        + ToPrimitive
311        + std::fmt::Debug
312        + std::ops::Mul<f64>
313        + PartialOrd<f64>
314        + PartialOrd
315        + Send
316        + Sync,
317    Complex<T::Coeff>: std::ops::Mul<Output = Complex<T::Coeff>>
318        + From<Complex64>
319        + std::ops::MulAssign
320        + std::ops::AddAssign
321        + One
322        + ComplexFloat
323        + Copy,
324    I: TableauIndex + Debug + Send + Sync,
325    F: Fn(usize) -> GeneralizedTableau<T, I, C> + Sync,
326{
327    // Below ~4 shots per thread, rayon's scheduling overhead outweighs
328    // the gain, so stay serial; with a single thread there is no upside.
329    let n_threads = rayon::current_num_threads();
330    if n_threads <= 1 || num_shots < 4 * n_threads {
331        sample_serial_validated(instructions, measurement_count, num_shots, make_tableau)
332    } else {
333        sample_parallel_validated(instructions, measurement_count, num_shots, make_tableau)
334    }
335}

509/// Dispatch a slice of validated instructions onto a tableau, appending
510/// measurement bits to `results`. Skips validation — the caller is responsible
511/// for having run [`validate`](fn@validate) on the originating program. Used by
512/// [`execute`] / [`sample`] internally and by the Python `tab.run()` path
513/// where `StimProgram` already cached the validate step.
514pub fn execute_validated<T, I, C>(
515    instructions: &[ExtendedInstruction],
516    tab: &mut GeneralizedTableau<T, I, C>,
517    results: &mut Vec<Option<bool>>,
518) where
519    T: Config,
520    <<T as Config>::Storage as BitView>::Store: PrimInt,
521    C: SparseVector<Complex<T::Coeff>, I> + std::fmt::Debug,
522    T::Coeff: One
523        + Zero
524        + Clone
525        + num::Num
526        + ToPrimitive
527        + std::fmt::Debug
528        + std::ops::Mul<f64>
529        + PartialOrd<f64>
530        + PartialOrd
531        + Send
532        + Sync,
533    Complex<T::Coeff>: std::ops::Mul<Output = Complex<T::Coeff>>
534        + From<Complex64>
535        + std::ops::MulAssign
536        + std::ops::AddAssign
537        + One
538        + ComplexFloat
539        + Copy,
540    I: TableauIndex + Debug + Send + Sync,
541{
542    for instr in instructions {
543        match instr {
544            ExtendedInstruction::Gate(GateOp { name, targets, .. }) => match name {
545                GateName::Reset | GateName::ResetZ => {
546                    targets.iter().for_each(|&t| tab.reset(qubit(t)));
547                }
548                // RX: reset to |+> (Z-basis reset, then rotate into the X basis).
549                GateName::ResetX => targets.iter().for_each(|&t| {
550                    let q = qubit(t);
551                    tab.reset(q);
552                    tab.h(q);
553                }),
554                // RY: reset to |i> (|0> -> |+> -> S|+> = |i>).
555                GateName::ResetY => targets.iter().for_each(|&t| {
556                    let q = qubit(t);
557                    tab.reset(q);
558                    tab.h(q);
559                    tab.s(q);
560                }),
561                GateName::X => tab.x_many(&qubits(targets)),
562                GateName::Y => tab.y_many(&qubits(targets)),
563                GateName::Z => tab.z_many(&qubits(targets)),
564                GateName::H | GateName::HXZ => tab.h_many(&qubits(targets)),
565                GateName::S | GateName::SqrtZ => tab.s_many(&qubits(targets)),
566                GateName::SDag | GateName::SqrtZDag => tab.s_dag_many(&qubits(targets)),
567                GateName::SqrtX => tab.sqrt_x_many(&qubits(targets)),
568                GateName::SqrtXDag => tab.sqrt_x_dag_many(&qubits(targets)),
569                GateName::SqrtY => tab.sqrt_y_many(&qubits(targets)),
570                GateName::SqrtYDag => tab.sqrt_y_dag_many(&qubits(targets)),
571                GateName::Identity => {}
572                // Controlled Paulis. The control slot may be a measurement record
573                // `rec[-k]` (classical feed-forward): apply the target Pauli iff
574                // the recorded bit is 1, exactly as Stim's `single_cx`/`single_cy`.
575                // With no record control present, keep the batched fast path.
576                GateName::CX | GateName::ZCX | GateName::CNot => {
577                    if has_record_control(targets) {
578                        for p in targets.chunks_exact(2) {
579                            match p[0] {
580                                Target::Qubit(c) => tab.cnot(c, qubit(p[1])),
581                                Target::Rec(k) => {
582                                    if record_bit(&tab.measurement_record, k) {
583                                        tab.x(qubit(p[1]));
584                                    }
585                                }
586                            }
587                        }
588                    } else {
589                        tab.cnot_many(&qubit_pairs(targets));
590                    }
591                }
592                GateName::CY | GateName::ZCY => {
593                    if has_record_control(targets) {
594                        for p in targets.chunks_exact(2) {
595                            match p[0] {
596                                Target::Qubit(c) => tab.cy(c, qubit(p[1])),
597                                Target::Rec(k) => {
598                                    if record_bit(&tab.measurement_record, k) {
599                                        tab.y(qubit(p[1]));
600                                    }
601                                }
602                            }
603                        }
604                    } else {
605                        tab.cy_many(&qubit_pairs(targets));
606                    }
607                }
608                GateName::CZ | GateName::ZCZ => {
609                    if has_record_control(targets) {
610                        for p in targets.chunks_exact(2) {
611                            match p[0] {
612                                Target::Qubit(c) => tab.cz(c, qubit(p[1])),
613                                Target::Rec(k) => {
614                                    if record_bit(&tab.measurement_record, k) {
615                                        tab.z(qubit(p[1]));
616                                    }
617                                }
618                            }
619                        }
620                    } else {
621                        tab.cz_many(&qubit_pairs(targets));
622                    }
623                }
624                GateName::Swap
625                | GateName::ISwap
626                | GateName::ISwapDag
627                | GateName::SqrtXX
628                | GateName::SqrtYY
629                | GateName::SqrtZZ
630                | GateName::CXSwap
631                | GateName::SwapCX
632                | GateName::XCX
633                | GateName::XCY
634                | GateName::XCZ
635                | GateName::YCX
636                | GateName::YCY
637                | GateName::YCZ
638                | GateName::CXYZ
639                | GateName::CZYX
640                | GateName::HXY
641                | GateName::HYZ => {
642                    unreachable!("unsupported gate {name:?} should have been rejected by validate")
643                }
644                GateName::T | GateName::TDag => {
645                    unreachable!("T/T_DAG are lowered to ExtendedInstruction::T/TDag by interpret")
646                }
647            },
648            ExtendedInstruction::T { targets, .. } => targets.iter().for_each(|&q| tab.t(q)),
649            ExtendedInstruction::TDag { targets, .. } => {
650                targets.iter().for_each(|&q| tab.t_dag(q));
651            }
652            ExtendedInstruction::Rotation {
653                axis,
654                theta,
655                targets,
656                ..
657            } => match axis {
658                Axis::X => targets.iter().for_each(|&q| tab.rx(q, *theta)),
659                Axis::Y => targets.iter().for_each(|&q| tab.ry(q, *theta)),
660                Axis::Z => targets.iter().for_each(|&q| tab.rz(q, *theta)),
661            },
662            ExtendedInstruction::U3 {
663                theta,
664                phi,
665                lambda,
666                targets,
667                ..
668            } => targets
669                .iter()
670                .for_each(|&q| tab.u3(q, (*theta).into(), (*phi).into(), (*lambda).into())),
671            ExtendedInstruction::Noise(NoiseOp {
672                name,
673                targets,
674                args,
675                ..
676            }) => match name {
677                NoiseName::Depolarize1 => {
678                    debug_assert_eq!(args.len(), 1);
679                    let p = args[0];
680                    for &q in targets {
681                        tab.depolarize1(q, p.into());
682                    }
683                }
684                NoiseName::Depolarize2 => {
685                    debug_assert_eq!(args.len(), 1);
686                    let p = args[0];
687                    for (a, b) in targets.iter().copied().tuples() {
688                        tab.depolarize2(a, b, p.into());
689                    }
690                }
691                NoiseName::PauliChannel1 => {
692                    debug_assert_eq!(args.len(), 3);
693                    let ps: [T::Coeff; 3] = [args[0].into(), args[1].into(), args[2].into()];
694                    for &q in targets {
695                        tab.pauli_error(q, ps.clone());
696                    }
697                }
698                NoiseName::PauliChannel2 => {
699                    debug_assert_eq!(args.len(), 15);
700                    let ps: [T::Coeff; 15] = std::array::from_fn(|i| args[i].into());
701                    debug_assert!(targets.len().is_even());
702                    for (a, b) in targets.iter().copied().tuples() {
703                        tab.two_qubit_pauli_error(a, b, ps.clone());
704                    }
705                }
706                NoiseName::XError | NoiseName::YError | NoiseName::ZError => {
707                    debug_assert_eq!(args.len(), 1);
708                    let p: T::Coeff = args[0].into();
709                    let zero = T::Coeff::zero();
710                    let ps: [T::Coeff; 3] = match name {
711                        NoiseName::XError => [p, zero.clone(), zero],
712                        NoiseName::YError => [zero.clone(), p, zero],
713                        NoiseName::ZError => [zero.clone(), zero, p],
714                        _ => unreachable!(),
715                    };
716                    for &q in targets {
717                        tab.pauli_error(q, ps.clone());
718                    }
719                }
720                NoiseName::IError
721                | NoiseName::HeraldedErase
722                | NoiseName::HeraldedPauliChannel1
723                | NoiseName::CorrelatedError
724                | NoiseName::ElseCorrelatedError => {
725                    unreachable!("unsupported noise {name:?} should have been rejected by validate")
726                }
727            },
728            ExtendedInstruction::Loss { p, targets, .. } => {
729                for &q in targets {
730                    tab.loss_channel(q, (*p).into());
731                }
732            }
733            ExtendedInstruction::CorrelatedLoss { ps, targets, .. } => {
734                let ps: [T::Coeff; 3] = [ps[0].into(), ps[1].into(), ps[2].into()];
735                for &(a, b) in targets {
736                    tab.correlated_loss_channel(a, b, ps.clone());
737                }
738            }
739            ExtendedInstruction::Measure(MeasureOp {
740                name,
741                args,
742                targets,
743                ..
744            }) => {
745                let noise = args.first().copied().unwrap_or(0.0);
746                match name {
747                    MeasureName::M | MeasureName::MZ => {
748                        // Noiseless readout: measure the whole target list at once
749                        // so the shared scratch is reused across targets, mirroring
750                        // the batched gate paths above. With readout noise, keep the
751                        // per-qubit loop so each measure/flip pair's RNG draws stay
752                        // interleaved exactly as before.
753                        if noise > 0.0 {
754                            for &q in targets {
755                                results.push(tab.measure_noisy(q, noise));
756                            }
757                        } else {
758                            results.extend(tab.measure_many(targets));
759                        }
760                    }
761                    // MR cannot delegate to `measure_noisy` because the reset must use
762                    // the *true* outcome — but it shares the readout flip via
763                    // `flip_with_prob`, so the RNG-draw shape matches MZ exactly.
764                    MeasureName::MR => {
765                        for &q in targets {
766                            results.push(measure_reset_z(tab, q, noise));
767                        }
768                    }
769                    // X/Y-basis measurements via Stim's basis-change decomposition:
770                    // conjugate so the measured axis maps to Z (`MX = H M H`,
771                    // `MY = S_DAG H M H S`), do the Z-basis measurement, then
772                    // conjugate back. Reuses `measure_noisy` so the readout-flip /
773                    // RNG-draw shape matches `M` exactly.
774                    MeasureName::MX => {
775                        for &q in targets {
776                            tab.h(q);
777                            results.push(tab.measure_noisy(q, noise));
778                            tab.h(q);
779                        }
780                    }
781                    MeasureName::MY => {
782                        for &q in targets {
783                            tab.s_dag(q);
784                            tab.h(q);
785                            results.push(tab.measure_noisy(q, noise));
786                            tab.h(q);
787                            tab.s(q);
788                        }
789                    }
790                    // MRX/MRY add a reset to the measured eigenstate
791                    // (`MRX = H M R H`, `MRY = S_DAG H M R H S`). The reset must
792                    // use the *true* outcome, so it mirrors the MR arm rather than
793                    // delegating to `measure_noisy`.
794                    MeasureName::MRX => {
795                        for &q in targets {
796                            tab.h(q);
797                            results.push(measure_reset_z(tab, q, noise));
798                            tab.h(q);
799                        }
800                    }
801                    MeasureName::MRY => {
802                        for &q in targets {
803                            tab.s_dag(q);
804                            tab.h(q);
805                            results.push(measure_reset_z(tab, q, noise));
806                            tab.h(q);
807                            tab.s(q);
808                        }
809                    }
810                    MeasureName::MXX | MeasureName::MYY | MeasureName::MZZ | MeasureName::MPP => {
811                        unreachable!(
812                            "unsupported measure {name:?} should have been rejected by validate"
813                        )
814                    }
815                }
816            }
817            ExtendedInstruction::MPad { bits, prob, .. } => {
818                let noise = prob.unwrap_or(0.0);
819                for &bit in bits {
820                    let recorded = Some(tab.flip_with_prob(bit, noise));
821                    tab.append_measurement_record(recorded);
822                    results.push(recorded);
823                }
824            }
825            // MPP: measure each Pauli product non-destructively via Stim's
826            // basis-change + CX-ladder gadget. Each factor's axis is rotated to
827            // Z (`X = H · Z · H`, `Y = (S H) · Z · (H S_DAG)`), a CX ladder onto
828            // the first qubit maps the product `Z_0 Z_1 ... Z_{m-1}` to a single
829            // `Z_0`, that qubit is measured, then the ladder and basis changes
830            // are undone so only the product operator is projected.
831            ExtendedInstruction::Mpp(MppOp { products, args, .. }) => {
832                let noise = args.first().copied().unwrap_or(0.0);
833                for product in products {
834                    for f in product {
835                        match f.axis {
836                            PauliAxis::X => tab.h(f.qubit),
837                            PauliAxis::Y => {
838                                tab.s_dag(f.qubit);
839                                tab.h(f.qubit);
840                            }
841                            PauliAxis::Z => {}
842                        }
843                    }
844                    let q0 = product[0].qubit;
845                    for f in &product[1..] {
846                        tab.cnot(f.qubit, q0);
847                    }
848                    results.push(tab.measure_noisy(q0, noise));
849                    for f in product[1..].iter().rev() {
850                        tab.cnot(f.qubit, q0);
851                    }
852                    for f in product {
853                        match f.axis {
854                            PauliAxis::X => tab.h(f.qubit),
855                            PauliAxis::Y => {
856                                tab.h(f.qubit);
857                                tab.s(f.qubit);
858                            }
859                            PauliAxis::Z => {}
860                        }
861                    }
862                }
863            }
864            ExtendedInstruction::Annotation(_) => { /* no-op */ }
865            ExtendedInstruction::Repeat { count, body, .. } => {
866                for _ in 0..*count {
867                    execute_validated(body, tab, results);
868                }
869            }
870        }
871    }
