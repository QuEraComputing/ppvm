20/// Unwrap a plain qubit target. The `validate` pass guarantees that only the
21/// control slot of a controlled Pauli may be a `rec[...]`, so every target
22/// reached here through a non-control path is a qubit.
23#[inline]
24fn qubit(t: Target) -> usize {
25    t.as_qubit()
26        .expect("non-control gate targets are validated as qubits by validate")
27}
28
29// Inline capacity for the gate-target buffers below. Sized so typical narrow
30// and moderate-width gate instructions stay on the stack; wider broadcasts
31// spill to the heap (same as a plain `Vec`), so this is never worse.
32const TARGETS_INLINE: usize = 16;
33
34/// Collect plain qubit targets, for the batched single-qubit gate paths.
35/// Stack-allocates (no per-instruction heap alloc) for up to [`TARGETS_INLINE`]
36/// targets, since `*_many` takes a `&[usize]` and the source is `&[Target]`.
37fn qubits(targets: &[Target]) -> SmallVec<[usize; TARGETS_INLINE]> {
38    targets.iter().map(|&t| qubit(t)).collect()
39}
40
41/// Collect (control, target) qubit pairs, for the batched two-qubit gate paths.
42/// Only valid when no target is a measurement record (see [`has_record_control`]).
43fn qubit_pairs(targets: &[Target]) -> SmallVec<[(usize, usize); TARGETS_INLINE / 2]> {
44    targets
45        .chunks_exact(2)
46        .map(|p| (qubit(p[0]), qubit(p[1])))
47        .collect()
48}

50/// Whether any target is a measurement-record control `rec[-k]`. Such a gate
51/// takes the per-pair feed-forward path rather than the batched fast path.
52fn has_record_control(targets: &[Target]) -> bool {
53    targets.iter().any(|t| matches!(t, Target::Rec(_)))
54}
55
56/// Resolve a measurement-record lookback `rec[-k]` against the running record.
57/// `k == 1` is the most recent measurement. A lost-qubit measurement (`None`)
58/// resolves to `false` (no feed-forward applied). An out-of-range lookback also
59/// resolves to `false`, but `validate` rejects those up front, so on a validated
60/// program the only `false`-from-missing case is a lost-qubit measurement.
61#[inline]
62fn record_bit(record: &[Option<bool>], k: usize) -> bool {
63    record
64        .len()
65        .checked_sub(k)
66        .and_then(|i| record.get(i).copied().flatten())
67        .unwrap_or(false)
68}

70/// Measure qubit `q` in the Z basis, reset it to |0>, and return the reported
71/// (readout-noise-flipped) outcome — the building block for `MR`/`MRX`/`MRY`.
72///
73/// The reset must act on the *true* (pre-flip) outcome, so this cannot delegate
74/// to `measure_noisy`; instead it applies the same `flip_with_prob` draw and
75/// overwrites the record entry with the reported value, so the measurement
76/// record matches the returned result exactly (consistent with `M`).
77fn measure_reset_z<T, I, C>(
78    tab: &mut GeneralizedTableau<T, I, C>,
79    q: usize,
80    noise: f64,
81) -> Option<bool>
82where
83    T: Config,
84    <<T as Config>::Storage as BitView>::Store: PrimInt,
85    C: SparseVector<Complex<T::Coeff>, I> + std::fmt::Debug,
86    T::Coeff: One
87        + Zero
88        + Clone
89        + num::Num
90        + ToPrimitive
91        + std::fmt::Debug
92        + std::ops::Mul<f64>
93        + PartialOrd<f64>
94        + PartialOrd
95        + Send
96        + Sync,
97    Complex<T::Coeff>: std::ops::Mul<Output = Complex<T::Coeff>>
98        + From<Complex64>
99        + std::ops::MulAssign
100        + std::ops::AddAssign
101        + One
102        + ComplexFloat
103        + Copy,
104    I: TableauIndex + Debug + Send + Sync,
105{
106    let true_outcome = tab.measure(q);
107    if true_outcome == Some(true) {
108        tab.x(q);
109    }
110    let recorded = true_outcome.map(|b| tab.flip_with_prob(b, noise));
111    tab.overwrite_last_measurement_record(recorded);
112    recorded
113}

115/// Validate and execute a parsed extended Stim program against a tableau,
116/// returning the per-measurement results in circuit order.
117pub fn execute<T, I, C>(
118    program: &ExtendedProgram,
119    tab: &mut GeneralizedTableau<T, I, C>,
120) -> Result<Vec<Option<bool>>, ExecError>
121where
122    T: Config,
123    <<T as Config>::Storage as BitView>::Store: PrimInt,
124    C: SparseVector<Complex<T::Coeff>, I> + std::fmt::Debug,
125    T::Coeff: One
126        + Zero
127        + Clone
128        + num::Num
129        + ToPrimitive
130        + std::fmt::Debug
131        + std::ops::Mul<f64>
132        + PartialOrd<f64>
133        + PartialOrd
134        + Send
135        + Sync,
136    Complex<T::Coeff>: std::ops::Mul<Output = Complex<T::Coeff>>
137        + From<Complex64>
138        + std::ops::MulAssign
139        + std::ops::AddAssign
140        + One
141        + ComplexFloat
142        + Copy,
143    I: TableauIndex + Debug + Send + Sync,
144{
145    validate(program)?;
146    let count = program.measurement_count();
147    let mut results = Vec::with_capacity(count);
148    execute_validated(&program.instructions, tab, &mut results);
149    Ok(results)
150}

287/// Like [`sample`] but skips validation — call only when the program has
288/// already been validated (e.g. via [`validate`](fn@validate)).
289///
290/// When the `rayon` feature is enabled this dispatches to
291/// [`sample_parallel_validated`] for large batches, falling back to
292/// [`sample_serial_validated`] for small ones.
293#[cfg(feature = "rayon")]
294pub fn sample_validated<T, I, C, F>(
295    instructions: &[stim_parser::extended::ExtendedInstruction],
296    measurement_count: usize,
297    num_shots: usize,
298    make_tableau: F,
299) -> Vec<Vec<Option<bool>>>
300where
301    T: Config,
302    <<T as Config>::Storage as BitView>::Store: PrimInt,
303    C: SparseVector<Complex<T::Coeff>, I> + std::fmt::Debug,
304    T::Coeff: One
305        + Zero
306        + Clone
307        + num::Num
308        + ToPrimitive
309        + std::fmt::Debug
310        + std::ops::Mul<f64>
311        + PartialOrd<f64>
312        + PartialOrd
313        + Send
314        + Sync,
315    Complex<T::Coeff>: std::ops::Mul<Output = Complex<T::Coeff>>
316        + From<Complex64>
317        + std::ops::MulAssign
318        + std::ops::AddAssign
319        + One
320        + ComplexFloat
321        + Copy,
322    I: TableauIndex + Debug + Send + Sync,
323    F: Fn(usize) -> GeneralizedTableau<T, I, C> + Sync,
324{
325    // Below ~4 shots per thread, rayon's scheduling overhead outweighs
326    // the gain, so stay serial; with a single thread there is no upside.
327    let n_threads = rayon::current_num_threads();
328    if n_threads <= 1 || num_shots < 4 * n_threads {
329        sample_serial_validated(instructions, measurement_count, num_shots, make_tableau)
330    } else {
331        sample_parallel_validated(instructions, measurement_count, num_shots, make_tableau)
332    }
333}

507/// Dispatch a slice of validated instructions onto a tableau, appending
508/// measurement bits to `results`. Skips validation — the caller is responsible
509/// for having run [`validate`](fn@validate) on the originating program. Used by
510/// [`execute`] / [`sample`] internally and by the Python `tab.run()` path
511/// where `StimProgram` already cached the validate step.
512pub fn execute_validated<T, I, C>(
513    instructions: &[ExtendedInstruction],
514    tab: &mut GeneralizedTableau<T, I, C>,
515    results: &mut Vec<Option<bool>>,
516) where
517    T: Config,
518    <<T as Config>::Storage as BitView>::Store: PrimInt,
519    C: SparseVector<Complex<T::Coeff>, I> + std::fmt::Debug,
520    T::Coeff: One
521        + Zero
522        + Clone
523        + num::Num
524        + ToPrimitive
525        + std::fmt::Debug
526        + std::ops::Mul<f64>
527        + PartialOrd<f64>
528        + PartialOrd
529        + Send
530        + Sync,
531    Complex<T::Coeff>: std::ops::Mul<Output = Complex<T::Coeff>>
532        + From<Complex64>
533        + std::ops::MulAssign
534        + std::ops::AddAssign
535        + One
536        + ComplexFloat
537        + Copy,
538    I: TableauIndex + Debug + Send + Sync,
539{
540    for instr in instructions {
541        match instr {
542            ExtendedInstruction::Raw(RawPassthrough::Gate { name, targets, .. }) => match name {
543                GateName::Reset | GateName::ResetZ => {
544                    targets.iter().for_each(|&t| tab.reset(qubit(t)));
545                }
546                // RX: reset to |+> (Z-basis reset, then rotate into the X basis).
547                GateName::ResetX => targets.iter().for_each(|&t| {
548                    let q = qubit(t);
549                    tab.reset(q);
550                    tab.h(q);
551                }),
552                // RY: reset to |i> (|0> -> |+> -> S|+> = |i>).
553                GateName::ResetY => targets.iter().for_each(|&t| {
554                    let q = qubit(t);
555                    tab.reset(q);
556                    tab.h(q);
557                    tab.s(q);
558                }),
559                GateName::X => tab.x_many(&qubits(targets)),
560                GateName::Y => tab.y_many(&qubits(targets)),
561                GateName::Z => tab.z_many(&qubits(targets)),
562                GateName::H | GateName::HXZ => tab.h_many(&qubits(targets)),
563                GateName::S | GateName::SqrtZ => tab.s_many(&qubits(targets)),
564                GateName::SDag | GateName::SqrtZDag => tab.s_dag_many(&qubits(targets)),
565                GateName::SqrtX => tab.sqrt_x_many(&qubits(targets)),
566                GateName::SqrtXDag => tab.sqrt_x_dag_many(&qubits(targets)),
567                GateName::SqrtY => tab.sqrt_y_many(&qubits(targets)),
568                GateName::SqrtYDag => tab.sqrt_y_dag_many(&qubits(targets)),
569                GateName::Identity => {}
570                // Controlled Paulis. The control slot may be a measurement record
571                // `rec[-k]` (classical feed-forward): apply the target Pauli iff
572                // the recorded bit is 1, exactly as Stim's `single_cx`/`single_cy`.
573                // With no record control present, keep the batched fast path.
574                GateName::CX | GateName::ZCX | GateName::CNot => {
575                    if has_record_control(targets) {
576                        for p in targets.chunks_exact(2) {
577                            match p[0] {
578                                Target::Qubit(c) => tab.cnot(c, qubit(p[1])),
579                                Target::Rec(k) => {
580                                    if record_bit(&tab.measurement_record, k) {
581                                        tab.x(qubit(p[1]));
582                                    }
583                                }
584                            }
585                        }
586                    } else {
587                        tab.cnot_many(&qubit_pairs(targets));
588                    }
589                }
590                GateName::CY | GateName::ZCY => {
591                    if has_record_control(targets) {
592                        for p in targets.chunks_exact(2) {
593                            match p[0] {
594                                Target::Qubit(c) => tab.cy(c, qubit(p[1])),
595                                Target::Rec(k) => {
596                                    if record_bit(&tab.measurement_record, k) {
597                                        tab.y(qubit(p[1]));
598                                    }
599                                }
600                            }
601                        }
602                    } else {
603                        tab.cy_many(&qubit_pairs(targets));
604                    }
605                }
606                GateName::CZ | GateName::ZCZ => {
607                    if has_record_control(targets) {
608                        for p in targets.chunks_exact(2) {
609                            match p[0] {
610                                Target::Qubit(c) => tab.cz(c, qubit(p[1])),
611                                Target::Rec(k) => {
612                                    if record_bit(&tab.measurement_record, k) {
613                                        tab.z(qubit(p[1]));
614                                    }
615                                }
616                            }
617                        }
618                    } else {
619                        tab.cz_many(&qubit_pairs(targets));
620                    }
621                }
622                GateName::Swap
623                | GateName::ISwap
624                | GateName::ISwapDag
625                | GateName::SqrtXX
626                | GateName::SqrtYY
627                | GateName::SqrtZZ
628                | GateName::CXSwap
629                | GateName::SwapCX
630                | GateName::XCX
631                | GateName::XCY
632                | GateName::XCZ
633                | GateName::YCX
634                | GateName::YCY
635                | GateName::YCZ
636                | GateName::CXYZ
637                | GateName::CZYX
638                | GateName::HXY
639                | GateName::HYZ => {
640                    unreachable!("unsupported gate {name:?} should have been rejected by validate")
641                }
642                GateName::T | GateName::TDag => {
643                    unreachable!("T/T_DAG are lowered to ExtendedInstruction::T/TDag by interpret")
644                }
645            },
646            ExtendedInstruction::T { targets, .. } => targets.iter().for_each(|&q| tab.t(q)),
647            ExtendedInstruction::TDag { targets, .. } => {
648                targets.iter().for_each(|&q| tab.t_dag(q));
649            }
650            ExtendedInstruction::Rotation {
651                axis,
652                theta,
653                targets,
654                ..
655            } => match axis {
656                Axis::X => targets.iter().for_each(|&q| tab.rx(q, *theta)),
657                Axis::Y => targets.iter().for_each(|&q| tab.ry(q, *theta)),
658                Axis::Z => targets.iter().for_each(|&q| tab.rz(q, *theta)),
659            },
660            ExtendedInstruction::U3 {
661                theta,
662                phi,
663                lambda,
664                targets,
665                ..
666            } => targets
667                .iter()
668                .for_each(|&q| tab.u3(q, (*theta).into(), (*phi).into(), (*lambda).into())),
669            ExtendedInstruction::Raw(RawPassthrough::Noise {
670                name,
671                targets,
672                args,
673                ..
674            }) => match name {
675                NoiseName::Depolarize1 => {
676                    debug_assert_eq!(args.len(), 1);
677                    let p = args[0];
678                    for &q in targets {
679                        tab.depolarize1(q, p.into());
680                    }
681                }
682                NoiseName::Depolarize2 => {
683                    debug_assert_eq!(args.len(), 1);
684                    let p = args[0];
685                    for (a, b) in targets.iter().copied().tuples() {
686                        tab.depolarize2(a, b, p.into());
687                    }
688                }
689                NoiseName::PauliChannel1 => {
690                    debug_assert_eq!(args.len(), 3);
691                    let ps: [T::Coeff; 3] = [args[0].into(), args[1].into(), args[2].into()];
692                    for &q in targets {
693                        tab.pauli_error(q, ps.clone());
694                    }
695                }
696                NoiseName::PauliChannel2 => {
697                    debug_assert_eq!(args.len(), 15);
698                    let ps: [T::Coeff; 15] = std::array::from_fn(|i| args[i].into());
699                    debug_assert!(targets.len().is_even());
700                    for (a, b) in targets.iter().copied().tuples() {
701                        tab.two_qubit_pauli_error(a, b, ps.clone());
702                    }
703                }
704                NoiseName::XError | NoiseName::YError | NoiseName::ZError => {
705                    debug_assert_eq!(args.len(), 1);
706                    let p: T::Coeff = args[0].into();
707                    let zero = T::Coeff::zero();
708                    let ps: [T::Coeff; 3] = match name {
709                        NoiseName::XError => [p, zero.clone(), zero],
710                        NoiseName::YError => [zero.clone(), p, zero],
711                        NoiseName::ZError => [zero.clone(), zero, p],
712                        _ => unreachable!(),
713                    };
714                    for &q in targets {
715                        tab.pauli_error(q, ps.clone());
716                    }
717                }
718                NoiseName::IError
719                | NoiseName::HeraldedErase
720                | NoiseName::HeraldedPauliChannel1
721                | NoiseName::CorrelatedError
722                | NoiseName::ElseCorrelatedError => {
723                    unreachable!("unsupported noise {name:?} should have been rejected by validate")
724                }
725            },
726            ExtendedInstruction::Loss { p, targets, .. } => {
727                for &q in targets {
728                    tab.loss_channel(q, (*p).into());
729                }
730            }
731            ExtendedInstruction::CorrelatedLoss { ps, targets, .. } => {
732                let ps: [T::Coeff; 3] = [ps[0].into(), ps[1].into(), ps[2].into()];
733                for &(a, b) in targets {
734                    tab.correlated_loss_channel(a, b, ps.clone());
735                }
736            }
737            ExtendedInstruction::Raw(RawPassthrough::Measure {
738                name,
739                args,
740                targets,
741                ..
742            }) => {
743                let noise = args.first().copied().unwrap_or(0.0);
744                match name {
745                    MeasureName::M | MeasureName::MZ => {
746                        // Noiseless readout: measure the whole target list at once
747                        // so the shared scratch is reused across targets, mirroring
748                        // the batched gate paths above. With readout noise, keep the
749                        // per-qubit loop so each measure/flip pair's RNG draws stay
750                        // interleaved exactly as before.
751                        if noise > 0.0 {
752                            for &q in targets {
753                                results.push(tab.measure_noisy(q, noise));
754                            }
755                        } else {
756                            results.extend(tab.measure_many(targets));
757                        }
758                    }
759                    // MR cannot delegate to `measure_noisy` because the reset must use
760                    // the *true* outcome — but it shares the readout flip via
761                    // `flip_with_prob`, so the RNG-draw shape matches MZ exactly.
762                    MeasureName::MR => {
763                        for &q in targets {
764                            results.push(measure_reset_z(tab, q, noise));
765                        }
766                    }
767                    // X/Y-basis measurements via Stim's basis-change decomposition:
768                    // conjugate so the measured axis maps to Z (`MX = H M H`,
769                    // `MY = S_DAG H M H S`), do the Z-basis measurement, then
770                    // conjugate back. Reuses `measure_noisy` so the readout-flip /
771                    // RNG-draw shape matches `M` exactly.
772                    MeasureName::MX => {
773                        for &q in targets {
774                            tab.h(q);
775                            results.push(tab.measure_noisy(q, noise));
776                            tab.h(q);
777                        }
778                    }
779                    MeasureName::MY => {
780                        for &q in targets {
781                            tab.s_dag(q);
782                            tab.h(q);
783                            results.push(tab.measure_noisy(q, noise));
784                            tab.h(q);
785                            tab.s(q);
786                        }
787                    }
788                    // MRX/MRY add a reset to the measured eigenstate
789                    // (`MRX = H M R H`, `MRY = S_DAG H M R H S`). The reset must
790                    // use the *true* outcome, so it mirrors the MR arm rather than
791                    // delegating to `measure_noisy`.
792                    MeasureName::MRX => {
793                        for &q in targets {
794                            tab.h(q);
795                            results.push(measure_reset_z(tab, q, noise));
796                            tab.h(q);
797                        }
798                    }
799                    MeasureName::MRY => {
800                        for &q in targets {
801                            tab.s_dag(q);
802                            tab.h(q);
803                            results.push(measure_reset_z(tab, q, noise));
804                            tab.h(q);
805                            tab.s(q);
806                        }
807                    }
808                    MeasureName::MXX | MeasureName::MYY | MeasureName::MZZ | MeasureName::MPP => {
809                        unreachable!(
810                            "unsupported measure {name:?} should have been rejected by validate"
811                        )
812                    }
813                }
814            }
815            ExtendedInstruction::MPad { bits, prob, .. } => {
816                let noise = prob.unwrap_or(0.0);
817                for &bit in bits {
818                    let recorded = Some(tab.flip_with_prob(bit, noise));
819                    tab.append_measurement_record(recorded);
820                    results.push(recorded);
821                }
822            }
823            // MPP: measure each Pauli product non-destructively via Stim's
824            // basis-change + CX-ladder gadget. Each factor's axis is rotated to
825            // Z (`X = H · Z · H`, `Y = (S H) · Z · (H S_DAG)`), a CX ladder onto
826            // the first qubit maps the product `Z_0 Z_1 ... Z_{m-1}` to a single
827            // `Z_0`, that qubit is measured, then the ladder and basis changes
828            // are undone so only the product operator is projected.
829            ExtendedInstruction::Mpp { products, args, .. } => {
830                let noise = args.first().copied().unwrap_or(0.0);
831                for product in products {
832                    for f in product {
833                        match f.axis {
834                            PauliAxis::X => tab.h(f.qubit),
835                            PauliAxis::Y => {
836                                tab.s_dag(f.qubit);
837                                tab.h(f.qubit);
838                            }
839                            PauliAxis::Z => {}
840                        }
841                    }
842                    let q0 = product[0].qubit;
843                    for f in &product[1..] {
844                        tab.cnot(f.qubit, q0);
845                    }
846                    results.push(tab.measure_noisy(q0, noise));
847                    for f in product[1..].iter().rev() {
848                        tab.cnot(f.qubit, q0);
849                    }
850                    for f in product {
851                        match f.axis {
852                            PauliAxis::X => tab.h(f.qubit),
853                            PauliAxis::Y => {
854                                tab.h(f.qubit);
855                                tab.s(f.qubit);
856                            }
857                            PauliAxis::Z => {}
858                        }
859                    }
860                }
861            }
862            ExtendedInstruction::Raw(RawPassthrough::Annotation { .. }) => { /* no-op */ }
863            ExtendedInstruction::Repeat { count, body, .. } => {
864                for _ in 0..*count {
865                    execute_validated(body, tab, results);
866                }
867            }
868        }
869    }
