1//! Parse and execute Stim circuits against a [`GeneralizedTableau`].
2//!
3//! Two-stage pipeline:
4//!
5//! 1. [`parse_extended`] — `&str` → [`ExtendedProgram`] (re-exported from
6//!    [`stim_parser`]).
7//! 2. [`execute`] / [`sample`] — call [`prepare`] to validate the
8//!    [`ExtendedProgram`], then apply it to a [`GeneralizedTableau`].
9//!
10//! Multi-shot usage should call [`parse_extended`] once and pass the parsed
11//! program to [`sample`]. The [`run_string`] / [`run_file`] convenience helpers
12//! re-parse on every call and are intended for single-shot demos only.
13//!
14//! # Multi-shot pattern (recommended)
15//!
16//! ```ignore
17//! use ppvm_stim::{parse_extended, sample};
18//! use ppvm_tableau::prelude::*;
19//!
20//! let prog = parse_extended(circuit_src)?;
21//! let shots = sample(&prog, 10_000, || {
22//!     GeneralizedTableau::<_, usize, _>::new(n_qubits, 1e-10)
23//! })?;
24//! # Ok::<(), ppvm_stim::Error>(())
25//! ```
26//!
27//! [`run_string`] / [`run_file`] re-parse on every call and exist only for
28//! single-shot demos — never call them from a shot loop.
29//!
30//! [`ExtendedProgram`]: stim_parser::extended::ExtendedProgram
31//! [`GeneralizedTableau`]: ppvm_tableau::prelude::GeneralizedTableau
32

43#[derive(Debug, thiserror::Error)]
44pub enum Error {
45    #[error(transparent)]
46    Parse(#[from] ExtendedParseError),
47    #[error(transparent)]
48    Exec(#[from] ExecError),
49    #[error("failed to read stim file {path}: {source}")]
50    Io {
51        path: PathBuf,
52        #[source]
53        source: std::io::Error,
54    },
55}

57/// Parse → validate → execute in one shot. Re-parses each call; do **not**
58/// use in shot loops — use [`parse_extended`] + [`sample`] instead.
59pub fn run_string<T, I, C>(
60    src: &str,
61    tab: &mut ppvm_tableau::prelude::GeneralizedTableau<T, I, C>,
62) -> Result<Vec<Option<bool>>, Error>
63where
64    T: ppvm_runtime::prelude::Config,
65    <<T as ppvm_runtime::prelude::Config>::Storage as bitvec::view::BitView>::Store: num::PrimInt,
66    C: ppvm_tableau::prelude::SparseVector<num::Complex<T::Coeff>, I> + std::fmt::Debug,
67    T::Coeff: num::One
68        + num::Zero
69        + Clone
70        + num::Num
71        + num::ToPrimitive
72        + std::fmt::Debug
73        + std::ops::Mul<f64>
74        + PartialOrd<f64>
75        + Send
76        + Sync,
77    num::Complex<T::Coeff>: std::ops::Mul<Output = num::Complex<T::Coeff>>
78        + From<num::complex::Complex64>
79        + std::ops::MulAssign
80        + std::ops::AddAssign
81        + num::One
82        + num::complex::ComplexFloat
83        + Copy,
84    I: ppvm_tableau::prelude::TableauIndex + std::fmt::Debug + Send + Sync,
85{
86    let prog = parse_extended(src)?;
87    let results = execute(&prog, tab)?;
88    Ok(results)
89}

91pub fn run_file<T, I, C>(
92    path: &Path,
93    tab: &mut ppvm_tableau::prelude::GeneralizedTableau<T, I, C>,
94) -> Result<Vec<Option<bool>>, Error>
95where
96    T: ppvm_runtime::prelude::Config,
97    <<T as ppvm_runtime::prelude::Config>::Storage as bitvec::view::BitView>::Store: num::PrimInt,
98    C: ppvm_tableau::prelude::SparseVector<num::Complex<T::Coeff>, I> + std::fmt::Debug,
99    T::Coeff: num::One
100        + num::Zero
101        + Clone
102        + num::Num
103        + num::ToPrimitive
104        + std::fmt::Debug
105        + std::ops::Mul<f64>
106        + PartialOrd<f64>
107        + Send
108        + Sync,
109    num::Complex<T::Coeff>: std::ops::Mul<Output = num::Complex<T::Coeff>>
110        + From<num::complex::Complex64>
111        + std::ops::MulAssign
112        + std::ops::AddAssign
113        + num::One
114        + num::complex::ComplexFloat
115        + Copy,
116    I: ppvm_tableau::prelude::TableauIndex + std::fmt::Debug + Send + Sync,
117{
118    let src = std::fs::read_to_string(path).map_err(|source| Error::Io {
119        path: path.to_path_buf(),
120        source,
121    })?;
122    run_string(&src, tab)
123}
