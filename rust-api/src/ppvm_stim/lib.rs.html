4//! Parse and execute Stim circuits against a [`GeneralizedTableau`].
5//!
6//! Two-stage pipeline:
7//!
8//! 1. [`parse_extended`] — `&str` → [`ExtendedProgram`] (re-exported from
9//!    [`stim_parser`]).
10//! 2. [`execute`] / [`sample`] — call [`prepare`] to validate the
11//!    [`ExtendedProgram`], then apply it to a [`GeneralizedTableau`].
12//!
13//! Multi-shot usage should call [`parse_extended`] once and pass the parsed
14//! program to [`sample`]. The [`run_string`] / [`run_file`] convenience helpers
15//! re-parse on every call and are intended for single-shot demos only.
16//!
17//! # Multi-shot pattern (recommended)
18//!
19//! ```ignore
20//! use ppvm_stim::{parse_extended, sample};
21//! use ppvm_tableau::prelude::*;
22//!
23//! let prog = parse_extended(circuit_src)?;
24//! let shots = sample(&prog, 10_000, || {
25//!     GeneralizedTableau::<_, usize, _>::new(n_qubits, 1e-10)
26//! })?;
27//! # Ok::<(), ppvm_stim::Error>(())
28//! ```
29//!
30//! [`run_string`] / [`run_file`] re-parse on every call and exist only for
31//! single-shot demos — never call them from a shot loop.
32//!
33//! [`ExtendedProgram`]: stim_parser::extended::ExtendedProgram
34//! [`GeneralizedTableau`]: ppvm_tableau::prelude::GeneralizedTableau
35
36pub mod executor;
37pub mod prepare;

44use std::path::{Path, PathBuf};

46#[derive(Debug, thiserror::Error)]
47pub enum Error {
48    #[error(transparent)]
49    Parse(#[from] ExtendedParseError),
50    #[error(transparent)]
51    Exec(#[from] ExecError),
52    #[error("failed to read stim file {path}: {source}")]
53    Io {
54        path: PathBuf,
55        #[source]
56        source: std::io::Error,
57    },
58}
59
60/// Parse → validate → execute in one shot. Re-parses each call; do **not**
61/// use in shot loops — use [`parse_extended`] + [`sample`] instead.
62pub fn run_string<T, I, C>(
63    src: &str,
64    tab: &mut ppvm_tableau::prelude::GeneralizedTableau<T, I, C>,
65) -> Result<Vec<Option<bool>>, Error>
66where
67    T: ppvm_runtime::prelude::Config,
68    <<T as ppvm_runtime::prelude::Config>::Storage as bitvec::view::BitView>::Store: num::PrimInt,
69    C: ppvm_tableau::prelude::SparseVector<num::Complex<T::Coeff>, I> + std::fmt::Debug,
70    T::Coeff: num::One
71        + num::Zero
72        + Clone
73        + num::Num
74        + num::ToPrimitive
75        + std::fmt::Debug
76        + std::ops::Mul<f64>
77        + PartialOrd<f64>
78        + Send
79        + Sync,
80    num::Complex<T::Coeff>: std::ops::Mul<Output = num::Complex<T::Coeff>>
81        + From<num::complex::Complex64>
82        + std::ops::MulAssign
83        + std::ops::AddAssign
84        + num::One
85        + num::complex::ComplexFloat
86        + Copy,
87    I: ppvm_tableau::prelude::TableauIndex + std::fmt::Debug + Send + Sync,
88{
89    let prog = parse_extended(src)?;
90    let results = execute(&prog, tab)?;
91    Ok(results)
92}

94pub fn run_file<T, I, C>(
95    path: &Path,
96    tab: &mut ppvm_tableau::prelude::GeneralizedTableau<T, I, C>,
97) -> Result<Vec<Option<bool>>, Error>
98where
99    T: ppvm_runtime::prelude::Config,
100    <<T as ppvm_runtime::prelude::Config>::Storage as bitvec::view::BitView>::Store: num::PrimInt,
101    C: ppvm_tableau::prelude::SparseVector<num::Complex<T::Coeff>, I> + std::fmt::Debug,
102    T::Coeff: num::One
103        + num::Zero
104        + Clone
105        + num::Num
106        + num::ToPrimitive
107        + std::fmt::Debug
108        + std::ops::Mul<f64>
109        + PartialOrd<f64>
110        + Send
111        + Sync,
112    num::Complex<T::Coeff>: std::ops::Mul<Output = num::Complex<T::Coeff>>
113        + From<num::complex::Complex64>
114        + std::ops::MulAssign
115        + std::ops::AddAssign
116        + num::One
117        + num::complex::ComplexFloat
118        + Copy,
119    I: ppvm_tableau::prelude::TableauIndex + std::fmt::Debug + Send + Sync,
120{
121    let src = std::fs::read_to_string(path).map_err(|source| Error::Io {
122        path: path.to_path_buf(),
123        source,
124    })?;
125    run_string(&src, tab)
126}
