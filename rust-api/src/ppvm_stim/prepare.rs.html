4use stim_parser::ast::{GateName, MeasureName, NoiseName};
5use stim_parser::extended::{ExtendedInstruction, ExtendedProgram, RawPassthrough};
6
7#[derive(Debug, thiserror::Error, PartialEq)]
8pub enum ExecError {
9    #[error("unsupported instruction '{name}' at line {line}")]
10    Unsupported { name: String, line: usize },
11    #[error("invalid probability {value} for '{name}' at line {line}; expected value in [0, 1]")]
12    InvalidProbability {
13        name: String,
14        line: usize,
15        value: f64,
16    },
17}
18
19pub fn prepare(program: &ExtendedProgram) -> Result<(), ExecError> {
20    validate_slice(&program.instructions)
21}
22
23fn validate_slice(instructions: &[ExtendedInstruction]) -> Result<(), ExecError> {
24    for instr in instructions {
25        match instr {
26            ExtendedInstruction::Raw(RawPassthrough::Gate { name, line, .. }) => {
27                check_gate_supported(*name, *line)?;
28            }
29            ExtendedInstruction::Raw(RawPassthrough::Noise { name, line, .. }) => {
30                check_noise_supported(*name, *line)?;
31            }
32            ExtendedInstruction::Raw(RawPassthrough::Measure {
33                name, args, line, ..
34            }) => {
35                check_measure_supported(*name, *line)?;
36                if let Some(&p) = args.first() {
37                    check_probability(p, name.canonical_name(), *line)?;
38                }
39            }
40            ExtendedInstruction::Repeat { body, .. } => validate_slice(body)?,
41            ExtendedInstruction::Raw(RawPassthrough::Annotation { .. })
42            | ExtendedInstruction::T { .. }
43            | ExtendedInstruction::TDag { .. }
44            | ExtendedInstruction::Rotation { .. }
45            | ExtendedInstruction::U3 { .. }
46            | ExtendedInstruction::Loss { .. }
47            | ExtendedInstruction::CorrelatedLoss { .. } => {}
48            ExtendedInstruction::MPad { prob, line, .. } => {
49                if let Some(p) = prob {
50                    check_probability(*p, "MPAD", *line)?;
51                }
52            }
53        }
54    }
55    Ok(())
56}
57
58fn check_gate_supported(name: GateName, line: usize) -> Result<(), ExecError> {
59    use GateName::*;
60    match name {
61        Reset | ResetZ | X | Y | Z | H | HXZ | S | SqrtZ | SDag | SqrtZDag | SqrtX | SqrtXDag
62        | SqrtY | SqrtYDag | Identity | CX | ZCX | CNot | CY | ZCY | CZ | ZCZ => Ok(()),
63        Swap | ISwap | ISwapDag | SqrtXX | SqrtYY | SqrtZZ | CXSwap | SwapCX | XCX | XCY | XCZ
64        | YCX | YCY | YCZ | CXYZ | CZYX | HXY | HYZ => Err(ExecError::Unsupported {
65            name: name.canonical_name().to_string(),
66            line,
67        }),
68    }
69}
70
71fn check_noise_supported(name: NoiseName, line: usize) -> Result<(), ExecError> {
72    use NoiseName::*;
73    match name {
74        Depolarize1 | Depolarize2 | PauliChannel1 | PauliChannel2 | XError | YError | ZError => {
75            Ok(())
76        }
77        IError | HeraldedErase | HeraldedPauliChannel1 | CorrelatedError | ElseCorrelatedError => {
78            Err(ExecError::Unsupported {
79                name: name.canonical_name().to_string(),
80                line,
81            })
82        }
83    }
84}
85
86fn check_measure_supported(name: MeasureName, line: usize) -> Result<(), ExecError> {
87    use MeasureName::*;
88    match name {
89        M | MZ | MR => Ok(()),
90        MX | MY | MRX | MRY | MXX | MYY | MZZ | MPP => Err(ExecError::Unsupported {
91            name: name.canonical_name().to_string(),
92            line,
93        }),
94    }
95}
96
97fn check_probability(p: f64, name: &str, line: usize) -> Result<(), ExecError> {
98    if p.is_finite() && (0.0..=1.0).contains(&p) {
99        Ok(())
100    } else {
101        Err(ExecError::InvalidProbability {
102            name: name.to_string(),
103            line,
104            value: p,
105        })
106    }
107}
