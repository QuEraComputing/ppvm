22/// A `2n`-row stabilizer / destabilizer tableau.
23///
24/// Rows `0..n` hold the destabilizers; rows `n..2n` hold the
25/// stabilizers. Each row is a [`PhasedPauliWord`] tracking both its
26/// `X`/`Z` bits and a phase in `{±1, ±i}`. Implements every
27/// Clifford-only operation natively (Hadamard, phase, CNOT, CZ, etc.).
28///
29/// # Examples
30///
31/// ```
32/// use ppvm_runtime::config::fxhash::ByteF64;
33/// use ppvm_runtime::traits::Clifford;
34/// use ppvm_tableau::data::Tableau;
35///
36/// let mut tab: Tableau<ByteF64<1>> = Tableau::new(2);
37/// tab.h(0);
38/// tab.cnot(0, 1);
39/// assert_eq!(tab.n_qubits, 2);
40/// assert_eq!(tab.stabilizers().len(), 2);
41/// ```
42#[derive(Clone, Debug)]
43pub struct Tableau<T: Config> {
44    /// Number of qubits.
45    pub n_qubits: usize,
46    /// Destabilizer / Stabilizer tableau
47    /// * Entries 0..n are the destabilizers
48    /// * Entries n..2n are the stabilizers
49    pub data: Vec<PhasedPauliWordNoHash<T::Storage, T::BuildHasher>>,
50    pub(crate) rng: SmallRng,
51}
52
53impl<T: Config> Tableau<T> {
54    /// Construct a fresh tableau initialised to `|0…0⟩`.
55    pub fn new(n_qubits: usize) -> Self {
56        // Initialize tableau for 0 state
57        let mut data: Vec<PhasedPauliWordNoHash<T::Storage, T::BuildHasher>> =
58            Vec::with_capacity(2 * n_qubits);
59        let pw_cache = PhasedPauliWordNoHash::<T::Storage, T::BuildHasher>::new(n_qubits);
60        for i in 0..n_qubits {
61            // destabilizer
62            let mut pw = pw_cache.clone();
63            pw.set(i, Pauli::X);
64            data.push(pw);
65        }
66        for i in 0..n_qubits {
67            // stabilizer
68            let mut pw = pw_cache.clone();
69            pw.set(i, Pauli::Z);
70            data.push(pw);
71        }
72
73        Self {
74            n_qubits,
75            data,
76            rng: rand::make_rng(),
77        }
78    }
79
80    /// Same as [`Tableau::new`], but seed the RNG deterministically.
81    pub fn new_with_seed(n_qubits: usize, seed: u64) -> Self {
82        let mut t = Self::new(n_qubits);
83        t.rng = SmallRng::seed_from_u64(seed);
84        t
85    }

113    // some helper functions for measurement impl
114    pub(crate) fn find_z_anticommuting_stabilizer(&self, addr0: usize) -> Option<usize> {
115        // Find first stabilizer that anticommutes with Z_addr0
116        self.stabilizers()
117            .iter()
118            .position(|stab| stab.word.anticommutes_at(addr0, (false, true)))
119    }

121    pub(crate) fn get_deterministic_outcome(&self, addr0: usize) -> bool
122    where
123        <<T as Config>::Storage as BitView>::Store: PrimInt,
124    {
125        // find the outcome: either Z_addr0 or -Z_addr0 is a stabilizer
126        // the stabilizer can be computed as the product of all stabilizers S_i
127        // whose corresponding destabilizer D_i anticommutes with Z_addr0 (has X at addr0).
128        // We have to actually multiply Paulis to also account for products of +i/-i
129        let destabilizers = self.destabilizers();
130        let stabilizers = self.stabilizers();
131        let n = self.n_qubits;
132        let mut result = PhasedPauliWordNoHash::<T::Storage, T::BuildHasher>::new(n);
133        for (i, destab) in destabilizers.iter().enumerate() {
134            if destab.word.xbits[addr0] {
135                result *= &stabilizers[i];
136            }
137        }
138
139        // phase >= 2 means -Z eigenvalue → outcome |1⟩ (true)
140        debug_assert!(
141            result.phase == 0 || result.phase == 2,
142            "Measurement result cannot be imaginary!"
143        );
144        result.phase >= 2
145    }

147    pub(crate) fn update_tableau_according_to_outcome(
148        &mut self,
149        addr0: usize,
150        q_idx: usize,
151        outcome: bool,
152    ) where
153        <<T as Config>::Storage as BitView>::Store: PrimInt,
154    {
155        let n = self.n_qubits;
156        let (destabilizers, stabilizers) = self.data.split_at_mut(n);
157
158        // Clone g_q once before the loop
159        let g_q = stabilizers[q_idx].clone();
160
161        // Check if there are other stabilizers that anticommute with Z_addr0
162        // If so, replace with g_j = g_j * g_q
163        for i in 0..n {
164            if i == q_idx {
165                continue;
166            }
167            if stabilizers[i].word.xbits[addr0] {
168                // Stabilizer i also anticommutes, so multiply by g_q to eliminate
169                stabilizers[i] *= &g_q;
170            }
171            if destabilizers[i].word.xbits[addr0] {
172                destabilizers[i] *= &g_q;
173            }
174        }
175
176        // Update destabilizer q to be the old stabilizer q (before replacement)
177        destabilizers[q_idx] = g_q;
178
179        // Finally, replace g_q by ±Z
180        let stab_q = &mut stabilizers[q_idx];
181        stab_q.word.xbits = BitArray::ZERO;
182        stab_q.word.zbits = BitArray::ZERO;
183        stab_q.word.zbits.set(addr0, true);
184        stab_q.phase = if outcome { 2 } else { 0 };
185    }
186
187    /// Apply CZ to N pairs with constant offset: (base+i, base+offset+i) for i in 0..count.
188    /// All pairs must be in the same u64 word. This replaces N individual CZ calls
189    /// with a single word-level shift+XOR operation per row.
190    ///
191    /// # Panics
192    /// Debug-asserts that all bits are within the same word.
193    #[inline]
194    pub fn cz_block_pairs(&mut self, base: usize, offset: usize, count: usize)
195    where
196        <<T::Storage as BitView>::Store as TryFrom<usize>>::Error: Debug,
197        <T::Storage as BitView>::Store: PrimInt + TryFrom<usize>,
198    {
199        if count == 0 {
200            return;
201        }
202        let bits_per_word = std::mem::size_of::<<T::Storage as BitView>::Store>() * 8;
203        let base_bit = base % bits_per_word;
204        let word_idx = base / bits_per_word;
205
206        debug_assert_eq!(
207            (base + offset + count - 1) / bits_per_word,
208            word_idx,
209            "All CZ pairs must be in the same word"
210        );
211
212        let one = <T::Storage as BitView>::Store::one();
213        let zero = <T::Storage as BitView>::Store::zero();
214        let count_mask = if count >= bits_per_word {
215            !zero
216        } else {
217            (one << count) - one
218        };
219        let mask_c = count_mask << base_bit;
220        let mask_t = count_mask << (base_bit + offset);
221
222        self.data.iter_mut().for_each(|pw| {
223            let xp = pw.word.xbits.data.as_raw_mut_slice();
224            let zp = pw.word.zbits.data.as_raw_mut_slice();
225            let x = xp[word_idx];
226            let z = zp[word_idx];
227
228            // Phase computation (must use original z before update)
229            let xc = (x >> base_bit) & count_mask;
230            let xt = (x >> (base_bit + offset)) & count_mask;
231            let zc = (z >> base_bit) & count_mask;
232            let zt = (z >> (base_bit + offset)) & count_mask;
233            let phase_bits = xc & xt & (zc ^ zt);
234            pw.phase ^= ((phase_bits.count_ones() & 1) as u8) << 1;
235
236            // Z update: z[c] ^= x[t], z[t] ^= x[c]
237            let z_delta = ((x >> offset) & mask_c) | ((x << offset) & mask_t);
238            zp[word_idx] = z ^ z_delta;
239        });
240    }
241
242    /// Apply CZ to N pairs with constant offset across two different words.
243    /// Controls at (word_c, base_bit_c+i) and targets at (word_t, base_bit_t+i) for i in 0..count.
244    /// word_c and word_t must be different.
245    #[inline]
246    pub fn cz_block_pairs_cross_word(
247        &mut self,
248        word_c: usize,
249        base_bit_c: usize,
250        word_t: usize,
251        base_bit_t: usize,
252        count: usize,
253    ) where
254        <T::Storage as BitView>::Store: PrimInt,
255    {
256        if count == 0 {
257            return;
258        }
259        let one = <T::Storage as BitView>::Store::one();
260        let zero = <T::Storage as BitView>::Store::zero();
261        let bits_per_word = std::mem::size_of::<<T::Storage as BitView>::Store>() * 8;
262
263        debug_assert!(base_bit_c + count <= bits_per_word);
264        debug_assert!(base_bit_t + count <= bits_per_word);
265        debug_assert_ne!(word_c, word_t);
266
267        let count_mask = if count >= bits_per_word {
268            !zero
269        } else {
270            (one << count) - one
271        };
272
273        self.data.iter_mut().for_each(|pw| {
274            let xp = pw.word.xbits.data.as_raw_mut_slice();
275            let zp = pw.word.zbits.data.as_raw_mut_slice();
276
277            // Extract aligned bits (shifted to 0..count-1)
278            let xc = (xp[word_c] >> base_bit_c) & count_mask;
279            let xt = (xp[word_t] >> base_bit_t) & count_mask;
280            let zc = (zp[word_c] >> base_bit_c) & count_mask;
281            let zt = (zp[word_t] >> base_bit_t) & count_mask;
282
283            // Phase: x[c] & x[t] & (z[c] ^ z[t]) per pair
284            let phase_bits = xc & xt & (zc ^ zt);
285            pw.phase ^= ((phase_bits.count_ones() & 1) as u8) << 1;
286
287            // z[c] ^= x[t]: place target x-bits at control positions
288            zp[word_c] = zp[word_c] ^ (xt << base_bit_c);
289            // z[t] ^= x[c]: place control x-bits at target positions
290            zp[word_t] = zp[word_t] ^ (xc << base_bit_t);
291        });
292    }
293}

295const COMPLEX_PHASE_CONVERSION: [Complex64; 4] = [
296    Complex64 { re: 1.0, im: 0.0 },  // +1
297    Complex64 { re: 0.0, im: 1.0 },  // +i
298    Complex64 { re: -1.0, im: 0.0 }, // -1
299    Complex64 { re: 0.0, im: -1.0 }, // -i
300];

302/// Symplectic inner product of two tableau index values — the count of
303/// shared set bits, used in stabilizer phase calculations.
304#[inline]
305pub fn symplectic_inner<I>(alpha: I, beta: I) -> u32
306where
307    I: TableauIndex,
308{
309    (alpha & beta).count_ones()
310}
311
312/// Compute the phase contribution from destabilizer anticommutation and odd-phase masks.
313/// This is a pure function of its arguments (no self access needed), extracted to enable
314/// use in parallel contexts where borrowing self is not possible.
315#[inline]
316pub(crate) fn compute_phase_with_mask_static<I: TableauIndex>(
317    destab_anticomm_bits: I,
318    basis_index: I,
319    stab_anticomm_bits: I,
320    odd_phase_mask: I,
321) -> u8 {
322    let mut phase = (2 * symplectic_inner(destab_anticomm_bits, basis_index) as u8) % 4;
323    let active = basis_index & stab_anticomm_bits;
324    let parity = (active & odd_phase_mask).count_ones() % 2;
325    phase = (phase + 2 * parity as u8) % 4;
326    phase
327}

329/// Minimum number of coefficients before engaging rayon parallelism.
330/// Below this threshold, the sequential path is always used even with the `rayon` feature.
331/// Benchmarked: at 8K coefficients rayon has ~24% overhead; at 32K it's 35% faster.
332/// Set to 16384 to avoid regressions while capturing the large-coefficient wins.
333#[cfg(feature = "rayon")]
334pub(crate) const RAYON_COEFF_THRESHOLD: usize = 16384;
335
336/// Sequential accumulation of branch coefficients.
337fn branch_coefficients_seq<I, CoeffType>(
338    items: impl IntoIterator<Item = (Complex<CoeffType>, I)>,
339    stab_anticomm_bits: I,
340    destab_anticomm_bits: I,
341    odd_phase_mask: I,
342    phase_decomp: u8,
343    coefficient_factor: Complex<CoeffType>,
344    branch_factor: Complex<CoeffType>,
345) -> HashMap<I, Complex<CoeffType>>
346where
347    I: TableauIndex,
348    CoeffType: One + Zero + Clone + num::Num,
349    Complex<CoeffType>:
350        std::ops::Mul<Output = Complex<CoeffType>> + std::ops::AddAssign + From<Complex64> + Copy,
351{
352    let mut map: HashMap<I, Complex<CoeffType>> = HashMap::default();
353    for (coeff, idx) in items {
354        debug_assert!(
355            !(coeff.re == CoeffType::zero() && coeff.im == CoeffType::zero()),
356            "Coefficient should not be zero"
357        );
358        let branch_index = idx ^ stab_anticomm_bits;
359        let branch_phase_contribution = compute_phase_with_mask_static(
360            destab_anticomm_bits,
361            idx,
362            stab_anticomm_bits,
363            odd_phase_mask,
364        );
365        let branch_phase = (branch_phase_contribution + phase_decomp) % 4;
366        let phase_factor: Complex<CoeffType> =
367            COMPLEX_PHASE_CONVERSION[branch_phase as usize].into();
368        let branch_coefficient = phase_factor * coeff * branch_factor;
369        let nonbranch_coefficient = coeff * coefficient_factor;
370        *map.entry(branch_index).or_insert(Complex::zero()) += branch_coefficient;
371        *map.entry(idx).or_insert(Complex::zero()) += nonbranch_coefficient;
372    }
373    map
374}

376/// Accumulate branch coefficients. When the coefficient count exceeds
377/// `RAYON_COEFF_THRESHOLD`, uses parallel map/collect into a Vec followed
378/// by sequential accumulation. Below the threshold, falls back to sequential.
379#[cfg(feature = "rayon")]
380fn branch_coefficients_parallel<I, CoeffType>(
381    items: &[(Complex<CoeffType>, I)],
382    stab_anticomm_bits: I,
383    destab_anticomm_bits: I,
384    odd_phase_mask: I,
385    phase_decomp: u8,
386    coefficient_factor: Complex<CoeffType>,
387    branch_factor: Complex<CoeffType>,
388) -> HashMap<I, Complex<CoeffType>>
389where
390    I: TableauIndex + Send + Sync,
391    CoeffType: One + Zero + Clone + Send + Sync + num::Num,
392    Complex<CoeffType>:
393        std::ops::Mul<Output = Complex<CoeffType>> + std::ops::AddAssign + From<Complex64> + Copy,
394{
395    if items.len() >= RAYON_COEFF_THRESHOLD {
396        use rayon::prelude::*;
397
398        // Parallel phase: compute all (branch_idx, branch_coeff, idx, nonbranch_coeff) tuples.
399        // This is pure math with no shared mutable state.
400        let pairs: Vec<(I, Complex<CoeffType>, I, Complex<CoeffType>)> = items
401            .par_iter()
402            .map(|&(coeff, idx)| {
403                let branch_index = idx ^ stab_anticomm_bits;
404                let branch_phase_contribution = compute_phase_with_mask_static(
405                    destab_anticomm_bits,
406                    idx,
407                    stab_anticomm_bits,
408                    odd_phase_mask,
409                );
410                let branch_phase = (branch_phase_contribution + phase_decomp) % 4;
411                let phase_factor: Complex<CoeffType> =
412                    COMPLEX_PHASE_CONVERSION[branch_phase as usize].into();
413                (
414                    branch_index,
415                    phase_factor * coeff * branch_factor,
416                    idx,
417                    coeff * coefficient_factor,
418                )
419            })
420            .collect();
421
422        // Sequential phase: accumulate into a pre-sized HashMap.
423        // HashMap inserts dominate the cost but benefit from cache locality.
424        let mut map: HashMap<I, Complex<CoeffType>> =
425            HashMap::with_capacity_and_hasher(2 * pairs.len(), Default::default());
426        for (branch_idx, branch_coeff, idx, nonbranch_coeff) in pairs {
427            *map.entry(branch_idx).or_insert(Complex::zero()) += branch_coeff;
428            *map.entry(idx).or_insert(Complex::zero()) += nonbranch_coeff;
429        }
430        return map;
431    }
432
433    branch_coefficients_seq(
434        items.iter().copied(),
435        stab_anticomm_bits,
436        destab_anticomm_bits,
437        odd_phase_mask,
438        phase_decomp,
439        coefficient_factor,
440        branch_factor,
441    )
442}

444/// Sequential accumulation of apply coefficients.
445fn apply_coefficients_seq<I, CoeffType>(
446    items: impl IntoIterator<Item = (Complex<CoeffType>, I)>,
447    stab_anticomm_bits: I,
448    destab_anticomm_bits: I,
449    odd_phase_mask: I,
450    phase_decomp: u8,
451) -> HashMap<I, Complex<CoeffType>>
452where
453    I: TableauIndex,
454    CoeffType: One + Zero + Clone + num::Num,
455    Complex<CoeffType>:
456        std::ops::Mul<Output = Complex<CoeffType>> + std::ops::AddAssign + From<Complex64> + Copy,
457{
458    let mut map: HashMap<I, Complex<CoeffType>> = HashMap::default();
459    for (coeff, idx) in items {
460        debug_assert!(
461            !(coeff.re == CoeffType::zero() && coeff.im == CoeffType::zero()),
462            "Coefficient should not be zero"
463        );
464        let branch_index = idx ^ stab_anticomm_bits;
465        let branch_phase_contribution = compute_phase_with_mask_static(
466            destab_anticomm_bits,
467            idx,
468            stab_anticomm_bits,
469            odd_phase_mask,
470        );
471        let branch_phase = (branch_phase_contribution + phase_decomp) % 4;
472        let phase_factor: Complex<CoeffType> =
473            COMPLEX_PHASE_CONVERSION[branch_phase as usize].into();
474        let branch_coefficient = phase_factor * coeff;
475        *map.entry(branch_index).or_insert(Complex::zero()) += branch_coefficient;
476    }
477    map
478}

480/// Accumulate coefficients for pauli application. When the coefficient count
481/// exceeds `RAYON_COEFF_THRESHOLD`, uses parallel map/collect followed by
482/// sequential accumulation. Below the threshold, falls back to sequential.
483#[cfg(feature = "rayon")]
484fn apply_coefficients_parallel<I, CoeffType>(
485    items: &[(Complex<CoeffType>, I)],
486    stab_anticomm_bits: I,
487    destab_anticomm_bits: I,
488    odd_phase_mask: I,
489    phase_decomp: u8,
490) -> HashMap<I, Complex<CoeffType>>
491where
492    I: TableauIndex + Send + Sync,
493    CoeffType: One + Zero + Clone + Send + Sync + num::Num,
494    Complex<CoeffType>:
495        std::ops::Mul<Output = Complex<CoeffType>> + std::ops::AddAssign + From<Complex64> + Copy,
496{
497    if items.len() >= RAYON_COEFF_THRESHOLD {
498        use rayon::prelude::*;
499
500        let pairs: Vec<(I, Complex<CoeffType>)> = items
501            .par_iter()
502            .map(|&(coeff, idx)| {
503                let branch_index = idx ^ stab_anticomm_bits;
504                let branch_phase_contribution = compute_phase_with_mask_static(
505                    destab_anticomm_bits,
506                    idx,
507                    stab_anticomm_bits,
508                    odd_phase_mask,
509                );
510                let branch_phase = (branch_phase_contribution + phase_decomp) % 4;
511                let phase_factor: Complex<CoeffType> =
512                    COMPLEX_PHASE_CONVERSION[branch_phase as usize].into();
513                (branch_index, phase_factor * coeff)
514            })
515            .collect();
516
517        let mut map: HashMap<I, Complex<CoeffType>> =
518            HashMap::with_capacity_and_hasher(pairs.len(), Default::default());
519        for (branch_idx, branch_coeff) in pairs {
520            *map.entry(branch_idx).or_insert(Complex::zero()) += branch_coeff;
521        }
522        return map;
523    }
524
525    apply_coefficients_seq(
526        items.iter().copied(),
527        stab_anticomm_bits,
528        destab_anticomm_bits,
529        odd_phase_mask,
530        phase_decomp,
531    )
532}

534/// A [`Tableau`] extended with sparse coefficient tracking to handle
535/// non-Clifford gates.
536///
537/// Non-Clifford gates (T, rotations) split a single tableau into a sum
538/// of weighted branches indexed by bitstrings. `GeneralizedTableau`
539/// stores those weights in a [`SparseVector`] keyed by an
540/// [`IndexType`](TableauIndex). Choose:
541/// * `IndexType = usize` for up to 64 qubits,
542/// * `IndexType = u128` for up to 128,
543/// * `IndexType = bnum::types::U256` and friends for the very wide
544///   regime.
545///
546/// Per-qubit loss is tracked in [`is_lost`](GeneralizedTableau::is_lost);
547/// gates respect it automatically.
548///
549/// # Examples
550///
551/// Prepare a Bell pair and sample one shot. With a fixed seed the two
552/// measurements are perfectly correlated on every shot:
553///
554/// ```
555/// use ppvm_runtime::config::fxhash::ByteF64;
556/// use ppvm_runtime::traits::{Clifford, LossyMeasure};
557/// use ppvm_tableau::data::GeneralizedTableau;
558///
559/// let mut tab: GeneralizedTableau<ByteF64<1>> =
560///     GeneralizedTableau::new_with_seed(2, 1e-12, 0);
561/// tab.h(0);
562/// tab.cnot(0, 1);
563///
564/// let r0 = LossyMeasure::measure(&mut tab, 0);
565/// let r1 = LossyMeasure::measure(&mut tab, 1);
566/// assert_eq!(r0, r1);
567/// ```
568///
569/// Non-Clifford gates work through the same interface — apply a `T` gate
570/// followed by `T†` and the state is unchanged:
571///
572/// ```
573/// use ppvm_runtime::config::fxhash::ByteF64;
574/// use ppvm_runtime::traits::{Clifford, TGate};
575/// use ppvm_tableau::data::GeneralizedTableau;
576///
577/// let mut tab: GeneralizedTableau<ByteF64<1>> =
578///     GeneralizedTableau::new_with_seed(1, 1e-12, 0);
579/// tab.h(0);
580/// tab.t(0);
581/// tab.t_adj(0);
582/// // T followed by T† is the identity; the |+⟩ state is restored.
583/// ```
584#[derive(Clone)]
585pub struct GeneralizedTableau<
586    T: Config,
587    IndexType = usize,
588    SparseVectorType: SparseVector<Complex<T::Coeff>, IndexType> = Vec<(Complex64, IndexType)>,
589> {
590    /// Underlying Clifford tableau.
591    pub tableau: Tableau<T>,
592    /// Sparse coefficient vector indexed by bitstrings.
593    pub coefficients: SparseVectorType,
594    /// Per-qubit loss flags.
595    pub is_lost: Vec<bool>,
596    /// Coefficient-magnitude threshold below which branches are dropped.
597    pub coefficient_threshold: T::Coeff,
598    _index_phantom: PhantomData<IndexType>,
599}
600
601impl<T: Config, I, C: SparseVector<Complex<T::Coeff>, I>> GeneralizedTableau<T, I, C>
602where
603    T::Coeff: One + Zero + Clone + num::Num,
604    Complex<T::Coeff>: std::ops::Mul<Output = Complex<T::Coeff>>
605        + std::ops::AddAssign
606        + From<Complex64>
607        + ComplexFloat
608        + Copy,
609    I: TableauIndex,
610{
611    /// Construct a generalized tableau in the `|0…0⟩` state.
612    ///
613    /// Branches whose coefficient magnitude falls below
614    /// `coefficient_threshold` are dropped during gate application.
615    pub fn new(n_qubits: usize, coefficient_threshold: T::Coeff) -> Self {
616        let mut coefficients = C::new();
617        let complex_one = Complex {
618            re: T::Coeff::one(),
619            im: T::Coeff::zero(),
620        };
621        coefficients.unsafe_insert(I::zero(), complex_one);
622        Self {
623            tableau: Tableau::new(n_qubits),
624            coefficients,
625            is_lost: vec![false; n_qubits],
626            coefficient_threshold,
627            _index_phantom: PhantomData,
628        }
629    }
630
631    /// Same as [`GeneralizedTableau::new`], but seed the RNG deterministically.
632    pub fn new_with_seed(n_qubits: usize, coefficient_threshold: T::Coeff, seed: u64) -> Self {
633        let mut s = Self::new(n_qubits, coefficient_threshold);
634        s.tableau.rng = SmallRng::seed_from_u64(seed);
635        s
636    }
637
638    /// Clone the quantum state but reinitialize the RNG, producing an independent simulation
639    /// branch. If `seed` is `Some`, the new RNG is seeded deterministically; if `None`, it is
640    /// seeded from OS entropy.
641    pub fn fork(&self, seed: Option<u64>) -> Self {
642        let mut cloned = self.clone();
643        cloned.tableau.rng = match seed {
644            Some(s) => SmallRng::seed_from_u64(s),
645            None => rand::make_rng(),
646        };
647        cloned
648    }
649
650    /// Number of qubits.
651    pub fn n_qubits(&self) -> usize {
652        self.tableau.n_qubits
653    }
654
655    /// Apply CZ to N pairs with constant offset: (base+i, base+offset+i) for i in 0..count.
656    /// Falls back to individual CZ calls if any qubit in the range is lost.
657    pub fn cz_block_pairs(&mut self, base: usize, offset: usize, count: usize)
658    where
659        <<T::Storage as BitView>::Store as TryFrom<usize>>::Error: Debug,
660        <T::Storage as BitView>::Store: PrimInt + TryFrom<usize>,
661    {
662        // Check if any qubit in the range is lost
663        let any_lost =
664            (0..count).any(|i| self.is_lost[base + i] || self.is_lost[base + offset + i]);
665        if !any_lost {
666            self.tableau.cz_block_pairs(base, offset, count);
667        } else {
668            // Fallback to individual CZ calls
669            for i in 0..count {
670                let c = base + i;
671                let t = base + offset + i;
672                if !self.is_lost[c] && !self.is_lost[t] {
673                    Clifford::cz(&mut self.tableau, c, t);
674                }
675            }
676        }
677    }
678
679    /// Apply CZ to N cross-word pairs. Controls at word_c, targets at word_t.
680    /// Falls back to individual CZ calls if any qubit is lost.
681    pub fn cz_block_pairs_cross_word(
682        &mut self,
683        word_c: usize,
684        base_bit_c: usize,
685        word_t: usize,
686        base_bit_t: usize,
687        count: usize,
688    ) where
689        <T::Storage as BitView>::Store: PrimInt,
690    {
691        let bits_per_word = std::mem::size_of::<<T::Storage as BitView>::Store>() * 8;
692        let any_lost = (0..count).any(|i| {
693            let c = word_c * bits_per_word + base_bit_c + i;
694            let t = word_t * bits_per_word + base_bit_t + i;
695            self.is_lost[c] || self.is_lost[t]
696        });
697        if !any_lost {
698            self.tableau
699                .cz_block_pairs_cross_word(word_c, base_bit_c, word_t, base_bit_t, count);
700        } else {
701            for i in 0..count {
702                let c = word_c * bits_per_word + base_bit_c + i;
703                let t = word_t * bits_per_word + base_bit_t + i;
704                if !self.is_lost[c] && !self.is_lost[t] {
705                    Clifford::cz(&mut self.tableau, c, t);
706                }
707            }
708        }
709    }

711    // helper functions

713    /// Compute the decomposition of a pauli into stabilizer destabilizer products
714    /// Any Pauli can be written as P_addr0 = phase * prod(d_k ^ gamma_k) * prod(s_l ^ lambda_l)
715    /// where: gamma_k == 1 iff {P_addr0, s_k} = 0
716    /// lambda_l == 1 iff {P_addr0, d_l} = 0
717    /// Lemma 5. from T. J. Yoder (2012)
718    /// NOTE: this is O(n^2)
719    ///
720    /// The function returns `(phase, stab_anticomm_bits, destab_anticomm_bits)`, where
721    /// `stab_anticomm_bits[k] = 1` iff P_addr0 anticommutes with stabilizer s_k, and
722    /// `destab_anticomm_bits[l] = 1` iff P_addr0 anticommutes with destabilizer d_l.
723    /// Note that stab_anticomm_bits is equal to the shift of the index when branching
724    /// (`beta` in Eq(4) of the SOFT paper).
725    pub fn compute_decomposition(&self, addr0: usize, pauli: Pauli) -> (u8, I, I)
726    where
727        <<T as Config>::Storage as BitView>::Store: PrimInt,
728    {
729        let n = self.n_qubits();
730
731        // the actual decomposition, which we need to track the phase
732        let mut p_word = PhasedPauliWordNoHash::<T::Storage, T::BuildHasher>::new(n);
733        p_word.set(addr0, pauli);
734
735        // the bit strings defining the contributions
736        let mut destab_anticomm_bits = I::zero();
737        let mut stab_anticomm_bits = I::zero();
738
739        debug_assert_ne!(pauli, Pauli::I);
740        let pauli_bits = match pauli {
741            Pauli::I => (false, false),
742            Pauli::X => (true, false),
743            Pauli::Y => (true, true),
744            Pauli::Z => (false, true),
745            _ => unreachable!("Pauli L cannot occur in tableau"),
746        };
747
748        let stabilizers = self.tableau.stabilizers();
749        let destabilizers = self.tableau.destabilizers();
750        let one = I::one();
751
752        for (i, stab) in stabilizers.iter().enumerate() {
753            if !destabilizers[i].word.anticommutes_at(addr0, pauli_bits) {
754                // commutes
755                continue;
756            }
757
758            // contributes, so set the corresponding bit in destab_anticomm_bits to 1
759            destab_anticomm_bits |= one << i;
760
761            // destabilizer anti-commutes, so the stabilizer contributes
762            // the stabilizer is its own inverse up to its phase
763            // to avoid inverting the stabilizer, we just multiply by it
764            // and then divide by its phase squared
765            p_word *= stab;
766            p_word.add_phase(8 - 2 * stab.phase);
767        }
768
769        // NOTE: destabilizers also commute with one another in a valid tableau
770        // since the form a basis together with stabilizers
771        for (i, destab) in destabilizers.iter().enumerate() {
772            if !stabilizers[i].word.anticommutes_at(addr0, pauli_bits) {
773                // commutes
774                continue;
775            }
776
777            // contributes, so set the corresponding bit in stab_anticomm_bits to 1
778            stab_anticomm_bits |= one << i;
779
780            // stabilizer anti-commutes, so the destabilizer contributes
781            p_word *= destab;
782            p_word.add_phase(8 - 2 * destab.phase);
783        }
784
785        (p_word.phase, stab_anticomm_bits, destab_anticomm_bits)
786    }
787
788    /// every basis index is a bit string alpha defining the basis state
789    /// the phase when applying a Pauli is the product of all destabilizer phases
790    /// and the phase contributions from the commutation relations
791    /// we need to check every destabilizer where the basis index has a 1 bit.
792    #[allow(dead_code)]
793    pub(crate) fn compute_phase(
794        &self,
795        destab_anticomm_bits: I,
796        basis_index: I,
797        stab_anticomm_bits: I,
798    ) -> u8 {
799        // phase convention: 0: +1, 1: +i, 2: -1, 3: -i
800        let one = I::one();
801        let zero = I::zero();
802
803        // contribution 1: each destabilizer D_i with basis_index[i]=1 that anticommutes
804        // with P (destab_anticomm_bits[i]=1) contributes a -1 sign; this is the symplectic inner product
805        let mut phase = (2 * symplectic_inner(destab_anticomm_bits, basis_index) as u8) % 4;
806
807        // contribution 2: destabilizers that appear twice (basis_index[i]=1 and stab_anticomm_bits[i]=1)
808        // contribute an extra -1 if their phase is imaginary
809        let active = basis_index & stab_anticomm_bits;
810        for (i, destab) in self.tableau.destabilizers().iter().enumerate() {
811            if active & (one << i) == zero {
812                continue;
813            }
814            if destab.phase % 2 != 0 {
815                phase = (phase + 2) % 4;
816            }
817        }
818
819        phase
820    }
821
822    /// Build a bitmask where bit i is set if destabilizer i has odd phase (phase % 2 != 0).
823    pub fn odd_phase_destabilizer_mask(&self) -> I {
824        let mut mask = I::zero();
825        let one = I::one();
826        for (i, destab) in self.tableau.destabilizers().iter().enumerate() {
827            if destab.phase % 2 != 0 {
828                mask |= one << i;
829            }
830        }
831        mask
832    }
833}

835impl<T: Config, I, C: SparseVector<Complex<T::Coeff>, I>> GeneralizedTableau<T, I, C>
836where
837    T::Coeff: One + Zero + Clone + Send + Sync + num::Num + PartialOrd,
838    Complex<T::Coeff>: std::ops::Mul<Output = Complex<T::Coeff>>
839        + std::ops::AddAssign
840        + From<Complex64>
841        + ComplexFloat
842        + Copy,
843    I: TableauIndex + Send + Sync,
844{
845    pub(crate) fn branch_with_coefficients(
846        &mut self,
847        addr0: usize,
848        pauli: Pauli,
849        coefficient_factor: Complex<T::Coeff>,
850        branch_factor: Complex<T::Coeff>,
851    ) where
852        <<T as Config>::Storage as BitView>::Store: PrimInt,
853    {
854        if self.is_lost[addr0] {
855            return;
856        }
857
858        let (phase_decomp, stab_anticomm_bits, destab_anticomm_bits) =
859            self.compute_decomposition(addr0, pauli);
860
861        let odd_phase_mask = self.odd_phase_destabilizer_mask();
862        let old_coefficients = std::mem::replace(&mut self.coefficients, C::new());
863        #[cfg(feature = "rayon")]
864        let n_coefficients = old_coefficients.len();
865
866        // When rayon is enabled and above the threshold, collect to Vec for parallel map;
867        // otherwise iterate directly with the sequential path.
868        #[cfg(feature = "rayon")]
869        let new_coefficients = if n_coefficients >= RAYON_COEFF_THRESHOLD {
870            let items: Vec<_> = old_coefficients.into_iter().collect();
871            branch_coefficients_parallel(
872                &items,
873                stab_anticomm_bits,
874                destab_anticomm_bits,
875                odd_phase_mask,
876                phase_decomp,
877                coefficient_factor,
878                branch_factor,
879            )
880        } else {
881            branch_coefficients_seq(
882                old_coefficients,
883                stab_anticomm_bits,
884                destab_anticomm_bits,
885                odd_phase_mask,
886                phase_decomp,
887                coefficient_factor,
888                branch_factor,
889            )
890        };
891
892        #[cfg(not(feature = "rayon"))]
893        let new_coefficients = branch_coefficients_seq(
894            old_coefficients,
895            stab_anticomm_bits,
896            destab_anticomm_bits,
897            odd_phase_mask,
898            phase_decomp,
899            coefficient_factor,
900            branch_factor,
901        );
902
903        let cutoff_sq = self.coefficient_threshold.clone() * self.coefficient_threshold.clone();
904        for (idx, coeff) in new_coefficients {
905            if coeff.norm_sqr() > cutoff_sq {
906                self.coefficients.unsafe_insert(idx, coeff);
907            }
908        }
909    }

911    pub(crate) fn compute_coefficients_after_pauli_apply(
912        &self,
913        coefficients: &mut C,
914        addr0: usize,
915        pauli: Pauli,
916    ) where
917        <<T as Config>::Storage as BitView>::Store: PrimInt,
918    {
919        if self.is_lost[addr0] {
920            return;
921        }
922
923        let (phase_decomp, stab_anticomm_bits, destab_anticomm_bits) =
924            self.compute_decomposition(addr0, pauli);
925
926        let odd_phase_mask = self.odd_phase_destabilizer_mask();
927        #[cfg(feature = "rayon")]
928        let n_coefficients = coefficients.len();
929        let old_coefficients = std::mem::replace(coefficients, C::new());
930
931        #[cfg(feature = "rayon")]
932        let new_coefficients = if n_coefficients >= RAYON_COEFF_THRESHOLD {
933            let items: Vec<_> = old_coefficients.into_iter().collect();
934            apply_coefficients_parallel(
935                &items,
936                stab_anticomm_bits,
937                destab_anticomm_bits,
938                odd_phase_mask,
939                phase_decomp,
940            )
941        } else {
942            apply_coefficients_seq(
943                old_coefficients,
944                stab_anticomm_bits,
945                destab_anticomm_bits,
946                odd_phase_mask,
947                phase_decomp,
948            )
949        };
950
951        #[cfg(not(feature = "rayon"))]
952        let new_coefficients = apply_coefficients_seq(
953            old_coefficients,
954            stab_anticomm_bits,
955            destab_anticomm_bits,
956            odd_phase_mask,
957            phase_decomp,
958        );
959
960        let cutoff_sq = self.coefficient_threshold.clone() * self.coefficient_threshold.clone();
961        for (idx, coeff) in new_coefficients {
962            if coeff.norm_sqr() > cutoff_sq {
963                coefficients.unsafe_insert(idx, coeff);
964            }
965        }
966    }
