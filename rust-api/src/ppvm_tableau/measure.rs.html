14impl<T: Config> Measure for Tableau<T>
15where
16    <<T as Config>::Storage as BitView>::Store: PrimInt,
17{
18    /// Measure qubit `addr0` in Z basis
19    fn measure(&mut self, addr0: usize) -> bool {
20        let q = self.find_z_anticommuting_stabilizer(addr0);
21        match q {
22            Some(q_idx) => {
23                // Case a: random measurement outcome
24                // At least one stabilizer anticommutes with Z_addr0
25
26                // Generate random measurement outcome (50/50)
27                let outcome = self.rng.random::<bool>();
28
29                self.update_tableau_according_to_outcome(addr0, q_idx, outcome);
30
31                outcome
32            }
33            None => {
34                // Case b: deterministic measurement outcome
35
36                self.get_deterministic_outcome(addr0)
37            }
38        }
39    }
40}

42const COMPLEX_PHASE_CONVERSION: [Complex64; 4] = [
43    Complex64::new(1.0, 0.0),  // +1
44    Complex64::new(0.0, 1.0),  // +i
45    Complex64::new(-1.0, 0.0), // -1
46    Complex64::new(0.0, -1.0), // -i
47];

49/// Per-measurement scratch buffers, reused across qubits within a single
50/// `measure_all` invocation — and, when threaded through
51/// [`measure_all_with_scratch`](GeneralizedTableau::measure_all_with_scratch),
52/// across many shots of a sampler too. Reusing one scratch keeps the case-a
53/// HashMap and the b-entries Vec out of the per-shot allocator churn.
54///
55/// - `odd_phase_mask` is lazily computed and cached until the destabilizers
56///   change (i.e. until a case-a measurement runs `update_tableau_according_to_outcome`).
57/// - `coeff_map` is the case-a HashMap holding `(idx → amplitude)` between
58///   the overlap, partition, and merge passes.
59/// - `b_entries` is the case-a partition's "k-bit = 1" scratch Vec.
60///
61/// Construct one per active sampling thread; the type is not meant to be
62/// shared across threads concurrently.
63#[derive(Clone)]
64pub struct MeasureScratch<I, R> {
65    pub odd_phase_mask: Option<I>,
66    pub coeff_map: HashMap<I, Complex<R>>,
67    pub b_entries: Vec<(I, Complex<R>)>,
68}
69
70impl<I, R> MeasureScratch<I, R> {
71    pub fn new() -> Self {
72        Self {
73            odd_phase_mask: None,
74            coeff_map: HashMap::default(),
75            b_entries: Vec::new(),
76        }
77    }
78}
79
80impl<I, R> Default for MeasureScratch<I, R> {
81    fn default() -> Self {
82        Self::new()
83    }

86impl<T: Config, I, C: SparseVector<Complex<T::Coeff>, I>> LossyMeasure
87    for GeneralizedTableau<T, I, C>
88where
89    T: Config,
90    <<T as Config>::Storage as BitView>::Store: PrimInt,
91    C: SparseVector<Complex<T::Coeff>, I> + std::fmt::Debug,
92    T::Coeff: One
93        + Zero
94        + Clone
95        + num::Num
96        + ToPrimitive
97        + std::fmt::Debug
98        + std::ops::Mul<f64>
99        + PartialOrd<f64>,
100    Complex<T::Coeff>: std::ops::Mul<Output = Complex<T::Coeff>>
101        + From<Complex64>
102        + std::ops::MulAssign
103        + std::ops::AddAssign
104        + One
105        + ComplexFloat
106        + Copy,
107    I: TableauIndex + Debug,
108{
109    fn measure(&mut self, addr0: usize) -> Option<bool> {
110        if self.is_lost[addr0] {
111            return None;
112        }
113
114        let (phase_decomp, stab_anticomm_bits, destab_anticomm_bits) =
115            self.compute_decomposition(addr0, Pauli::Z);
116
117        // Standalone callers don't get cross-call cache benefits; `measure_all`
118        // threads through a long-lived scratch.
119        let mut scratch = MeasureScratch::new();
120        self.measure_with_scratch(
121            addr0,
122            &mut scratch,
123            phase_decomp,
124            stab_anticomm_bits,
125            destab_anticomm_bits,
126        )
127    }
128}

130impl<T: Config, I, C: SparseVector<Complex<T::Coeff>, I>> GeneralizedTableau<T, I, C>
131where
132    T: Config,
133    <<T as Config>::Storage as BitView>::Store: PrimInt,
134    C: SparseVector<Complex<T::Coeff>, I> + std::fmt::Debug,
135    T::Coeff: One
136        + Zero
137        + Clone
138        + num::Num
139        + ToPrimitive
140        + std::fmt::Debug
141        + std::ops::Mul<f64>
142        + PartialOrd<f64>,
143    Complex<T::Coeff>: std::ops::Mul<Output = Complex<T::Coeff>>
144        + From<Complex64>
145        + std::ops::MulAssign
146        + std::ops::AddAssign
147        + One
148        + ComplexFloat
149        + Copy,
150    I: TableauIndex + Debug,
151{
152    pub(crate) fn measure_with_scratch(
153        &mut self,
154        addr0: usize,
155        scratch: &mut MeasureScratch<I, T::Coeff>,
156        phase_decomp: u8,
157        stab_anticomm_bits: I,
158        destab_anticomm_bits: I,
159    ) -> Option<bool> {
160        if stab_anticomm_bits == I::zero() {
161            // Case b (fast path): Z is already a stabilizer.
162            // branch_index = idx ^ 0 = idx, so the overlap is self-pairing:
163            //   z_overlap = sum_alpha phase_factor(alpha).conj() * |c_alpha|^2
164            // No HashMap needed — work directly on the Vec.
165
166            // Collect into a Vec so we can iterate twice: once for overlap,
167            // once for filtering. The collect from Vec IntoIter→Vec is a no-op.
168            let entries: Vec<(Complex<T::Coeff>, I)> =
169                std::mem::replace(&mut self.coefficients, C::new())
170                    .into_iter()
171                    .collect();
172
173            // Pass 1: compute overlap (read-only, real-only accumulation)
174            // Since conj(c)*c = |c|^2 (always real), the phase factor contribution
175            // to z_overlap.re is: phase 0 → +|c|^2, phase 2 → −|c|^2,
176            // phase 1,3 → 0 (imaginary × real = imaginary, doesn't contribute to .re)
177            let z_overlap_re =
178                Self::compute_overlap_case_b(&entries, phase_decomp, destab_anticomm_bits);
179
180            let prob_1 = 0.5 - 0.5 * z_overlap_re;
181            let outcome = self.tableau.rng.random::<f64>() < prob_1;
182
183            debug_assert!(
184                phase_decomp == 0 || phase_decomp == 2,
185                "Measurement result cannot be imaginary!"
186            );
187
188            self.project_case_b(&entries, outcome, phase_decomp, destab_anticomm_bits);
189
190            // Case-b doesn't mutate destabilizers, so the cached mask remains valid.
191            Some(outcome)
192        } else {
193            // Case a: Z is not a stabilizer — need HashMap for cross-index lookups.
194            // Drain self.coefficients into scratch.coeff_map via `retain` so the
195            // Vec's capacity survives and we can refill it at the end without a
196            // fresh allocation.
197            scratch.coeff_map.clear();
198            scratch.coeff_map.reserve(self.coefficients.len());
199            {
200                let coeff_map = &mut scratch.coeff_map;
201                self.coefficients.retain(|(v, i)| {
202                    coeff_map.insert(*i, *v);
203                    false // drain — keeps allocation
204                });
205            }
206
207            // Compute z_overlap.re directly (the imaginary part is always ~0).
208            // The mask is a pure function of destabilizer phases — cache it across
209            // measurements until `update_tableau_according_to_outcome` invalidates it.
210            let odd_phase_mask = *scratch
211                .odd_phase_mask
212                .get_or_insert_with(|| self.odd_phase_destabilizer_mask());
213            let z_overlap_re = Self::compute_overlap_case_a(
214                &scratch.coeff_map,
215                phase_decomp,
216                destab_anticomm_bits,
217                stab_anticomm_bits,
218                odd_phase_mask,
219            );
220
221            let prob_1 = 0.5 - 0.5 * z_overlap_re;
222            let outcome = self.tableau.rng.random::<f64>() < prob_1;
223            self.project_case_a(
224                outcome,
225                scratch,
226                phase_decomp,
227                stab_anticomm_bits,
228                destab_anticomm_bits,
229                addr0,
230            );
231            Some(outcome)
232        }
233    }

235    pub fn project_case_a(
236        &mut self,
237        outcome: bool,
238        scratch: &mut MeasureScratch<I, T::Coeff>,
239        phase_decomp: u8,
240        stab_anticomm_bits: I,
241        destab_anticomm_bits: I,
242        addr0: usize,
243    ) {
244        // Case a: Z is not a stabilizer — need HashMap for cross-index lookups.
245        // Drain self.coefficients into scratch.coeff_map via `retain` so the
246        // Vec's capacity survives and we can refill it at the end without a
247        // fresh allocation.
248
249        let q_idx = stab_anticomm_bits.trailing_zeros() as usize;
250
251        let one = I::one();
252        let zero = I::zero();
253        let k = one << q_idx;
254
255        let alpha = if outcome {
256            (phase_decomp + 2) % 4
257        } else {
258            phase_decomp
259        };
260
261        // Partition into A (k-bit=0) and B (k-bit=1) via retain, then merge.
262        // Split the borrow so `retain` can mutate coeff_map while the closure
263        // pushes into b_entries.
264        scratch.b_entries.clear();
265        let MeasureScratch {
266            coeff_map,
267            b_entries,
268            ..
269        } = scratch;
270        b_entries.reserve(coeff_map.len() / 2 + 1);
271        coeff_map.retain(|idx, coeff| {
272            if (*idx & k) != zero {
273                b_entries.push((*idx, *coeff));
274                false // remove B entry
275            } else {
276                true // keep A entry
277            }
278        });
279        // Merge B entries into their A partners with phase adjustment.
280        Self::merge_b_into_a(
281            coeff_map,
282            b_entries,
283            alpha,
284            destab_anticomm_bits,
285            stab_anticomm_bits,
286        );
287
288        // Keep entries where |c|/norm > threshold.
289        let norm_sqr = coeff_map
290            .values()
291            .fold(T::Coeff::zero(), |acc, c: &Complex<T::Coeff>| {
292                acc + c.norm_sqr()
293            });
294
295        let cutoff_sq = self.coefficient_threshold.clone() * self.coefficient_threshold.clone();
296        let threshold = cutoff_sq.to_f64().unwrap_or(0.0) * norm_sqr.to_f64().unwrap_or(0.0);
297        // self.coefficients is already empty here (drained via retain above);
298        // reserve is mostly a no-op since the prior capacity is still there.
299        self.coefficients.reserve(coeff_map.len());
300        for (idx, coeff) in coeff_map.drain() {
301            if coeff.norm_sqr() > threshold {
302                self.coefficients.unsafe_insert(idx, coeff);
303            }
304        }
305
306        self.coefficients.normalize();
307
308        self.tableau
309            .update_tableau_according_to_outcome(addr0, q_idx, outcome);
310        // Destabilizer phases just changed, invalidate the cached mask.
311        scratch.odd_phase_mask = None;
312    }

314    /// project state in case b (Z is a stabilizer) according to sampled outcome
315    pub fn project_case_b(
316        &mut self,
317        entries: &[(Complex<T::Coeff>, I)],
318        outcome: bool,
319        phase_decomp: u8,
320        destab_anticomm_bits: I,
321    ) {
322        let old_len = entries.len();
323
324        let z_sign = phase_decomp == 2;
325
326        // Pass 2: filter directly into self.coefficients (no retain needed)
327        self.coefficients.reserve(entries.len());
328        for &(coeff, alpha) in entries {
329            let parity = symplectic_inner(alpha, destab_anticomm_bits) % 2 != 0;
330            if (parity ^ outcome) == z_sign {
331                self.coefficients.unsafe_insert(alpha, coeff);
332            }
333        }
334
335        if self.coefficients.len() < old_len {
336            self.coefficients.normalize();
337        }
338    }
339}

341impl<T, I, C> GeneralizedTableau<T, I, C>
342where
343    T: Config,

345    C: SparseVector<Complex<T::Coeff>, I> + std::fmt::Debug,
346    T::Coeff: One

352        + std::ops::Mul<f64>
353        + PartialOrd<f64>,
354    Complex<T::Coeff>: std::ops::Mul<Output = Complex<T::Coeff>>
355        + From<Complex64>
356        + std::ops::MulAssign
357        + std::ops::AddAssign
358        + One

362{
363    /// Measure qubit `addr0` in Z basis with optional readout noise.
364    ///
365    /// Behaves like [`measure`](LossyMeasure::measure), then with probability
366    /// `flip_prob` flips the *recorded* bit. The qubit's quantum state stays
367    /// consistent with the true outcome — only the returned value flips.
368    /// `flip_prob = 0.0` is equivalent to `measure`.
369    ///
370    /// If the qubit is lost, returns `None` regardless of `flip_prob`.
371    pub fn measure_noisy(&mut self, addr0: usize, flip_prob: f64) -> Option<bool> {
372        debug_assert!(
373            (0.0..=1.0).contains(&flip_prob),
374            "flip_prob must be in [0, 1], got {flip_prob}"
375        );
376        let outcome = self.measure(addr0)?;
377        Some(self.flip_with_prob(outcome, flip_prob))
378    }
379
380    /// Sample a Bernoulli(`p`) outcome using the tableau's internal RNG.
381    /// Used by Stim measurement-noise dispatch in `ppvm-stim`.
382    pub fn bernoulli(&mut self, p: f64) -> bool {
383        debug_assert!((0.0..=1.0).contains(&p), "p must be in [0, 1], got {p}");
384        self.tableau.rng.random::<f64>() < p
385    }
386
387    /// Flip `bit` with probability `p`. Used by Stim MR/MPad readout-noise
388    /// dispatch in `ppvm-stim`. Returns `bit` unchanged when `p <= 0.0`.
389    pub fn flip_with_prob(&mut self, bit: bool, p: f64) -> bool {
390        debug_assert!((0.0..=1.0).contains(&p), "p must be in [0, 1], got {p}");
391        if p > 0.0 && self.bernoulli(p) {
392            !bit
393        } else {
394            bit
395        }
396    }
397}

399/// Measurement overlap helper functions, with optional rayon parallelism.
400impl<T: Config, I, C: SparseVector<Complex<T::Coeff>, I>> GeneralizedTableau<T, I, C>
401where
402    T: Config,
403    <<T as Config>::Storage as BitView>::Store: PrimInt,
404    C: SparseVector<Complex<T::Coeff>, I> + std::fmt::Debug,
405    T::Coeff: One
406        + Zero
407        + Clone
408        + num::Num
409        + ToPrimitive
410        + std::fmt::Debug
411        + std::ops::Mul<f64>
412        + PartialOrd<f64>,
413    Complex<T::Coeff>: std::ops::Mul<Output = Complex<T::Coeff>>
414        + From<Complex64>
415        + std::ops::MulAssign
416        + std::ops::AddAssign
417        + One
418        + ComplexFloat
419        + Copy,
420    I: TableauIndex + Debug,
421{
422    /// Case_b overlap: self-pairing (branch_index = idx), so overlap = ±|c|^2.
423    /// Only even phases contribute to the real part.
424    pub fn compute_overlap_case_b(
425        entries: &[(Complex<T::Coeff>, I)],
426        phase_decomp: u8,
427        destab_anticomm_bits: I,
428    ) -> f64 {
429        let mut z_overlap_re = 0.0f64;
430        for &(coeff, idx) in entries {
431            let phase =
432                (phase_decomp + (2 * symplectic_inner(destab_anticomm_bits, idx) as u8) % 4) % 4;
433            if !phase.is_multiple_of(2) {
434                continue;
435            }
436            let norm_sq = coeff.norm_sqr().to_f64().unwrap_or(0.0);
437            if phase == 0 {
438                z_overlap_re += norm_sq;
439            } else {
440                z_overlap_re -= norm_sq;
441            }
442        }
443        z_overlap_re
444    }
445
446    /// Case_a overlap: cross-index pairing via HashMap lookup.
447    /// Accumulates only the real part of z_overlap.
448    pub fn compute_overlap_case_a(
449        coeff_map: &HashMap<I, Complex<T::Coeff>>,
450        phase_decomp: u8,
451        destab_anticomm_bits: I,
452        stab_anticomm_bits: I,
453        odd_phase_mask: I,
454    ) -> f64 {
455        let mut z_overlap_re = 0.0f64;
456        for (&idx, coeff) in coeff_map {
457            let branch_index = idx ^ stab_anticomm_bits;
458            let phase = (phase_decomp
459                + compute_phase_with_mask_static(
460                    destab_anticomm_bits,
461                    idx,
462                    stab_anticomm_bits,
463                    odd_phase_mask,
464                ))
465                % 4;
466            let Some(coeff_branch) = coeff_map.get(&branch_index).copied() else {
467                continue;
468            };
469            let a_re = coeff.re.to_f64().unwrap_or(0.0);
470            let a_im = coeff.im.to_f64().unwrap_or(0.0);
471            let b_re = coeff_branch.re.to_f64().unwrap_or(0.0);
472            let b_im = coeff_branch.im.to_f64().unwrap_or(0.0);
473            let re_w = a_re * b_re + a_im * b_im;
474            let im_w = a_re * b_im - a_im * b_re;
475            match phase {
476                0 => z_overlap_re += re_w,
477                1 => z_overlap_re += im_w,
478                2 => z_overlap_re -= re_w,
479                3 => z_overlap_re -= im_w,
480                _ => unreachable!(),
481            }
482        }
483        z_overlap_re
484    }
485
486    /// Merge B entries (k-bit=1) into their A counterparts in coeff_map.
487    /// With rayon: parallel phase computation, sequential HashMap accumulation.
488    fn merge_b_into_a(
489        coeff_map: &mut HashMap<I, Complex<T::Coeff>>,
490        b_entries: &[(I, Complex<T::Coeff>)],
491        alpha: u8,
492        destab_anticomm_bits: I,
493        stab_anticomm_bits: I,
494    ) {
495        for &(idx, coeff) in b_entries {
496            let symp_inner = symplectic_inner(idx, destab_anticomm_bits);
497            let phase_idx = ((alpha as i32 + if symp_inner % 2 == 1 { 2 } else { 0 }) % 4) as usize;
498            let q: Complex<T::Coeff> = COMPLEX_PHASE_CONVERSION[phase_idx].into();
499            *coeff_map
500                .entry(idx ^ stab_anticomm_bits)
501                .or_insert(Complex::zero()) += q * coeff;
502        }
503    }
