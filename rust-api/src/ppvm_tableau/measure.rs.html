14impl<T: Config> Measure for Tableau<T>
15where
16    <<T as Config>::Storage as BitView>::Store: PrimInt,
17{
18    /// Measure qubit `addr0` in Z basis
19    fn measure(&mut self, addr0: usize) -> bool {
20        let q = self.find_z_anticommuting_stabilizer(addr0);
21        match q {
22            Some(q_idx) => {
23                // Case a: random measurement outcome
24                // At least one stabilizer anticommutes with Z_addr0
25
26                // Generate random measurement outcome (50/50)
27                let outcome = self.rng.random::<bool>();
28
29                self.update_tableau_according_to_outcome(addr0, q_idx, outcome);
30
31                outcome
32            }
33            None => {
34                // Case b: deterministic measurement outcome
35
36                self.get_deterministic_outcome(addr0)
37            }
38        }
39    }
40}

43    Complex64::new(1.0, 0.0),  // +1
44    Complex64::new(0.0, 1.0),  // +i
45    Complex64::new(-1.0, 0.0), // -1
46    Complex64::new(0.0, -1.0), // -i
47];

49impl<T: Config, I, C: SparseVector<Complex<T::Coeff>, I>> LossyMeasure
50    for GeneralizedTableau<T, I, C>
51where
52    T: Config,
53    <<T as Config>::Storage as BitView>::Store: PrimInt,
54    C: SparseVector<Complex<T::Coeff>, I> + std::fmt::Debug,
55    T::Coeff: One
56        + Zero
57        + Clone
58        + num::Num
59        + ToPrimitive
60        + std::fmt::Debug
61        + std::ops::Mul<f64>
62        + PartialOrd<f64>
63        + Send
64        + Sync,
65    Complex<T::Coeff>: std::ops::Mul<Output = Complex<T::Coeff>>
66        + From<Complex64>
67        + std::ops::MulAssign
68        + std::ops::AddAssign
69        + One
70        + ComplexFloat
71        + Copy,
72    I: TableauIndex + Debug + Send + Sync,
73{
74    fn measure(&mut self, addr0: usize) -> Option<bool> {
75        if self.is_lost[addr0] {
76            return None;
77        }
78
79        let (phase_decomp, stab_anticomm_bits, destab_anticomm_bits) =
80            self.compute_decomposition(addr0, Pauli::Z);
81
82        if stab_anticomm_bits == I::zero() {
83            // Case b (fast path): Z is already a stabilizer.
84            // branch_index = idx ^ 0 = idx, so the overlap is self-pairing:
85            //   z_overlap = sum_alpha phase_factor(alpha).conj() * |c_alpha|^2
86            // No HashMap needed — work directly on the Vec.
87
88            // Collect into a Vec so we can iterate twice: once for overlap,
89            // once for filtering. The collect from Vec IntoIter→Vec is a no-op.
90            let entries: Vec<(Complex<T::Coeff>, I)> =
91                std::mem::replace(&mut self.coefficients, C::new())
92                    .into_iter()
93                    .collect();
94            let old_len = entries.len();
95
96            // Pass 1: compute overlap (read-only, real-only accumulation)
97            // Since conj(c)*c = |c|^2 (always real), the phase factor contribution
98            // to z_overlap.re is: phase 0 → +|c|^2, phase 2 → −|c|^2,
99            // phase 1,3 → 0 (imaginary × real = imaginary, doesn't contribute to .re)
100            let z_overlap_re =
101                Self::compute_overlap_case_b(&entries, phase_decomp, destab_anticomm_bits);
102
103            let prob_1 = 0.5 - 0.5 * z_overlap_re;
104            let outcome = self.tableau.rng.random::<f64>() < prob_1;
105
106            debug_assert!(
107                phase_decomp == 0 || phase_decomp == 2,
108                "Measurement result cannot be imaginary!"
109            );
110            let z_sign = phase_decomp == 2;
111
112            // Pass 2: filter directly into self.coefficients (no retain needed)
113            for (coeff, alpha) in entries {
114                let parity = symplectic_inner(alpha, destab_anticomm_bits) % 2 != 0;
115                if (parity ^ outcome) == z_sign {
116                    self.coefficients.unsafe_insert(alpha, coeff);
117                }
118            }
119
120            if self.coefficients.len() < old_len {
121                self.coefficients.normalize();
122            }
123
124            Some(outcome)
125        } else {
126            // Case a: Z is not a stabilizer — need HashMap for cross-index lookups
127            let mut coeff_map: HashMap<I, Complex<T::Coeff>> =
128                std::mem::replace(&mut self.coefficients, C::new())
129                    .into_iter()
130                    .map(|(v, i)| (i, v))
131                    .collect();
132
133            // Compute z_overlap.re directly (the imaginary part is always ~0)
134            let odd_phase_mask = self.odd_phase_destabilizer_mask();
135            let z_overlap_re = Self::compute_overlap_case_a(
136                &coeff_map,
137                phase_decomp,
138                destab_anticomm_bits,
139                stab_anticomm_bits,
140                odd_phase_mask,
141            );
142
143            let prob_1 = 0.5 - 0.5 * z_overlap_re;
144            let outcome = self.tableau.rng.random::<f64>() < prob_1;
145
146            let q_idx = stab_anticomm_bits.trailing_zeros() as usize;
147
148            let one = I::one();
149            let zero = I::zero();
150            let k = one << q_idx;
151
152            let alpha = if outcome {
153                (phase_decomp + 2) % 4
154            } else {
155                phase_decomp
156            };
157
158            // Partition into A (k-bit=0) and B (k-bit=1) via retain, then merge.
159            // This avoids allocating a separate b_keys Vec and eliminates
160            // individual HashMap removes (which require probe+shift).
161            let half = coeff_map.len() / 2 + 1;
162            let mut b_entries: Vec<(I, Complex<T::Coeff>)> = Vec::with_capacity(half);
163            coeff_map.retain(|idx, coeff| {
164                if (*idx & k) != zero {
165                    b_entries.push((*idx, *coeff));
166                    false // remove B entry
167                } else {
168                    true // keep A entry
169                }
170            });
171            // Merge B entries into their A partners with phase adjustment.
172            Self::merge_b_into_a(
173                &mut coeff_map,
174                &b_entries,
175                alpha,
176                destab_anticomm_bits,
177                stab_anticomm_bits,
178            );
179
180            // Keep entries where |c|/norm > threshold.
181            let norm_sqr = coeff_map
182                .values()
183                .fold(T::Coeff::zero(), |acc, c: &Complex<T::Coeff>| {
184                    acc + c.norm_sqr()
185                });
186
187            let cutoff_sq = self.coefficient_threshold.clone() * self.coefficient_threshold.clone();
188            let threshold = cutoff_sq.to_f64().unwrap_or(0.0) * norm_sqr.to_f64().unwrap_or(0.0);
189            self.coefficients = C::new();
190            for (idx, coeff) in coeff_map {
191                if coeff.norm_sqr() > threshold {
192                    self.coefficients.unsafe_insert(idx, coeff);
193                }
194            }
195
196            self.coefficients.normalize();
197
198            self.tableau
199                .update_tableau_according_to_outcome(addr0, q_idx, outcome);
200
201            Some(outcome)
202        }
203    }
204}

206impl<T, I, C> GeneralizedTableau<T, I, C>
207where
208    T: Config,
209    <<T as Config>::Storage as BitView>::Store: PrimInt,
210    C: SparseVector<Complex<T::Coeff>, I> + std::fmt::Debug,
211    T::Coeff: One
212        + Zero
213        + Clone
214        + num::Num
215        + ToPrimitive
216        + std::fmt::Debug
217        + std::ops::Mul<f64>
218        + PartialOrd<f64>
219        + Send
220        + Sync,
221    Complex<T::Coeff>: std::ops::Mul<Output = Complex<T::Coeff>>
222        + From<Complex64>
223        + std::ops::MulAssign
224        + std::ops::AddAssign
225        + One
226        + ComplexFloat
227        + Copy,
228    I: TableauIndex + Debug + Send + Sync,
229{
230    /// Measure qubit `addr0` in Z basis with optional readout noise.
231    ///
232    /// Behaves like [`measure`](LossyMeasure::measure), then with probability
233    /// `flip_prob` flips the *recorded* bit. The qubit's quantum state stays
234    /// consistent with the true outcome — only the returned value flips.
235    /// `flip_prob = 0.0` is equivalent to `measure`.
236    ///
237    /// If the qubit is lost, returns `None` regardless of `flip_prob`.
238    pub fn measure_noisy(&mut self, addr0: usize, flip_prob: f64) -> Option<bool> {
239        debug_assert!(
240            (0.0..=1.0).contains(&flip_prob),
241            "flip_prob must be in [0, 1], got {flip_prob}"
242        );
243        let outcome = self.measure(addr0)?;
244        Some(self.flip_with_prob(outcome, flip_prob))
245    }
246
247    /// Sample a Bernoulli(`p`) outcome using the tableau's internal RNG.
248    /// Used by Stim measurement-noise dispatch in `ppvm-stim`.
249    pub fn bernoulli(&mut self, p: f64) -> bool {
250        debug_assert!((0.0..=1.0).contains(&p), "p must be in [0, 1], got {p}");
251        self.tableau.rng.random::<f64>() < p
252    }
253
254    /// Flip `bit` with probability `p`. Used by Stim MR/MPad readout-noise
255    /// dispatch in `ppvm-stim`. Returns `bit` unchanged when `p <= 0.0`.
256    pub fn flip_with_prob(&mut self, bit: bool, p: f64) -> bool {
257        debug_assert!((0.0..=1.0).contains(&p), "p must be in [0, 1], got {p}");
258        if p > 0.0 && self.bernoulli(p) {
259            !bit
260        } else {
261            bit
262        }
263    }
264}

266/// Measurement overlap helper functions, with optional rayon parallelism.
267impl<T: Config, I, C: SparseVector<Complex<T::Coeff>, I>> GeneralizedTableau<T, I, C>
268where
269    T: Config,
270    <<T as Config>::Storage as BitView>::Store: PrimInt,
271    C: SparseVector<Complex<T::Coeff>, I> + std::fmt::Debug,
272    T::Coeff: One
273        + Zero
274        + Clone
275        + num::Num
276        + ToPrimitive
277        + std::fmt::Debug
278        + std::ops::Mul<f64>
279        + PartialOrd<f64>
280        + Send
281        + Sync,
282    Complex<T::Coeff>: std::ops::Mul<Output = Complex<T::Coeff>>
283        + From<Complex64>
284        + std::ops::MulAssign
285        + std::ops::AddAssign
286        + One
287        + ComplexFloat
288        + Copy,
289    I: TableauIndex + Debug + Send + Sync,
290{
291    /// Case_b overlap: self-pairing (branch_index = idx), so overlap = ±|c|^2.
292    /// Only even phases contribute to the real part.
293    fn compute_overlap_case_b(
294        entries: &[(Complex<T::Coeff>, I)],
295        phase_decomp: u8,
296        destab_anticomm_bits: I,
297    ) -> f64 {
298        let mut z_overlap_re = 0.0f64;
299        for &(coeff, idx) in entries {
300            let phase =
301                (phase_decomp + (2 * symplectic_inner(destab_anticomm_bits, idx) as u8) % 4) % 4;
302            if !phase.is_multiple_of(2) {
303                continue;
304            }
305            let norm_sq = coeff.norm_sqr().to_f64().unwrap_or(0.0);
306            if phase == 0 {
307                z_overlap_re += norm_sq;
308            } else {
309                z_overlap_re -= norm_sq;
310            }
311        }
312        z_overlap_re
313    }
314
315    /// Case_a overlap: cross-index pairing via HashMap lookup.
316    /// Accumulates only the real part of z_overlap.
317    fn compute_overlap_case_a(
318        coeff_map: &HashMap<I, Complex<T::Coeff>>,
319        phase_decomp: u8,
320        destab_anticomm_bits: I,
321        stab_anticomm_bits: I,
322        odd_phase_mask: I,
323    ) -> f64 {
324        let mut z_overlap_re = 0.0f64;
325        for (&idx, coeff) in coeff_map {
326            let branch_index = idx ^ stab_anticomm_bits;
327            let phase = (phase_decomp
328                + compute_phase_with_mask_static(
329                    destab_anticomm_bits,
330                    idx,
331                    stab_anticomm_bits,
332                    odd_phase_mask,
333                ))
334                % 4;
335            let Some(coeff_branch) = coeff_map.get(&branch_index).copied() else {
336                continue;
337            };
338            let a_re = coeff.re.to_f64().unwrap_or(0.0);
339            let a_im = coeff.im.to_f64().unwrap_or(0.0);
340            let b_re = coeff_branch.re.to_f64().unwrap_or(0.0);
341            let b_im = coeff_branch.im.to_f64().unwrap_or(0.0);
342            let re_w = a_re * b_re + a_im * b_im;
343            let im_w = a_re * b_im - a_im * b_re;
344            match phase {
345                0 => z_overlap_re += re_w,
346                1 => z_overlap_re += im_w,
347                2 => z_overlap_re -= re_w,
348                3 => z_overlap_re -= im_w,
349                _ => unreachable!(),
350            }
351        }
352        z_overlap_re
353    }
354
355    /// Merge B entries (k-bit=1) into their A counterparts in coeff_map.
356    /// With rayon: parallel phase computation, sequential HashMap accumulation.
357    fn merge_b_into_a(
358        coeff_map: &mut HashMap<I, Complex<T::Coeff>>,
359        b_entries: &[(I, Complex<T::Coeff>)],
360        alpha: u8,
361        destab_anticomm_bits: I,
362        stab_anticomm_bits: I,
363    ) {
364        for &(idx, coeff) in b_entries {
365            let symp_inner = symplectic_inner(idx, destab_anticomm_bits);
366            let phase_idx = ((alpha as i32 + if symp_inner % 2 == 1 { 2 } else { 0 }) % 4) as usize;
367            let q: Complex<T::Coeff> = COMPLEX_PHASE_CONVERSION[phase_idx].into();
368            *coeff_map
369                .entry(idx ^ stab_anticomm_bits)
370                .or_insert(Complex::zero()) += q * coeff;
371        }
372    }
