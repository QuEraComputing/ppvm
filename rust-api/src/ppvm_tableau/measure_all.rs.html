16pub trait LossyMeasureAll {
17    fn measure_all(&mut self) -> Vec<Option<bool>>;
18}
19
20impl<T: Config, I, C: SparseVector<Complex<T::Coeff>, I>> LossyMeasureAll
21    for GeneralizedTableau<T, I, C>
22where
23    T: Config,
24    <<T as Config>::Storage as BitView>::Store: PrimInt,
25    C: SparseVector<Complex<T::Coeff>, I> + std::fmt::Debug,
26    T::Coeff: One
27        + Zero
28        + Clone
29        + num::Num
30        + ToPrimitive
31        + std::fmt::Debug
32        + std::ops::Mul<f64>
33        + PartialOrd<f64>,
34    Complex<T::Coeff>: std::ops::Mul<Output = Complex<T::Coeff>>
35        + From<Complex64>
36        + std::ops::MulAssign
37        + std::ops::AddAssign
38        + One
39        + ComplexFloat
40        + Copy,
41    I: TableauIndex + Debug,
42{
43    fn measure_all(&mut self) -> Vec<Option<bool>> {
44        // One scratch reused across all n measurements: keeps the HashMap and
45        // b_entries Vec allocations across qubits, and caches the
46        // odd-phase-destabilizer mask between case-a measurements (only
47        // recomputed when `update_tableau_according_to_outcome` modifies
48        // destabilizer phases).
49        let mut scratch: MeasureScratch<I, T::Coeff> = MeasureScratch::new();
50        self.measure_all_with_scratch(&mut scratch)
51    }
52}
53
54impl<T: Config, I, C: SparseVector<Complex<T::Coeff>, I>> GeneralizedTableau<T, I, C>
55where
56    T: Config,
57    <<T as Config>::Storage as BitView>::Store: PrimInt,
58    C: SparseVector<Complex<T::Coeff>, I> + std::fmt::Debug,
59    T::Coeff: One
60        + Zero
61        + Clone
62        + num::Num
63        + ToPrimitive
64        + std::fmt::Debug
65        + std::ops::Mul<f64>
66        + PartialOrd<f64>,
67    Complex<T::Coeff>: std::ops::Mul<Output = Complex<T::Coeff>>
68        + From<Complex64>
69        + std::ops::MulAssign
70        + std::ops::AddAssign
71        + One
72        + ComplexFloat
73        + Copy,
74    I: TableauIndex + Debug,
75{
76    /// Same as [`LossyMeasureAll::measure_all`], but the caller supplies a
77    /// `MeasureScratch` that's reused across the n per-qubit measurements
78    /// (and, if the caller chooses, across many invocations / shots).
79    ///
80    /// This is the entry point samplers should use when running many shots:
81    /// initialize one scratch alongside the sampler and thread it through
82    /// every shot to amortize the case-a HashMap and b_entries allocations.
83    pub fn measure_all_with_scratch(
84        &mut self,
85        scratch: &mut MeasureScratch<I, T::Coeff>,
86    ) -> Vec<Option<bool>> {
87        (0..self.n_qubits())
88            .map(|idx| self.measure_one_with_scratch(idx, scratch))
89            .collect()
90    }
91
92    /// Measure the given qubit `indices` in order, reusing a caller-supplied
93    /// `MeasureScratch` across the per-index measurements (and, if the caller
94    /// chooses, across many invocations / shots) — the explicit-index analogue
95    /// of [`measure_all_with_scratch`](Self::measure_all_with_scratch). This is
96    /// the scratch-reusing engine behind the tableau's `LossyMeasure::measure_many`
97    /// override (in `measure.rs`), which the Stim executor and the Python
98    /// `measure_many` binding both go through.
99    pub fn measure_many_with_scratch(
100        &mut self,
101        indices: &[usize],
102        scratch: &mut MeasureScratch<I, T::Coeff>,
103    ) -> Vec<Option<bool>> {
104        indices
105            .iter()
106            .map(|&idx| self.measure_one_with_scratch(idx, scratch))
107            .collect()
108    }
109
110    /// Measure a single qubit `idx` in the Z basis, reusing `scratch`. A lost
111    /// qubit pushes `None` onto the measurement record and returns `None`,
112    /// exactly as the standalone `measure` path does. Shared by
113    /// `measure_all_with_scratch` and `measure_many_with_scratch` so every
114    /// batched path measures — and records — identically to per-qubit `measure`.
115    fn measure_one_with_scratch(
116        &mut self,
117        idx: usize,
118        scratch: &mut MeasureScratch<I, T::Coeff>,
119    ) -> Option<bool> {
120        if self.is_lost[idx] {
121            self.measurement_record.push(None);
122            return None;
123        }
124        let (phase_decomp, stab_anticomm_bits, destab_anticomm_bits) =
125            self.compute_decomposition(idx, Pauli::Z);
126        self.measure_with_scratch(
127            idx,
128            scratch,
129            phase_decomp,
130            stab_anticomm_bits,
131            destab_anticomm_bits,
132        )
133    }
134}
