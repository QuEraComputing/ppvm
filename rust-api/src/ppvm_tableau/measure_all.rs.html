16pub trait LossyMeasureAll {
17    fn measure_all(&mut self) -> Vec<Option<bool>>;
18}
19
20impl<T: Config, I, C: SparseVector<Complex<T::Coeff>, I>> LossyMeasureAll
21    for GeneralizedTableau<T, I, C>
22where
23    T: Config,
24    <<T as Config>::Storage as BitView>::Store: PrimInt,
25    C: SparseVector<Complex<T::Coeff>, I> + std::fmt::Debug,
26    T::Coeff: One
27        + Zero
28        + Clone
29        + num::Num
30        + ToPrimitive
31        + std::fmt::Debug
32        + std::ops::Mul<f64>
33        + PartialOrd<f64>,
34    Complex<T::Coeff>: std::ops::Mul<Output = Complex<T::Coeff>>
35        + From<Complex64>
36        + std::ops::MulAssign
37        + std::ops::AddAssign
38        + One
39        + ComplexFloat
40        + Copy,
41    I: TableauIndex + Debug,
42{
43    fn measure_all(&mut self) -> Vec<Option<bool>> {
44        // One scratch reused across all n measurements: keeps the HashMap and
45        // b_entries Vec allocations across qubits, and caches the
46        // odd-phase-destabilizer mask between case-a measurements (only
47        // recomputed when `update_tableau_according_to_outcome` modifies
48        // destabilizer phases).
49        let mut scratch: MeasureScratch<I, T::Coeff> = MeasureScratch::new();
50        self.measure_all_with_scratch(&mut scratch)
51    }
52}
53
54impl<T: Config, I, C: SparseVector<Complex<T::Coeff>, I>> GeneralizedTableau<T, I, C>
55where
56    T: Config,
57    <<T as Config>::Storage as BitView>::Store: PrimInt,
58    C: SparseVector<Complex<T::Coeff>, I> + std::fmt::Debug,
59    T::Coeff: One
60        + Zero
61        + Clone
62        + num::Num
63        + ToPrimitive
64        + std::fmt::Debug
65        + std::ops::Mul<f64>
66        + PartialOrd<f64>,
67    Complex<T::Coeff>: std::ops::Mul<Output = Complex<T::Coeff>>
68        + From<Complex64>
69        + std::ops::MulAssign
70        + std::ops::AddAssign
71        + One
72        + ComplexFloat
73        + Copy,
74    I: TableauIndex + Debug,
75{
76    /// Same as [`LossyMeasureAll::measure_all`], but the caller supplies a
77    /// `MeasureScratch` that's reused across the n per-qubit measurements
78    /// (and, if the caller chooses, across many invocations / shots).
79    ///
80    /// This is the entry point samplers should use when running many shots:
81    /// initialize one scratch alongside the sampler and thread it through
82    /// every shot to amortize the case-a HashMap and b_entries allocations.
83    pub fn measure_all_with_scratch(
84        &mut self,
85        scratch: &mut MeasureScratch<I, T::Coeff>,
86    ) -> Vec<Option<bool>> {
87        (0..self.n_qubits())
88            .map(|idx| {
89                if self.is_lost[idx] {
90                    return None;
91                }
92                let (phase_decomp, stab_anticomm_bits, destab_anticomm_bits) =
93                    self.compute_decomposition(idx, Pauli::Z);
94                self.measure_with_scratch(
95                    idx,
96                    scratch,
97                    phase_decomp,
98                    stab_anticomm_bits,
99                    destab_anticomm_bits,
100                )
101            })
102            .collect()
103    }
104}
