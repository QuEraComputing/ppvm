11/// Check that a value is a valid probability (in [0, 1]).
12/// Extracted to avoid triggering `clippy::manual_range_contains` when
13/// the generic `T::Coeff` does not implement `PartialOrd` in both directions.
14#[inline]
15fn is_probability(p: &impl PartialOrd<f64>) -> bool {
16    *p >= 0.0 && *p <= 1.0
17}
18
19impl<T: Config> Depolarizing<T> for Tableau<T>
20where
21    T::Coeff: PartialOrd<f64>,
22{
23    fn depolarize(&mut self, addr0: usize, p: T::Coeff) {
24        debug_assert!(is_probability(&p));
25        let r = self.rng.random::<f64>();
26        if p <= r {
27            return;
28        }
29        if p > r * 3.0 {
30            // p / 3 > r >= 0
31            self.x(addr0);
32        } else if p > r * 1.5 {
33            // 2p/3 > r >= p / 3
34            self.y(addr0);
35        } else {
36            // p > r >= 2p/3
37            self.z(addr0);
38        }
39    }
40}

67impl<T: Config> PauliError<T> for Tableau<T>
68where
69    T::Coeff: PartialOrd<f64> + Zero,
70{
71    fn pauli_error(&mut self, addr0: usize, p: [<T as Config>::Coeff; 3]) {
72        let r = self.rng.random::<f64>();
73        let mut cumulative = T::Coeff::zero();
74        for (i, p_) in p.iter().enumerate() {
75            cumulative += p_.clone();
76            if cumulative > r {
77                match i {
78                    0 => self.x(addr0),
79                    1 => self.y(addr0),
80                    _ => self.z(addr0),
81                }
82                return;
83            }
84        }
85    }
86}

88impl<T: Config, I: TableauIndex, C: SparseVector<Complex<T::Coeff>, I>> PauliError<T>
89    for GeneralizedTableau<T, I, C>
90where
91    T::Coeff: PartialOrd<f64> + Zero,
92    Complex<<T as Config>::Coeff>: From<Complex<f64>>,
93{
94    fn pauli_error(&mut self, addr0: usize, p: [<T as Config>::Coeff; 3]) {
95        debug_assert!(p.iter().all(|p_| *p_ >= 0.0 && *p_ <= 1.0));
96        debug_assert!(p[0].clone() + p[1].clone() + p[2].clone() - 1.0 < 1e-7);
97        let r = self.tableau.rng.random::<f64>();
98        let mut cumulative = T::Coeff::zero();
99        for (i, p_) in p.iter().enumerate() {
100            cumulative += p_.clone();
101            if cumulative > r {
102                match i {
103                    0 => self.x(addr0),
104                    1 => self.y(addr0),
105                    _ => self.z(addr0),
106                }
107                return;
108            }
109        }
110    }

113fn two_qubit_pauli_error_impl<T: Config>(
114    this: &mut impl Clifford,
115    addr0: usize,
116    addr1: usize,
117    p: [T::Coeff; 15],
118    r: f64,
119) where
120    T::Coeff: PartialOrd<f64> + Zero,
121{
122    debug_assert!(p.iter().all(|p_| *p_ >= 0.0 && *p_ <= 1.0));
123    // debug_assert!(p.iter().sum() - 1.0 < 1e-7);
124    let sum = T::Coeff::zero();
125    let idx = p
126        .iter()
127        .scan(sum, |acc, p_| {
128            *acc += p_.clone();
129            Some(acc.clone())
130        })
131        .position(|cum_prob| cum_prob > r);
132
133    if let Some(i) = idx {
134        #[rustfmt::skip]
135        const PAULI_PAIRS: [(u8, u8); 16] = [
136            (0,0),(0,1),(0,2),(0,3),
137            (1,0),(1,1),(1,2),(1,3),
138            (2,0),(2,1),(2,2),(2,3),
139            (3,0),(3,1),(3,2),(3,3),
140        ];
141        let cartesian_index = PAULI_PAIRS[i + 1]; // skip II entry
142
143        match cartesian_index.0 {
144            0 => {}
145            1 => this.x(addr0),
146            2 => this.y(addr0),
147            _ => this.z(addr0),
148        }
149
150        match cartesian_index.1 {
151            0 => {}
152            1 => this.x(addr1),
153            2 => this.y(addr1),
154            _ => this.z(addr1),
155        }
156    }
157}

213impl<T: Config, I: TableauIndex + Send + Sync, C: SparseVector<Complex<T::Coeff>, I>> LossChannel<T>
214    for GeneralizedTableau<T, I, C>
215where
216    <<T as Config>::Storage as BitView>::Store: PrimInt,
217    C: std::fmt::Debug,
218    T::Coeff: PartialOrd<f64>
219        + One
220        + Zero
221        + Clone
222        + num::Num
223        + ToPrimitive
224        + std::fmt::Debug
225        + Send
226        + Sync,
227    Complex<T::Coeff>: std::ops::Mul<Output = Complex<T::Coeff>>
228        + From<Complex64>
229        + std::ops::MulAssign
230        + std::ops::AddAssign
231        + One
232        + ComplexFloat
233        + Copy,
234    I: Debug,
235{
236    fn loss_channel(&mut self, addr0: usize, p: <T as Config>::Coeff) {
237        if p < self.tableau.rng.random::<f64>() {
238            return;
239        }
240
241        // NOTE: this is O(n^2) but also potentially removes coefficients, which is nice
242        let outcome = self.measure(addr0);
243        if let Some(true) = outcome {
244            // flip back to 0
245            self.x(addr0);
246        }
247        self.is_lost[addr0] = true;
248    }
249}

251impl<T: Config, I: TableauIndex + Send + Sync, C: SparseVector<Complex<T::Coeff>, I>>
252    CorrelatedLossChannel<T> for GeneralizedTableau<T, I, C>
253where
254    <<T as Config>::Storage as BitView>::Store: PrimInt,
255    C: std::fmt::Debug,
256    T::Coeff: PartialOrd<f64>
257        + One
258        + Zero
259        + Clone
260        + num::Num
261        + ToPrimitive
262        + std::fmt::Debug
263        + Send
264        + Sync,
265    Complex<T::Coeff>: std::ops::Mul<Output = Complex<T::Coeff>>
266        + From<Complex64>
267        + std::ops::MulAssign
268        + std::ops::AddAssign
269        + One
270        + ComplexFloat
271        + Copy,
272    I: Debug,
273{
274    /// Apply a correlated loss channel to qubits at `addr0` and `addr1`.
275    ///
276    /// The three probabilities are:
277    /// * `p[0]`: The probability of losing both qubits simultaneously when
278    ///   both of them are in the qubit subspace.
279    /// * `p[1]`: The probability of losing either one qubit when both of them are
280    ///   in the qubit subspace.
281    /// * `p[2]`: The probability of losing one qubit when the other one has already
282    ///   been lost prior to the channel.
283    fn correlated_loss_channel(
284        &mut self,
285        addr0: usize,
286        addr1: usize,
287        p: [<T as Config>::Coeff; 3],
288    ) {
289        if self.is_lost[addr0] {
290            self.loss_channel(addr1, p[2].clone());
291            return;
292        } else if self.is_lost[addr1] {
293            self.loss_channel(addr0, p[2].clone());
294            return;
295        }
296
297        let r = self.tableau.rng.random::<f64>();
298        let mut cumulative = T::Coeff::zero();
299        for (i, p_i) in p[..2].iter().enumerate() {
300            cumulative += p_i.clone();
301            if cumulative > r {
302                if i == 0 {
303                    // both lost
304                    self.reset(addr0);
305                    self.reset(addr1);
306                    self.is_lost[addr0] = true;
307                    self.is_lost[addr1] = true;
308                } else {
309                    // only losing a single qubit,
310                    let choice = self.tableau.rng.random::<bool>();
311                    if choice {
312                        self.reset(addr1);
313                        self.is_lost[addr1] = true;
314                    } else {
315                        self.reset(addr0);
316                        self.is_lost[addr0] = true;
317                    }
318                }
319                return;
320            }
321        }
322    }
323}

590    #[test]
591    fn depolarize_statistics() {
592        // Starting from |0⟩, P(measure 1) = P(X) + P(Y) = p/3 + p/3 = 2p/3.
593        // Z leaves |0⟩ unchanged; I leaves |0⟩ unchanged.
594        let p = 0.6_f64;
595        let expected = 2.0 * p / 3.0; // 0.4
596        let trials = 500;
597
598        let ones = (0..trials)
599            .filter(|_| {
600                let mut t = tab(1);
601                t.depolarize(0, p);
602                t.measure(0).unwrap()
603            })
604            .count();
605
606        let fraction = ones as f64 / trials as f64;
607        // tolerance ~5σ: σ = sqrt(expected*(1-expected)/trials) ≈ 0.022
608        assert!(
609            (fraction - expected).abs() < 0.1,
610            "Expected fraction {expected:.3}, got {fraction:.3}"
611        );
612    }

614    #[test]
615    fn depolarize2_statistics() {
616        // Starting from |00⟩, errors that flip qubit 0 to |1⟩ are X and Y on that qubit:
617        // XI, XX, XY, XZ, YI, YX, YY, YZ — 8 out of 15, so P(q0=1) = 8p/15.
618        let p = 0.6_f64;
619        let expected = 8.0 * p / 15.0; // 0.32
620        let trials = 500;
621
622        let ones = (0..trials)
623            .filter(|_| {
624                let mut t = tab(2);
625                t.depolarize2(0, 1, p);
626                t.measure(0).unwrap()
627            })
628            .count();
