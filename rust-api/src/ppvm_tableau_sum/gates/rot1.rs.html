4use bitvec::view::BitView;
5use num::PrimInt;
6use num::{
7    Complex, One, Zero,
8    complex::{Complex64, ComplexFloat},
9};
10use ppvm_tableau::sparsevec::SparseVector;
11use ppvm_tableau::tableau_index::TableauIndex;
12use ppvm_traits::char::Pauli;
13use ppvm_traits::config::Config;
14use ppvm_traits::traits::RotationOne;
15
16use crate::data::GeneralizedTableauSum;
17use crate::storage::EntryStore;
18
19impl<T: Config, I, C: SparseVector<Complex<T::Coeff>, I>, S> RotationOne<T>
20    for GeneralizedTableauSum<T, I, C, S>
21where
22    <<T as Config>::Storage as BitView>::Store: PrimInt,
23    T::Coeff: Zero + One + Send + Sync + num::Num + PartialOrd,
24    I: TableauIndex + Send + Sync,
25    Complex<T::Coeff>: std::ops::Mul<Output = Complex<T::Coeff>>
26        + std::ops::AddAssign
27        + From<Complex64>
28        + ComplexFloat
29        + Copy,
30    S: EntryStore<T, I, C>,
31{
32    fn rotate_1(&mut self, axis: Pauli, addr0: usize, theta: T::Coeff) {
33        self.entries.for_each_mut(|tab, _p| {
34            tab.rotate_1(axis, addr0, theta.clone());
35        });
36        self.entries.mark_keys_dirty();
37    }
38}
