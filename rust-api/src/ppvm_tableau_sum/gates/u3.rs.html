4use bitvec::view::BitView;
5use num::{
6    Complex, One, PrimInt, Zero,
7    complex::{Complex64, ComplexFloat},
8};
9use ppvm_tableau::{sparsevec::SparseVector, tableau_index::TableauIndex};
10use ppvm_traits::{config::Config, traits::U3Gate};
11
12use crate::{data::GeneralizedTableauSum, storage::EntryStore};
13
14impl<T: Config, I, C: SparseVector<Complex<T::Coeff>, I>, S> U3Gate<T>
15    for GeneralizedTableauSum<T, I, C, S>
16where
17    <<T as Config>::Storage as BitView>::Store: PrimInt,
18    T::Coeff: Zero + One + Send + Sync + num::Num + PartialOrd,
19    I: TableauIndex + Send + Sync,
20    Complex<T::Coeff>: std::ops::Mul<Output = Complex<T::Coeff>>
21        + std::ops::AddAssign
22        + From<Complex64>
23        + ComplexFloat
24        + Copy,
25    S: EntryStore<T, I, C>,
26{
27    fn u3(&mut self, addr0: usize, theta: T::Coeff, phi: T::Coeff, lambda: T::Coeff) {
28        self.entries.for_each_mut(|tab, _p| {
29            tab.u3(addr0, theta.clone(), phi.clone(), lambda.clone());
30        });
31        self.entries.mark_keys_dirty();
32    }
33}
