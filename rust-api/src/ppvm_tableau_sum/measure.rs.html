45    /// Branch each entry on a Z-basis measurement of `addr0`. `per_branch`
46    /// is invoked once per resulting sub-branch with `(tab, outcome,
47    /// p_branch)`:
48    ///
49    /// * `outcome` is `None` for entries where `addr0` is already lost (no
50    ///   projection happens), and `Some(true)` / `Some(false)` for the two
51    ///   projection outcomes on a live qubit.
52    /// * `p_branch` is the absolute branch probability — the parent
53    ///   entry's `p_sum` scaled by the conditional outcome probability.
54    ///
55    /// `per_branch` may mutate `tab` (e.g. reset's conditional X) or just
56    /// accumulate statistics (e.g. measure's per-outcome aggregator), and
57    /// must return `true` iff it mutated the tableau. The helper uses the
58    /// flag to skip recomputing the branch's fingerprints in case b
59    /// (`project_case_b` leaves the tableau structure untouched, so the
60    /// parent's fingerprint stays valid unless the closure perturbed it).
61    /// Case a always recomputes because `project_case_a` rewrites tableau
62    /// rows.
63    pub(crate) fn for_each_z_branch<F>(&mut self, addr0: usize, mut per_branch: F)
64    where
65        F: FnMut(&mut GeneralizedTableau<T, I, C>, Option<bool>, &T::Coeff) -> bool,
66    {
67        let n_entries = self.entries.len();
68        let mut branches =
69            Vec::<(GeneralizedTableau<T, I, C>, T::Coeff, u64, u64)>::with_capacity(n_entries);
70
71        let mut scratch = MeasureScratch::<I, T::Coeff>::new();
72        let mut scratch_other_outcome = scratch.clone();
73        let mut any_tableau_mutation = false;
74
75        self.entries
76            .for_each_mut_with_keys(|tab, p_sum, word_fp, phase_loss_fp| {
77                if tab.is_lost[addr0] {
78                    // NOTE: deterministically outputs lost, no branching
79                    let _ = per_branch(tab, None, &*p_sum);
80                    return;
81                }
82
83                let (phase_decomp, stab_anticomm_bits, destab_anticomm_bits) =
84                    tab.compute_decomposition(addr0, Pauli::Z);
85
86                let tab_seed = self.rng.random::<u64>();
87
88                if stab_anticomm_bits == I::zero() {
89                    // case b
90                    let entries: Vec<(Complex<T::Coeff>, I)> =
91                        std::mem::replace(&mut tab.coefficients, C::new())
92                            .into_iter()
93                            .collect();
94
95                    // NOTE: fork AFTER draining coefficients, so we only copy an
96                    // empty coefficients vec
97                    let mut tab_other_outcome = tab.fork(Some(tab_seed));
98
99                    // Pass 1: compute overlap (read-only, real-only accumulation)
100                    // Since conj(c)*c = |c|^2 (always real), the phase factor contribution
101                    // to z_overlap.re is: phase 0 → +|c|^2, phase 2 → −|c|^2,
102                    // phase 1,3 → 0 (imaginary × real = imaginary, doesn't contribute to .re)
103                    let z_overlap_re = GeneralizedTableau::<T, I, C>::compute_overlap_case_b(
104                        &entries,
105                        phase_decomp,
106                        destab_anticomm_bits,
107                    );
108
109                    let prob_1 = 0.5 - 0.5 * z_overlap_re;
110                    let prob_0 = 1.0 - prob_1;
111
112                    // make the existing term the more likely outcome
113                    let likely_outcome = prob_1 > 0.5;
114                    let (p_likely, p_other) = if likely_outcome {
115                        (prob_1, prob_0)
116                    } else {
117                        (prob_0, prob_1)
118                    };
119
120                    debug_assert!(
121                        phase_decomp == 0 || phase_decomp == 2,
122                        "Measurement result cannot be imaginary!"
123                    );
124
125                    let p_branch_other = p_sum.clone() * p_other;
126
127                    // project
128                    // NOTE: avoid projecting into zero amplitude state
129                    // intentionally stricter than normal truncate
130                    // dropping terms < sum_cutoff even if they'd merge with another one
131                    if Into::<T::Coeff>::into(p_other) > self.sum_cutoff {
132                        tab_other_outcome.project_case_b(
133                            &entries,
134                            !likely_outcome,
135                            phase_decomp,
136                            destab_anticomm_bits,
137                        );
138                        let closure_mutated = per_branch(
139                            &mut tab_other_outcome,
140                            Some(!likely_outcome),
141                            &p_branch_other,
142                        );
143                        // project_case_b leaves tableau structure intact; only
144                        // re-hash when the closure perturbed it.
145                        let (word_fp_other, phase_loss_other) = if closure_mutated {
146                            any_tableau_mutation = true;
147                            (
148                                word_fingerprint(&tab_other_outcome),
149                                phase_loss_hash(&tab_other_outcome),
150                            )
151                        } else {
152                            (word_fp, phase_loss_fp)
153                        };
154                        branches.push((
155                            tab_other_outcome,
156                            p_branch_other,
157                            word_fp_other,
158                            phase_loss_other,
159                        ));
160                    }
161
162                    // update existing entry
163                    tab.project_case_b(
164                        &entries,
165                        likely_outcome,
166                        phase_decomp,
167                        destab_anticomm_bits,
168                    );
169                    *p_sum *= p_likely;
170                    if per_branch(tab, Some(likely_outcome), &*p_sum) {
171                        any_tableau_mutation = true;
172                    }
173                } else {
174                    // case a
175                    scratch.coeff_map.clear();
176                    scratch.coeff_map.reserve(tab.coefficients.len());
177                    {
178                        let coeff_map = &mut scratch.coeff_map;
179                        tab.coefficients.retain(|(v, i)| {
180                            coeff_map.insert(*i, *v);
181                            false // drain — keeps allocation
182                        });
183                    }
184
185                    scratch_other_outcome
186                        .coeff_map
187                        .clone_from(&scratch.coeff_map);
188
189                    // NOTE: fork AFTER draining coefficients, so we only copy an
190                    // empty coefficients vec
191                    let mut tab_other_outcome = tab.fork(Some(tab_seed));
192
193                    // Compute z_overlap.re directly (the imaginary part is always ~0).
194                    // The mask is a pure function of destabilizer phases — cache it across
195                    // measurements until `update_tableau_according_to_outcome` invalidates it.
196                    let odd_phase_mask = *scratch
197                        .odd_phase_mask
198                        .get_or_insert_with(|| tab.odd_phase_destabilizer_mask());
199                    let z_overlap_re = GeneralizedTableau::<T, I, C>::compute_overlap_case_a(
200                        &scratch.coeff_map,
201                        phase_decomp,
202                        destab_anticomm_bits,
203                        stab_anticomm_bits,
204                        odd_phase_mask,
205                    );
206
207                    let prob_1 = 0.5 - 0.5 * z_overlap_re;
208                    let prob_0 = 1.0 - prob_1;
209
210                    // make the existing term the more likely outcome
211                    let likely_outcome = prob_1 > 0.5;
212                    let (p_likely, p_other) = if likely_outcome {
213                        (prob_1, prob_0)
214                    } else {
215                        (prob_0, prob_1)
216                    };
217
218                    let p_branch_other = p_sum.clone() * p_other;
219
220                    // project
221                    if Into::<T::Coeff>::into(p_other) > self.sum_cutoff {
222                        tab_other_outcome.project_case_a(
223                            !likely_outcome,
224                            &mut scratch_other_outcome,
225                            phase_decomp,
226                            stab_anticomm_bits,
227                            destab_anticomm_bits,
228                            addr0,
229                        );
230                        let _ = per_branch(
231                            &mut tab_other_outcome,
232                            Some(!likely_outcome),
233                            &p_branch_other,
234                        );
235                        // project_case_a always rewrites the tableau, so we
236                        // must recompute the branch's fingerprint regardless
237                        // of what the closure did. The in-place project_case_a
238                        // below also mutates, so we record the flag once at
239                        // the end of the case-a arm.
240                        let word_fp_other = word_fingerprint(&tab_other_outcome);
241                        let phase_loss_other = phase_loss_hash(&tab_other_outcome);
242                        branches.push((
243                            tab_other_outcome,
244                            p_branch_other,
245                            word_fp_other,
246                            phase_loss_other,
247                        ));
248                    }
249
250                    // update exisiting entry in-place
251                    tab.project_case_a(
252                        likely_outcome,
253                        &mut scratch,
254                        phase_decomp,
255                        stab_anticomm_bits,
256                        destab_anticomm_bits,
257                        addr0,
258                    );
259                    *p_sum *= p_likely;
260                    let _ = per_branch(tab, Some(likely_outcome), &*p_sum);
261                    any_tableau_mutation = true;
262                }
263            });
264
265        if any_tableau_mutation {
266            self.entries.mark_keys_dirty();
267        }
268        let needs_normalize = self
269            .entries
270            .insert_or_merge_batch(branches, &self.sum_cutoff);
271        if needs_normalize {
272            self.normalize_probabilities();
273        }
274        self.truncate();
275    }

277    /// Routine for mid-circuit measurements (for obtaining results use sampling)
278    /// branch into different outcomes and return probabilities for outcomes (zero, one, lost)
279    pub fn measure(&mut self, addr0: usize) -> (T::Coeff, T::Coeff, T::Coeff) {
280        let mut p_zero = Vec::<T::Coeff>::new();
281        let mut p_one = Vec::<T::Coeff>::new();
282        let mut p_lost = Vec::<T::Coeff>::new();
283
284        self.for_each_z_branch(addr0, |_tab, outcome, p| {
285            match outcome {
286                Some(true) => p_one.push(p.clone()),
287                Some(false) => p_zero.push(p.clone()),
288                None => p_lost.push(p.clone()),
289            }
290            false // measure never mutates the tableau
291        });
292
293        let p_0 = p_zero
294            .iter()
295            .fold(T::Coeff::zero(), |acc, p| acc + p.clone());
296        let p_1 = p_one
297            .iter()
298            .fold(T::Coeff::zero(), |acc, p| acc + p.clone());
299        let p_l = p_lost
300            .iter()
301            .fold(T::Coeff::zero(), |acc, p| acc + p.clone());
302        (p_0, p_1, p_l)
303    }
