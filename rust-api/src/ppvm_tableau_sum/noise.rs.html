29fn single_qubit_loss_branch<T, I, C>(
30    addr0: usize,
31    p: &T::Coeff,
32    rng: &mut SmallRng,
33    branches: &mut Vec<Branch<T, I, C>>,
34    tab: &mut GeneralizedTableau<T, I, C>,
35    p_sum: &mut T::Coeff,
36    // The branch inherits its parent's cached fingerprint halves:
37    // `(word_fingerprint, phase_loss_hash)`.
38    (word_fp, phase_loss): (u64, u64),
39) where
40    T: Config,
41    <<T as Config>::Storage as BitView>::Store: PrimInt,
42    C: std::fmt::Debug,
43    T::Coeff: PartialOrd<f64>
44        + PartialOrd
45        + One
46        + Zero
47        + Clone
48        + num::Num
49        + ToPrimitive
50        + std::fmt::Debug
51        + Send
52        + Sync,
53    Complex<T::Coeff>: std::ops::Mul<Output = Complex<T::Coeff>>
54        + From<Complex64>
55        + std::ops::MulAssign
56        + std::ops::AddAssign
57        + One
58        + ComplexFloat
59        + Copy,
60    I: TableauIndex + Send + Sync + Debug,
61    C: SparseVector<Complex<T::Coeff>, I>,
62{
63    if tab.is_lost[addr0] {
64        // Don't branch if it's already lost
65        return;
66    }
67
68    let tab_seed = rng.random::<u64>();
69    let mut tab_branch = tab.fork(Some(tab_seed));
70    tab_branch.is_lost[addr0] = true;
71    // is_lost flip leaves the Pauli words and phases unchanged, so
72    // the branch reuses its parent's word-fingerprint and the only
73    // change to the phase/loss hash is the lost qubit's mask.
74    branches.push((
75        tab_branch,
76        p_sum.clone() * p.clone(),
77        word_fp,
78        phase_loss ^ loss_mask(addr0),
79    ));
80    *p_sum *= T::Coeff::one() - p.clone();
81}

83impl<
84    T: Config,
85    I: TableauIndex + Send + Sync,
86    C: SparseVector<Complex<T::Coeff>, I>,
87    S: EntryStore<T, I, C>,
88> LossChannel<T> for GeneralizedTableauSum<T, I, C, S>
89where
90    <<T as Config>::Storage as BitView>::Store: PrimInt,
91    C: std::fmt::Debug,
92    T::Coeff: PartialOrd<f64>
93        + PartialOrd
94        + One
95        + Zero
96        + Clone
97        + num::Num
98        + ToPrimitive
99        + std::fmt::Debug
100        + Send
101        + Sync,
102    Complex<T::Coeff>: std::ops::Mul<Output = Complex<T::Coeff>>
103        + From<Complex64>
104        + std::ops::MulAssign
105        + std::ops::AddAssign
106        + One
107        + ComplexFloat
108        + Copy,
109    I: Debug,
110{
111    fn loss_channel(&mut self, addr0: usize, p: <T as Config>::Coeff) {
112        // Lazy branch materialization: describe each loss branch as a mutation
113        // of its parent entry. The merge clones the parent only when the branch
114        // survives as a NEW entry; merges/below-cutoff drops never clone.
115        let mut branches =
116            Vec::<(usize, BranchMutation, T::Coeff, u64, u64)>::with_capacity(self.entries.len());
117        let mut idx = 0usize;
118        self.entries
119            .for_each_mut_with_keys(|tab, p_sum, word_fp, phase_loss| {
120                // Increment for EVERY entry, before the lost check, so
121                // parent_idx aligns with for_each_mut_with_keys' order.
122                let parent_idx = idx;
123                idx += 1;
124                if tab.is_lost[addr0] {
125                    return;
126                }
127                branches.push((
128                    parent_idx,
129                    BranchMutation::Loss { q: addr0 },
130                    p_sum.clone() * p.clone(),
131                    word_fp,
132                    phase_loss ^ loss_mask(addr0),
133                ));
134                *p_sum *= T::Coeff::one() - p.clone();
135            });
136
137        let needs_renormalize = self
138            .entries
139            .insert_or_merge_mutated_branches(branches, &self.sum_cutoff);
140        if needs_renormalize {
141            self.normalize_probabilities();
142        }
143        self.truncate();
144    }
145}

181impl<
182    T: Config,
183    I: TableauIndex + Send + Sync,
184    C: SparseVector<Complex<T::Coeff>, I>,
185    S: EntryStore<T, I, C>,
186> PauliError<T> for GeneralizedTableauSum<T, I, C, S>
187where
188    <<T as Config>::Storage as BitView>::Store: PrimInt,
189    C: std::fmt::Debug,
190    T::Coeff: PartialOrd<f64>
191        + PartialOrd
192        + One
193        + Zero
194        + Clone
195        + num::Num
196        + ToPrimitive
197        + std::fmt::Debug
198        + Send
199        + Sync,
200    Complex<T::Coeff>: std::ops::Mul<Output = Complex<T::Coeff>>
201        + From<Complex64>
202        + std::ops::MulAssign
203        + std::ops::AddAssign
204        + One
205        + ComplexFloat
206        + Copy,
207    I: Debug,
208{
209    fn pauli_error(&mut self, addr0: usize, p: [<T as Config>::Coeff; 3]) {
210        let p_total: T::Coeff = p[0].clone() + p[1].clone() + p[2].clone();
211        // Lazy branch materialization: describe each X/Y/Z branch as a Pauli
212        // mutation of its parent. The phase/loss delta is computed by walking the
213        // parent's column once (no clone) — X flips rows with z, Y with x^z, Z
214        // with x — matching what `pauli_branch_phase_loss` would produce.
215        let mut branches = Vec::<(usize, BranchMutation, T::Coeff, u64, u64)>::with_capacity(
216            3 * self.entries.len(),
217        );
218        // Precompute the per-row sign masks once instead of recomputing the
219        // splitmix `sign_mask` per row per entry in the hot loop below.
220        let masks = RowMasks::new(self.n_qubits);
221        // The store-word index / bit position of column `addr0` are the same for
222        // every entry and row, so resolve them once (Lsb0 convention).
223        let bits_per_word = std::mem::size_of::<<T::Storage as BitView>::Store>() * 8;
224        let word_idx = addr0 / bits_per_word;
225        let bit = addr0 % bits_per_word;
226        let mut idx = 0usize;
227        self.entries
228            .for_each_mut_with_keys(|tab, p_sum, word_fp, phase_loss| {
229                let parent_idx = idx;
230                idx += 1;
231                if tab.is_lost[addr0] {
232                    return;
233                }
234
235                let (mut dx, mut dy, mut dz) = (0u64, 0u64, 0u64);
236                for (row, pw) in tab.tableau.data.iter().enumerate() {
237                    let xw = pw.word.xbits.data.as_raw_slice();
238                    let zw = pw.word.zbits.data.as_raw_slice();
239                    let x: bool = bit_at(xw, word_idx, bit);
240                    let z: bool = bit_at(zw, word_idx, bit);
241                    let m = masks.sign[row];
242                    if z {
243                        dx ^= m;
244                    }
245                    if x ^ z {
246                        dy ^= m;
247                    }
248                    if x {
249                        dz ^= m;
250                    }
251                }
252
253                branches.push((
254                    parent_idx,
255                    BranchMutation::Pauli {
256                        op: NotIdentity::X,
257                        addr0,
258                    },
259                    p_sum.clone() * p[0].clone(),
260                    word_fp,
261                    phase_loss ^ dx,
262                ));
263                branches.push((
264                    parent_idx,
265                    BranchMutation::Pauli {
266                        op: NotIdentity::Y,
267                        addr0,
268                    },
269                    p_sum.clone() * p[1].clone(),
270                    word_fp,
271                    phase_loss ^ dy,
272                ));
273                branches.push((
274                    parent_idx,
275                    BranchMutation::Pauli {
276                        op: NotIdentity::Z,
277                        addr0,
278                    },
279                    p_sum.clone() * p[2].clone(),
280                    word_fp,
281                    phase_loss ^ dz,
282                ));
283
284                *p_sum *= T::Coeff::one() - p_total.clone();
285            });
286
287        let needs_normalize = self
288            .entries
289            .insert_or_merge_mutated_branches(branches, &self.sum_cutoff);
290        if needs_normalize {
291            self.normalize_probabilities();
292        }
293        self.truncate();
294    }
295}

297impl<
298    T: Config,
299    I: TableauIndex + Send + Sync,
300    C: SparseVector<Complex<T::Coeff>, I>,
301    S: EntryStore<T, I, C>,
302> TwoQubitPauliError<T> for GeneralizedTableauSum<T, I, C, S>
303where
304    <<T as Config>::Storage as BitView>::Store: PrimInt,
305    C: std::fmt::Debug,
306    T::Coeff: PartialOrd<f64>
307        + PartialOrd
308        + One
309        + Zero
310        + Clone
311        + num::Num
312        + ToPrimitive
313        + std::fmt::Debug
314        + Send
315        + Sync,
316    Complex<T::Coeff>: std::ops::Mul<Output = Complex<T::Coeff>>
317        + From<Complex64>
318        + std::ops::MulAssign
319        + std::ops::AddAssign
320        + One
321        + ComplexFloat
322        + Copy,
323    I: Debug,
324{
325    fn two_qubit_pauli_error(&mut self, addr0: usize, addr1: usize, p: [<T as Config>::Coeff; 15]) {
326        let p_total: T::Coeff = p
327            .iter()
328            .fold(T::Coeff::zero(), |acc, prob| acc + prob.clone());
329        let mut branches = Vec::<(GeneralizedTableau<T, I, C>, T::Coeff, u64, u64)>::with_capacity(
330            15 * self.entries.len(),
331        );
332
333        // The 15 non-identity two-qubit Paulis on (addr0, addr1), in the same
334        // order as the probability array `p`: IX, IY, IZ, XI, XX, XY, XZ, YI,
335        // YX, YY, YZ, ZI, ZX, ZY, ZZ. Encoding: 0 = I, 1 = X, 2 = Y, 3 = Z.
336        //
337        // `rustfmt::skip` keeps the rows grouped by the first Pauli (a readable
338        // 4-wide grid); without it rustfmt repacks the tuples to fill the line
339        // width and the grouping is lost.
340        #[rustfmt::skip]
341        const PAULI_PAIRS: [(u8, u8); 15] = [
342            (0, 1), (0, 2), (0, 3),
343            (1, 0), (1, 1), (1, 2), (1, 3),
344            (2, 0), (2, 1), (2, 2), (2, 3),
345            (3, 0), (3, 1), (3, 2), (3, 3),
346        ];
347
348        let apply = |t: &mut GeneralizedTableau<T, I, C>, op: u8, addr: usize| match op {
349            1 => t.x(addr),
350            2 => t.y(addr),
351            3 => t.z(addr),
352            _ => {}
353        };
354
355        self.entries
356            .for_each_mut_with_keys(|tab, p_sum, word_fp, phase_loss| {
357                if tab.is_lost[addr0] || tab.is_lost[addr1] {
358                    return;
359                }
360
361                for (k, &(op0, op1)) in PAULI_PAIRS.iter().enumerate() {
362                    let tab_seed = self.rng.random::<u64>();
363                    let mut tab_branch = tab.fork(Some(tab_seed));
364                    apply(&mut tab_branch, op0, addr0);
365                    apply(&mut tab_branch, op1, addr1);
366                    // X/Y/Z flips only phase bits, so the word-fingerprint is
367                    // preserved and the phase/loss hash is derived incrementally.
368                    let h = pauli_branch_phase_loss(tab, &tab_branch, phase_loss);
369                    branches.push((tab_branch, p_sum.clone() * p[k].clone(), word_fp, h));
370                }
371
372                *p_sum *= T::Coeff::one() - p_total.clone();
373            });
374
375        let needs_normalize = self
376            .entries
377            .insert_or_merge_batch(branches, &self.sum_cutoff);
378        if needs_normalize {
379            self.normalize_probabilities();
380        }
381        self.truncate();
382    }
383}

419impl<
420    T: Config,
421    I: TableauIndex + Send + Sync,
422    C: SparseVector<Complex<T::Coeff>, I>,
423    S: EntryStore<T, I, C>,
424> CorrelatedLossChannel<T> for GeneralizedTableauSum<T, I, C, S>
425where
426    <<T as Config>::Storage as BitView>::Store: PrimInt,
427    C: std::fmt::Debug,
428    T::Coeff: PartialOrd<f64>
429        + PartialOrd
430        + One
431        + Zero
432        + Clone
433        + num::Num
434        + ToPrimitive
435        + std::fmt::Debug
436        + Send
437        + Sync,
438    Complex<T::Coeff>: std::ops::Mul<Output = Complex<T::Coeff>>
439        + From<Complex64>
440        + std::ops::MulAssign
441        + std::ops::AddAssign
442        + One
443        + ComplexFloat
444        + Copy,
445    I: Debug,
446{
447    fn correlated_loss_channel(
448        &mut self,
449        addr0: usize,
450        addr1: usize,
451        p: [<T as Config>::Coeff; 3],
452    ) {
453        let mut branches = Vec::<(GeneralizedTableau<T, I, C>, T::Coeff, u64, u64)>::with_capacity(
454            3 * self.entries.len(),
455        );
456        self.entries
457            .for_each_mut_with_keys(|tab, p_sum, word_fp, phase_loss| {
458                // if either is lost already, we just lose the other with probability p[2]
459                if tab.is_lost[addr0] {
460                    single_qubit_loss_branch(
461                        addr1,
462                        &p[2],
463                        &mut self.rng,
464                        &mut branches,
465                        tab,
466                        p_sum,
467                        (word_fp, phase_loss),
468                    );
469                    return;
470                } else if tab.is_lost[addr1] {
471                    single_qubit_loss_branch(
472                        addr0,
473                        &p[2],
474                        &mut self.rng,
475                        &mut branches,
476                        tab,
477                        p_sum,
478                        (word_fp, phase_loss),
479                    );
480                    return;
481                }
482
483                // if both are present, then we create 3 new branches:
484                // losing both (p[0]), one, or the other qubit (p[1])
485
486                let tab_seed_both = self.rng.random::<u64>();
487                let mut tab_lose_both = tab.fork(Some(tab_seed_both));
488                tab_lose_both.is_lost[addr0] = true;
489                tab_lose_both.is_lost[addr1] = true;
490
491                // is_lost flip leaves the Pauli words and phases unchanged, so
492                // the branch reuses its parent's word-fingerprint and the only
493                // change to the phase/loss hash is the lost qubit's mask.
494                branches.push((
495                    tab_lose_both,
496                    p_sum.clone() * p[0].clone(),
497                    word_fp,
498                    phase_loss ^ loss_mask(addr0) ^ loss_mask(addr1),
499                ));
500
501                let tab_seed_0 = self.rng.random::<u64>();
502                let mut tab_lose_0 = tab.fork(Some(tab_seed_0));
503                tab_lose_0.is_lost[addr0] = true;
504                branches.push((
505                    tab_lose_0,
506                    p_sum.clone() * (p[1].clone() / 2.0.into()),
507                    word_fp,
508                    phase_loss ^ loss_mask(addr0),
509                ));
510
511                let tab_seed_1 = self.rng.random::<u64>();
512                let mut tab_lose_1 = tab.fork(Some(tab_seed_1));
513                tab_lose_1.is_lost[addr1] = true;
514                branches.push((
515                    tab_lose_1,
516                    p_sum.clone() * (p[1].clone() / 2.0.into()),
517                    word_fp,
518                    phase_loss ^ loss_mask(addr1),
519                ));
520
521                let p_total = p[0].clone() + p[1].clone();
522                *p_sum *= T::Coeff::one() - p_total;
523            });
524
525        let needs_renormalize = self
526            .entries
527            .insert_or_merge_batch(branches, &self.sum_cutoff);
528        if needs_renormalize {
529            self.normalize_probabilities();
530        }
531        self.truncate();
532    }
533}

535impl<T, I, C, S> ResetLossChannel<T> for GeneralizedTableauSum<T, I, C, S>
536where
537    T: Config,
538    I: TableauIndex + Send + Sync,
539    C: SparseVector<Complex<T::Coeff>, I>,
540    S: EntryStore<T, I, C>,
541    <<T as Config>::Storage as BitView>::Store: PrimInt,
542    C: std::fmt::Debug,
543    T::Coeff: PartialOrd<f64>
544        + PartialOrd
545        + One
546        + Zero
547        + Clone
548        + num::Num
549        + ToPrimitive
550        + std::fmt::Debug
551        + Send
552        + Sync,
553    Complex<T::Coeff>: std::ops::Mul<Output = Complex<T::Coeff>>
554        + From<Complex64>
555        + std::ops::MulAssign
556        + std::ops::AddAssign
557        + One
558        + ComplexFloat
559        + Copy,
560{
561    fn reset_loss_channel(&mut self, addr0: usize) {
562        let delta = loss_mask(addr0);
563        let mut branches = self.entries.drain_where(|tab| tab.is_lost[addr0]);
564        for (tab, _, _, phase_loss) in branches.iter_mut() {
565            tab.is_lost[addr0] = false;
566            *phase_loss ^= delta;
567        }
568        // reset_loss preserves total probability mass and never drops entries
569        // below sum_cutoff (merges only ever sum existing coefficients), so no
570        // renormalize is needed regardless of insert_or_merge_batch's result.
571        let _ = self
572            .entries
573            .insert_or_merge_batch(branches, &self.sum_cutoff);
574    }
