4use num::Complex;
5use ppvm_tableau::{data::GeneralizedTableau, sparsevec::SparseVector};
6use ppvm_traits::config::Config;
7
8/// One branch produced by a noise channel: its tableau, coefficient, and the
9/// cached `(word_fingerprint, phase_loss_hash)` pair, so a merge can recompute
10/// the full fingerprint (`word_fp ^ phase_loss`) without re-hashing the tableau.
11pub type Branch<T, I, C> = (GeneralizedTableau<T, I, C>, <T as Config>::Coeff, u64, u64);
12
13pub trait EntryStore<T: Config, I, C: SparseVector<Complex<T::Coeff>, I>>: Clone {
14    fn with_capacity(cap: usize) -> Self;
15    fn len(&self) -> usize;
16
17    fn is_empty(&self) -> bool {
18        self.len() == 0
19    }
20
21    fn iter<'a>(&'a self) -> impl Iterator<Item = (&'a GeneralizedTableau<T, I, C>, &'a T::Coeff)>
22    where
23        T: 'a,
24        I: 'a,
25        C: 'a;
26
27    /// Mutate each entry in place. `FnMut` so the closure can capture e.g.
28    /// `&mut Vec<branches>` from noise channels.
29    fn for_each_mut<F>(&mut self, f: F)
30    where
31        F: FnMut(&mut GeneralizedTableau<T, I, C>, &mut T::Coeff);
32
33    /// Like [`for_each_mut`](Self::for_each_mut), but first ensures cached
34    /// fingerprints are current, then passes each entry's `word_fingerprint`
35    /// and `phase_loss_hash` to the closure. Noise channels use this so a
36    /// spawned branch can inherit its parent's word-hash (invariant under
37    /// X/Y/Z and `is_lost`) and incrementally update the phase/loss hash,
38    /// instead of re-hashing the tableau.
39    fn for_each_mut_with_keys<F>(&mut self, f: F)
40    where
41        F: FnMut(&mut GeneralizedTableau<T, I, C>, &mut T::Coeff, u64, u64);
42
43    /// O(1) flag that gates have touched every tableau and cached fingerprints
44    /// are stale. Implementations recompute lazily on the next branching call.
45    fn mark_keys_dirty(&mut self);
46
47    /// Merge each branch into an existing entry whose tableau is structurally
48    /// equal within threshold, or push it as new if its coefficient exceeds
49    /// `cutoff`. Returns true if any incoming branch was dropped (caller
50    /// renormalizes).
51    ///
52    /// Each branch carries its parent's word-fingerprint and its own
53    /// phase/loss hash (third and fourth tuple fields), so the full
54    /// fingerprint is `word_fp ^ phase_loss` — no re-hashing of the tableau.
55    fn insert_or_merge_batch(&mut self, branches: Vec<Branch<T, I, C>>, cutoff: &T::Coeff) -> bool;
56
57    fn retain<F>(&mut self, f: F)
58    where
59        F: FnMut(&GeneralizedTableau<T, I, C>, &T::Coeff) -> bool;
60
61    /// Remove every entry whose tableau matches `pred` and return it together
62    /// with its cached fingerprint components, so the caller can mutate the
63    /// drained entries and re-insert them via [`insert_or_merge_batch`]. For
64    /// storage backends that don't cache `word_fp` and `phase_loss` separately
65    /// (e.g. map-keyed buckets), the split is `(fp, 0)`; either component may
66    /// be XORed against the change delta — the merge sees their XOR.
67    fn drain_where<F>(&mut self, pred: F) -> Vec<Branch<T, I, C>>
68    where
69        F: FnMut(&GeneralizedTableau<T, I, C>) -> bool;
70}
