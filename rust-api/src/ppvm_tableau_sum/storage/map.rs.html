25type Bucket<T, I, C> = SmallVec<[(GeneralizedTableau<T, I, C>, <T as Config>::Coeff); 1]>;
26
27#[derive(Clone)]
28pub struct MapStorage<T: Config, I: TableauIndex, C: SparseVector<Complex<T::Coeff>, I>> {
29    pub buckets: FxHashMap<u64, Bucket<T, I, C>>,
30    pub dirty: bool,
31    /// Reusable scratch buffer for `structurally_equal`'s coefficient lookup
32    /// map. Cleared and refilled per call; keeps its capacity across calls.
33    scratch: FxHashMap<I, Complex<T::Coeff>>,
34}
35
36impl<T, I, C> MapStorage<T, I, C>
37where
38    T: Config,
39    T::Coeff: One + Zero + Clone + num::Num + PartialOrd,
40    Complex<T::Coeff>: std::ops::Mul<Output = Complex<T::Coeff>>
41        + AddAssign
42        + From<Complex64>
43        + ComplexFloat
44        + Copy,
45    I: TableauIndex,
46    C: SparseVector<Complex<T::Coeff>, I>,
47{
48    /// Re-bucket every entry under its fresh fingerprint. Clifford/T gates
49    /// mutate tableau data in place, so after `mark_keys_dirty` the existing
50    /// keys are wrong. Iterates the old map directly to avoid materializing
51    /// the entries into a temporary Vec.
52    fn rebuild_if_dirty(&mut self) {
53        if !self.dirty {
54            return;
55        }
56        let entry_count = self.buckets.values().map(|v| v.len()).sum();
57        let old = std::mem::replace(
58            &mut self.buckets,
59            FxHashMap::with_capacity_and_hasher(entry_count, Default::default()),
60        );
61        for (_, bucket) in old {
62            for (tab, p) in bucket {
63                let fp = fingerprint(&tab);
64                self.buckets.entry(fp).or_default().push((tab, p));
65            }
66        }
67        self.dirty = false;
68    }
69}
70
71impl<T, I, C> EntryStore<T, I, C> for MapStorage<T, I, C>
72where
73    T: Config,
74    T::Coeff: One + Zero + Clone + num::Num + PartialOrd,
75    Complex<T::Coeff>: std::ops::Mul<Output = Complex<T::Coeff>>
76        + AddAssign
77        + From<Complex64>
78        + ComplexFloat
79        + Copy,
80    I: TableauIndex + Send + Sync,
81    C: SparseVector<Complex<T::Coeff>, I>,
82    GeneralizedTableau<T, I, C>: Clifford,
83    <<T as Config>::Storage as BitView>::Store: PrimInt,
84{
85    fn with_capacity(cap: usize) -> Self {
86        Self {
87            buckets: FxHashMap::with_capacity_and_hasher(cap, Default::default()),
88            dirty: false,
89            scratch: FxHashMap::default(),
90        }
91    }
92
93    fn len(&self) -> usize {
94        self.buckets.values().map(|v| v.len()).sum()
95    }
96
97    fn mark_keys_dirty(&mut self) {
98        self.dirty = true;
99    }
100
101    fn iter<'a>(
102        &'a self,
103    ) -> impl Iterator<Item = (&'a GeneralizedTableau<T, I, C>, &'a <T as Config>::Coeff)>
104    where
105        T: 'a,
106        I: 'a,
107        C: 'a,
108    {
109        self.buckets
110            .values()
111            .flat_map(|v| v.iter())
112            .map(|(t, c)| (t, c))
113    }
114
115    fn for_each_mut<F>(&mut self, mut f: F)
116    where
117        F: FnMut(&mut GeneralizedTableau<T, I, C>, &mut <T as Config>::Coeff),
118    {
119        for v in self.buckets.values_mut() {
120            for (tab, c) in v.iter_mut() {
121                f(tab, c);
122            }
123        }
124    }
125
126    fn for_each_mut_with_keys<F>(&mut self, mut f: F)
127    where
128        F: FnMut(&mut GeneralizedTableau<T, I, C>, &mut <T as Config>::Coeff, u64, u64),
129    {
130        self.rebuild_if_dirty();
131        // Build the per-row mask table once; every tableau in the sum shares the
132        // same qubit count. Skip when there are no entries.
133        let masks = self
134            .buckets
135            .values()
136            .flat_map(|v| v.iter())
137            .next()
138            .map(|(t, _)| RowMasks::new(t.is_lost.len()));
139        if let Some(masks) = masks {
140            for v in self.buckets.values_mut() {
141                for (tab, c) in v.iter_mut() {
142                    let word_fp = word_fingerprint(tab);
143                    let phase_loss = phase_loss_hash_with(tab, &masks);
144                    f(tab, c, word_fp, phase_loss);
145                }
146            }
147        }
148    }
149
150    fn insert_or_merge_batch(
151        &mut self,
152        branches: Vec<(GeneralizedTableau<T, I, C>, <T as Config>::Coeff, u64, u64)>,
153        cutoff: &<T as Config>::Coeff,
154    ) -> bool {
155        self.rebuild_if_dirty();
156
157        let mut needs_renormalize = false;
158        for (tab, p, word_fp, phase_loss) in branches {
159            // The branch carries both fingerprint components, so the full
160            // fingerprint is their XOR — no walk over the tableau.
161            let fp = word_fp ^ phase_loss;
162            let bucket = self.buckets.entry(fp).or_default();
163
164            let mut found: Option<usize> = None;
165            for (i, (existing, _)) in bucket.iter().enumerate() {
166                if structurally_equal(existing, &tab, &mut self.scratch) {
167                    found = Some(i);
168                    break;
169                }
170            }
171
172            match found {
173                Some(i) => {
174                    let p0 = &bucket[i].1;
175                    bucket[i].1 = p0.clone() + p;
176                }
177                None => {
178                    if &p > cutoff {
179                        bucket.push((tab, p));
180                    } else {
181                        // Drop the branch. If we just created the bucket via
182                        // `entry().or_default()`, remove it so iter() stays
183                        // clean and len() isn't skewed by empty slots.
184                        if bucket.is_empty() {
185                            self.buckets.remove(&fp);
186                        }
187                        needs_renormalize = true;
188                    }
189                }
190            }
191        }
192
193        needs_renormalize
194    }
195
196    fn insert_or_merge_mutated_branches(
197        &mut self,
198        branches: Vec<(usize, BranchMutation, <T as Config>::Coeff, u64, u64)>,
199        cutoff: &<T as Config>::Coeff,
200    ) -> bool {
201        self.rebuild_if_dirty();
202
203        // Materialize parents in the SAME order as for_each_mut_with_keys so
204        // parent_idx aligns. Correctness-only path: no clone savings here.
205        let parents: Vec<_> = self
206            .buckets
207            .values()
208            .flat_map(|v| v.iter())
209            .map(|(t, _)| t.clone())
210            .collect();
211
212        let real: Vec<Branch<T, I, C>> = branches
213            .into_iter()
214            .map(|(parent_idx, mutation, p, word_fp, phase_loss)| {
215                let mut tab = parents[parent_idx].clone();
216                apply_branch_mutation(&mut tab, mutation);
217                (tab, p, word_fp, phase_loss)
218            })
219            .collect();
220
221        self.insert_or_merge_batch(real, cutoff)
222    }
223
224    fn retain<F>(&mut self, mut f: F)
225    where
226        F: FnMut(&GeneralizedTableau<T, I, C>, &<T as Config>::Coeff) -> bool,
227    {
228        self.buckets.retain(|_, v| {
229            v.retain(|(tab, c)| f(tab, c));
230            !v.is_empty()
231        });
232    }
233
234    fn drain_where<F>(
235        &mut self,
236        mut pred: F,
237    ) -> Vec<(GeneralizedTableau<T, I, C>, <T as Config>::Coeff, u64, u64)>
238    where
239        F: FnMut(&GeneralizedTableau<T, I, C>) -> bool,
240    {
241        self.rebuild_if_dirty();
242
243        let mut drained = Vec::new();
244        // Buckets store entries under the full fingerprint, so the split is
245        // (fp, 0). XORing a delta into either component during the caller's
246        // mutation produces the right new bucket key in insert_or_merge_batch.
247        self.buckets.retain(|&fp, bucket| {
248            let mut i = 0;
249            while i < bucket.len() {
250                if pred(&bucket[i].0) {
251                    let (tab, p) = bucket.swap_remove(i);
252                    drained.push((tab, p, fp, 0));
253                } else {
254                    i += 1;
255                }
256            }
257            !bucket.is_empty()
258        });
259        drained
260    }
261}
