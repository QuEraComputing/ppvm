24#[derive(Clone)]
25pub struct VecStorage<T: Config, I: TableauIndex, C: SparseVector<Complex<T::Coeff>, I>> {
26    pub entries: Vec<(GeneralizedTableau<T, I, C>, T::Coeff)>,
27    pub fingerprints: Vec<u64>,
28    /// Cached `word_fingerprint` per entry, kept in lockstep with `entries`.
29    /// A branch inherits its parent's value (its Pauli words are unchanged),
30    /// so the merge avoids re-hashing the words — the dominant fingerprint cost.
31    pub word_fingerprints: Vec<u64>,
32    /// Cached `phase_loss_hash` per entry, kept in lockstep with `entries`.
33    /// XOR-combinable, so a branch inherits its parent's value and updates only
34    /// the rows it changed — the merge never walks the phases again.
35    pub phase_loss_hashes: Vec<u64>,
36    pub dirty: bool,
37    /// Reusable scratch buffer for `structurally_equal`'s coefficient lookup
38    /// map. Cleared and refilled per call; keeps its capacity across calls.
39    scratch: FxHashMap<I, Complex<T::Coeff>>,
40}
41
42impl<T, I, C> VecStorage<T, I, C>
43where
44    T: Config,
45    T::Coeff: One + Zero + Clone + num::Num + PartialOrd,
46    Complex<T::Coeff>: std::ops::Mul<Output = Complex<T::Coeff>>
47        + AddAssign
48        + From<Complex64>
49        + ComplexFloat
50        + Copy,
51    I: TableauIndex,
52    C: SparseVector<Complex<T::Coeff>, I>,
53{
54    fn insert_or_merge_entry(
55        &mut self,
56        tab: GeneralizedTableau<T, I, C>,
57        p: T::Coeff,
58        word_fp: u64,
59        phase_loss: u64,
60        fp_index: &mut FxHashMap<u64, Vec<usize>>,
61        sum_cutoff: &T::Coeff,
62    ) -> bool {
63        // The branch carries both components — its inherited word-fingerprint
64        // and its incrementally-maintained phase/loss hash — so the full
65        // fingerprint is their XOR, with no walk over the tableau.
66        let fp = word_fp ^ phase_loss;
67
68        // Only run the full equality check on entries whose fingerprint matches.
69        let mut found: Option<usize> = None;
70        if let Some(candidates) = fp_index.get(&fp) {
71            for &i in candidates {
72                if structurally_equal(&self.entries[i].0, &tab, &mut self.scratch) {
73                    found = Some(i);
74                    break;
75                }
76            }
77        }
78
79        let mut needs_normalize = false;
80        match found {
81            Some(i) => {
82                let p0 = &self.entries[i].1;
83                self.entries[i].1 = p0.clone() + p;
84            }
85
86            None => {
87                if &p > sum_cutoff {
88                    let new_idx = self.entries.len();
89                    self.entries.push((tab, p));
90                    self.fingerprints.push(fp);
91                    self.word_fingerprints.push(word_fp);
92                    self.phase_loss_hashes.push(phase_loss);
93                    fp_index.entry(fp).or_default().push(new_idx);
94                } else {
95                    needs_normalize = true;
96                }
97            }
98        }
99
100        needs_normalize
101    }
102
103    /// Recompute `fingerprints` and `word_fingerprints` from scratch when a
104    /// Clifford/T gate has invalidated them. No-op when clean.
105    fn rebuild_fingerprints_if_dirty(&mut self) {
106        if !self.dirty {
107            return;
108        }
109        self.fingerprints.clear();
110        self.word_fingerprints.clear();
111        self.phase_loss_hashes.clear();
112        // Build the per-row mask table once for all entries (every tableau in a
113        // sum shares the same qubit count). Skip when there are no entries.
114        if let Some((first, _)) = self.entries.first() {
115            let masks = RowMasks::new(first.is_lost.len());
116            for (t, _) in self.entries.iter() {
117                let wfp = word_fingerprint(t);
118                let plh = phase_loss_hash_with(t, &masks);
119                self.word_fingerprints.push(wfp);
120                self.phase_loss_hashes.push(plh);
121                self.fingerprints.push(wfp ^ plh);
122            }
123        }
124        self.dirty = false;
125    }
126}

128impl<T, I, C> EntryStore<T, I, C> for VecStorage<T, I, C>
129where
130    T: Config,
131    T::Coeff: One + Zero + Clone + num::Num + PartialOrd,
132    Complex<T::Coeff>: std::ops::Mul<Output = Complex<T::Coeff>>
133        + AddAssign
134        + From<Complex64>
135        + ComplexFloat
136        + Copy,
137    I: TableauIndex + Send + Sync,
138    C: SparseVector<Complex<T::Coeff>, I>,
139    GeneralizedTableau<T, I, C>: Clifford,
140    <<T as Config>::Storage as BitView>::Store: PrimInt,
141{
142    fn with_capacity(cap: usize) -> Self {
143        Self {
144            entries: Vec::with_capacity(cap),
145            fingerprints: Vec::with_capacity(cap),
146            word_fingerprints: Vec::with_capacity(cap),
147            phase_loss_hashes: Vec::with_capacity(cap),
148            dirty: false,
149            scratch: FxHashMap::default(),
150        }
151    }
152
153    fn len(&self) -> usize {
154        self.entries.len()
155    }
156
157    fn mark_keys_dirty(&mut self) {
158        self.dirty = true;
159    }
160
161    fn iter<'a>(
162        &'a self,
163    ) -> impl Iterator<Item = (&'a GeneralizedTableau<T, I, C>, &'a <T as Config>::Coeff)>
164    where
165        T: 'a,
166        I: 'a,
167        C: 'a,
168    {
169        self.entries.iter().map(|(t, c)| (t, c))
170    }
171
172    fn for_each_mut<F>(&mut self, mut f: F)
173    where
174        F: FnMut(&mut GeneralizedTableau<T, I, C>, &mut <T as Config>::Coeff),
175    {
176        self.entries.iter_mut().for_each(|(tab, c)| f(tab, c));
177    }
178
179    fn for_each_mut_with_keys<F>(&mut self, mut f: F)
180    where
181        F: FnMut(&mut GeneralizedTableau<T, I, C>, &mut <T as Config>::Coeff, u64, u64),
182    {
183        self.rebuild_fingerprints_if_dirty();
184        for (i, (tab, c)) in self.entries.iter_mut().enumerate() {
185            f(tab, c, self.word_fingerprints[i], self.phase_loss_hashes[i]);
186        }
187    }
188
189    fn insert_or_merge_batch(
190        &mut self,
191        branches: Vec<(GeneralizedTableau<T, I, C>, <T as Config>::Coeff, u64, u64)>,
192        cutoff: &<T as Config>::Coeff,
193    ) -> bool {
194        self.rebuild_fingerprints_if_dirty();
195
196        // Build a fingerprint index over the current entries so each branch
197        // lookup is O(1) expected instead of O(M) linear scan. Per-entry
198        // fingerprints are cached on `self.fingerprints` and reused across
199        // consecutive noise calls; Clifford/T gates clear the cache.
200        let mut fp_index: FxHashMap<u64, Vec<usize>> =
201            FxHashMap::with_capacity_and_hasher(self.entries.len(), Default::default());
202        for i in 0..self.entries.len() {
203            let fp = self.fingerprints[i];
204            fp_index.entry(fp).or_default().push(i);
205        }
206
207        let mut needs_renormalize = false;
208        for (tab, p, word_fp, phase_loss) in branches {
209            let dropped_any =
210                self.insert_or_merge_entry(tab, p, word_fp, phase_loss, &mut fp_index, cutoff);
211            needs_renormalize |= dropped_any;
212        }
213
214        needs_renormalize
215    }
216
217    fn insert_or_merge_mutated_branches(
218        &mut self,
219        branches: Vec<(usize, BranchMutation, <T as Config>::Coeff, u64, u64)>,
220        cutoff: &<T as Config>::Coeff,
221    ) -> bool {
222        // Defensive: should be a no-op since the caller's for_each_mut_with_keys
223        // already ran and the branch words were never mutated.
224        self.rebuild_fingerprints_if_dirty();
225
226        let mut fp_index: FxHashMap<u64, Vec<usize>> =
227            FxHashMap::with_capacity_and_hasher(self.entries.len(), Default::default());
228        for i in 0..self.entries.len() {
229            let fp = self.fingerprints[i];
230            fp_index.entry(fp).or_default().push(i);
231        }
232
233        let mut needs_renormalize = false;
234        for (parent_idx, mutation, p, word_fp, phase_loss) in branches {
235            let fp = word_fp ^ phase_loss;
236
237            // Find a structurally-equal existing entry among the fp candidates
238            // WITHOUT materializing the branch tableau. The disjoint-field borrow
239            // (`self.entries` immutable + `self.scratch` mutable) is allowed.
240            let mut found: Option<usize> = None;
241            if let Some(candidates) = fp_index.get(&fp) {
242                for &i in candidates {
243                    if structurally_equal_mutated(
244                        &self.entries[i].0,
245                        &self.entries[parent_idx].0,
246                        mutation,
247                        &mut self.scratch,
248                    ) {
249                        found = Some(i);
250                        break;
251                    }
252                }
253            }
254
255            match found {
256                Some(i) => {
257                    let p0 = &self.entries[i].1;
258                    self.entries[i].1 = p0.clone() + p;
259                }
260                None => {
261                    if &p > cutoff {
262                        // Surviving new entry: materialize now (clone parent +
263                        // apply mutation). Later branches' parent_idx still refer
264                        // to the original entries — push never moves them.
265                        let mut tab = self.entries[parent_idx].0.clone();
266                        apply_branch_mutation(&mut tab, mutation);
267                        let new_idx = self.entries.len();
268                        self.entries.push((tab, p));
269                        self.fingerprints.push(fp);
270                        self.word_fingerprints.push(word_fp);
271                        self.phase_loss_hashes.push(phase_loss);
272                        fp_index.entry(fp).or_default().push(new_idx);
273                    } else {
274                        needs_renormalize = true;
275                    }
276                }
277            }
278        }
279
280        needs_renormalize
281    }
282
283    fn retain<F>(&mut self, mut f: F)
284    where
285        F: FnMut(&GeneralizedTableau<T, I, C>, &<T as Config>::Coeff) -> bool,
286    {
287        // Walk entries + fingerprints in lockstep so they stay aligned.
288        // Order-preserving, O(N), no allocation.
289        let mut write = 0;
290        for read in 0..self.entries.len() {
291            let (tab, c) = &self.entries[read];
292            if f(tab, c) {
293                if read != write {
294                    self.entries.swap(read, write);
295                    self.fingerprints.swap(read, write);
296                    self.word_fingerprints.swap(read, write);
297                    self.phase_loss_hashes.swap(read, write);
298                }
299                write += 1;
300            }
301        }
302        self.entries.truncate(write);
303        self.fingerprints.truncate(write);
304        self.word_fingerprints.truncate(write);
305        self.phase_loss_hashes.truncate(write);
306    }
307
308    fn drain_where<F>(
309        &mut self,
310        mut pred: F,
311    ) -> Vec<(GeneralizedTableau<T, I, C>, <T as Config>::Coeff, u64, u64)>
312    where
313        F: FnMut(&GeneralizedTableau<T, I, C>) -> bool,
314    {
315        self.rebuild_fingerprints_if_dirty();
316
317        let mut indices: Vec<usize> = (0..self.entries.len())
318            .filter(|&i| pred(&self.entries[i].0))
319            .collect();
320        // Drain back-to-front so swap_remove indices remain valid.
321        indices.sort_unstable_by(|a, b| b.cmp(a));
322
323        let mut drained = Vec::with_capacity(indices.len());
324        for i in indices {
325            let (tab, p) = self.entries.swap_remove(i);
326            let word_fp = self.word_fingerprints.swap_remove(i);
327            let phase_loss = self.phase_loss_hashes.swap_remove(i);
328            self.fingerprints.swap_remove(i);
329            drained.push((tab, p, word_fp, phase_loss));
330        }
331        drained
332    }
