4//! A minimal scrollable list of lines with one optionally-highlighted row.
5//! Backs the Program panel (highlight = the program counter) and the REPL log
6//! (no highlight). Deliberately mirrors the shape of stellarscope's `CodeView`
7//! so the two stay interchangeable, but has no external dependencies.
8
9/// A list of displayable lines plus an optional cursor (the highlighted row).
10#[derive(Debug, Clone, Default)]
11pub struct CodeView<T> {
12    lines: Vec<T>,
13    cursor: Option<usize>,
14}
15
16impl<T> CodeView<T> {
17    pub fn new() -> Self {
18        Self {
19            lines: Vec::new(),
20            cursor: None,
21        }
22    }
23
24    /// Drop all lines and clear the cursor.
25    pub fn clear(&mut self) {
26        self.lines.clear();
27        self.cursor = None;
28    }
29
30    /// Append one line.
31    pub fn push(&mut self, line: T) {
32        self.lines.push(line);
33    }
34
35    /// Highlight row `idx` (or none).
36    pub fn set_cursor(&mut self, idx: Option<usize>) {
37        self.cursor = idx;
38    }
39
40    pub fn cursor(&self) -> Option<usize> {
41        self.cursor
42    }
43
44    pub fn lines(&self) -> &[T] {
45        &self.lines
46    }
47}
48
49#[cfg(test)]
50mod tests {
51    use super::*;
52
53    #[test]
54    fn push_and_cursor_round_trip() {
55        let mut cv: CodeView<String> = CodeView::new();
56        assert!(cv.lines().is_empty());
57        assert_eq!(cv.cursor(), None);
58
59        cv.push("0000: h".to_string());
60        cv.push("0001: measure".to_string());
61        cv.set_cursor(Some(1));
62
63        assert_eq!(cv.lines().len(), 2);
64        assert_eq!(cv.cursor(), Some(1));
65
66        cv.clear();
67        assert!(cv.lines().is_empty());
68        assert_eq!(cv.cursor(), None);
69    }
70}
