4//! Binary bytecode (`.ssb`) round-trip for PPVM modules.
5//!
6//! Serializes a resolved [`Module`] to a little-endian container — header
7//! (magic + version + device info) → strings section → code section — and
8//! reads it back. The per-instruction codec is the existing `WriteBytes` /
9//! `FromBytes` generated for [`PPVMInstruction`]; this module only frames the
10//! container around it.
11
12use std::io::{Read, Write};
13
14use vihaco::instruction::{FromBytes, WriteBytes};
15
16use crate::PPVMModule;
17use crate::composite::{BackendKind, PPVM_MAGIC, PPVMDeviceInfo, PPVMInstruction};
18
19/// Current `.ssb` format version. The reader rejects any other version.
20pub const PPVM_BYTECODE_VERSION: u16 = 1;
21
22/// Byte length of the fixed portion of the header. The actual `header_size`
23/// in the stream may exceed this when the optional `observable` string is
24/// populated; the reader uses `header_size` to skip to the strings section.
25///
26/// Field widths (bytes): magic(4) + version(2) + header_size(4) + n_qubits(4)
27/// + coefficient_threshold(8) + backend(1) + max_pauli_weight_present(1)
28/// + max_pauli_weight(8) + observable_present(1) = 33.
29const FIXED_HEADER_SIZE: u32 = 4 + 2 + 4 + 4 + 8 + 1 + 1 + 8 + 1;
30
31/// Serialize a resolved module to the v1 `.ssb` byte stream.
32pub fn write_module<W: Write>(module: &PPVMModule, w: &mut W) -> eyre::Result<()> {
33    // v1 serializes only code, strings, and device info. Refuse to silently
34    // drop any table a future feature might populate.
35    let populated = if !module.functions.is_empty() {
36        Some("functions")
37    } else if !module.labels.is_empty() {
38        Some("labels")
39    } else if !module.constants.is_empty() {
40        Some("constants")
41    } else if !module.source_symbols.is_empty() {
42        Some("source_symbols")
43    } else if module.main_function.is_some() {
44        Some("main_function")
45    } else if module.file != 0 {
46        Some("file")
47    } else {
48        None
49    };
50    if let Some(table) = populated {
51        return Err(eyre::eyre!(
52            "bytecode v1 cannot represent a populated `{table}`"
53        ));
54    }
55
56    let info = &module.extra;
57    let n_qubits = u32::try_from(info.n_qubits)
58        .map_err(|_| eyre::eyre!("n_qubits {} does not fit in u32", info.n_qubits))?;
59
60    // The header is `FIXED_HEADER_SIZE` bytes plus, when an observable is
61    // present, a u32 length followed by its UTF-8 bytes.
62    let observable_bytes: &[u8] = info
63        .observable
64        .as_ref()
65        .map(String::as_bytes)
66        .unwrap_or(&[]);
67    let observable_present: u8 = u8::from(info.observable.is_some());
68    let observable_len = u32::try_from(observable_bytes.len()).map_err(|_| {
69        eyre::eyre!(
70            "observable length {} does not fit in u32",
71            observable_bytes.len()
72        )
73    })?;
74    let observable_trailer: u32 = if info.observable.is_some() {
75        4 + observable_len
76    } else {
77        0
78    };
79    let header_size = FIXED_HEADER_SIZE + observable_trailer;
80
81    // Header.
82    w.write_all(&PPVM_MAGIC.to_le_bytes())?;
83    w.write_all(&PPVM_BYTECODE_VERSION.to_le_bytes())?;
84    w.write_all(&header_size.to_le_bytes())?;
85    w.write_all(&n_qubits.to_le_bytes())?;
86    w.write_all(&info.coefficient_threshold.to_le_bytes())?;
87    w.write_all(&[backend_to_u8(info.backend)])?;
88    let (mpw_present, mpw_value) = match info.max_pauli_weight {
89        Some(w) => (
90            1u8,
91            u64::try_from(w)
92                .map_err(|_| eyre::eyre!("max_pauli_weight {} does not fit in u64", w))?,
93        ),
94        None => (0u8, 0u64),
95    };
96    w.write_all(&[mpw_present])?;
97    w.write_all(&mpw_value.to_le_bytes())?;
98    w.write_all(&[observable_present])?;
99    if info.observable.is_some() {
100        w.write_all(&observable_len.to_le_bytes())?;
101        w.write_all(observable_bytes)?;
102    }
103
104    // Strings section: count, then each entry as len-prefixed UTF-8.
105    let string_count =
106        u32::try_from(module.strings.len()).map_err(|_| eyre::eyre!("string count exceeds u32"))?;
107    w.write_all(&string_count.to_le_bytes())?;
108    for s in &module.strings {
109        let len = u32::try_from(s.len()).map_err(|_| eyre::eyre!("string length exceeds u32"))?;
110        w.write_all(&len.to_le_bytes())?;
111        w.write_all(s.as_bytes())?;
112    }
113
114    // Code section: count, then each instruction's fixed-width frame.
115    let code_count =
116        u32::try_from(module.code.len()).map_err(|_| eyre::eyre!("code length exceeds u32"))?;
117    w.write_all(&code_count.to_le_bytes())?;
118    for inst in &module.code {
119        inst.write_bytes(w)?;
120    }
121
122    Ok(())
123}

125/// Reconstruct a module from a v1 `.ssb` byte stream.
126pub fn read_module<R: Read>(r: &mut R) -> eyre::Result<PPVMModule> {
127    // Header.
128    let magic = read_u32(r)?;
129    if magic != PPVM_MAGIC {
130        return Err(eyre::eyre!(
131            "not a PPVM bytecode file (magic 0x{magic:08X})"
132        ));
133    }
134    let version = read_u16(r)?;
135    if version != PPVM_BYTECODE_VERSION {
136        return Err(eyre::eyre!("unsupported bytecode version {version}"));
137    }
138    let header_size = read_u32(r)?;
139    let n_qubits = read_u32(r)? as usize;
140    let coefficient_threshold = read_f64(r)?;
141    let backend = backend_from_u8(read_u8(r)?)?;
142    let mpw_present = read_u8(r)?;
143    let mpw_value = read_u64(r)?;
144    let max_pauli_weight = match mpw_present {
145        0 => None,
146        1 => Some(usize::try_from(mpw_value).map_err(|_| {
147            eyre::eyre!("max_pauli_weight {mpw_value} does not fit in usize on this platform")
148        })?),
149        other => {
150            return Err(eyre::eyre!(
151                "invalid max_pauli_weight presence byte {other}"
152            ));
153        }
154    };
155    let observable_present = read_u8(r)?;
156    let observable = match observable_present {
157        0 => None,
158        1 => {
159            let len = read_u32(r)? as usize;
160            let mut bytes = vec![0u8; len];
161            r.read_exact(&mut bytes)?;
162            Some(String::from_utf8(bytes)?)
163        }
164        other => {
165            return Err(eyre::eyre!("invalid observable presence byte {other}"));
166        }
167    };
168
169    // Sections begin at `header_size`; skip any header bytes beyond what this
170    // reader knows about (forward compat / self-description).
171    let consumed = FIXED_HEADER_SIZE
172        + if observable.is_some() {
173            4 + u32::try_from(observable.as_deref().unwrap().len())
174                .map_err(|_| eyre::eyre!("observable length does not fit in u32"))?
175        } else {
176            0
177        };
178    if header_size < consumed {
179        return Err(eyre::eyre!(
180            "header_size {header_size} smaller than the {consumed} bytes already consumed"
181        ));
182    }
183    skip_bytes(r, u64::from(header_size - consumed))?;
184
185    // Don't pre-allocate from an untrusted count; grow as entries are read.
186    let string_count = read_u32(r)?;
187    let mut strings = Vec::new();
188    for _ in 0..string_count {
189        let len = read_u32(r)? as usize;
190        let mut bytes = vec![0u8; len];
191        r.read_exact(&mut bytes)?;
192        strings.push(String::from_utf8(bytes)?);
193    }
194
195    let code_count = read_u32(r)?;
196    let mut code = Vec::new();
197    for _ in 0..code_count {
198        code.push(PPVMInstruction::from_bytes(r)?);
199    }
200
201    Ok(PPVMModule {
202        extra: PPVMDeviceInfo {
203            magic,
204            n_qubits,
205            coefficient_threshold,
206            backend,
207            observable,
208            max_pauli_weight,
209        },
210        strings,
211        code,
212        ..Default::default()
213    })
214}

216fn backend_to_u8(backend: BackendKind) -> u8 {
217    match backend {
218        BackendKind::Tableau => 0,
219        BackendKind::PauliSum => 1,
220        BackendKind::LossyPauliSum => 2,
221    }
222}
223
224fn backend_from_u8(byte: u8) -> eyre::Result<BackendKind> {
225    match byte {
226        0 => Ok(BackendKind::Tableau),
227        1 => Ok(BackendKind::PauliSum),
228        2 => Ok(BackendKind::LossyPauliSum),
229        other => Err(eyre::eyre!("invalid backend tag {other}")),
230    }
231}

233/// Serialize a module to an owned byte vector.
234pub fn module_to_bytes(module: &PPVMModule) -> eyre::Result<Vec<u8>> {
235    let mut buf = Vec::new();
236    write_module(module, &mut buf)?;
237    Ok(buf)
238}
239
240/// Reconstruct a module from a byte slice.
241pub fn module_from_bytes(bytes: &[u8]) -> eyre::Result<PPVMModule> {
242    read_module(&mut &bytes[..])
243}
244
245/// Cheap sniff: does this byte stream begin with the PPVM `.ssb` magic?
246///
247/// Reads the leading four bytes the same way [`read_module`] does — as a
248/// little-endian `u32` — so a positive result here means [`read_module`] will
249/// accept the magic. A stream shorter than the magic is not bytecode.
250pub fn is_bytecode(bytes: &[u8]) -> bool {
251    bytes.len() >= 4 && u32::from_le_bytes([bytes[0], bytes[1], bytes[2], bytes[3]]) == PPVM_MAGIC
252}
253
254/// "Dump": compile `.sst` source straight to the `.ssb` byte stream.
255pub fn compile_to_bytes(source: &str) -> eyre::Result<Vec<u8>> {
256    let module = crate::compile_program(source)?;
257    module_to_bytes(&module)
258}

260fn read_u8<R: Read>(r: &mut R) -> eyre::Result<u8> {
261    let mut b = [0u8; 1];
262    r.read_exact(&mut b)?;
263    Ok(b[0])
264}
265
266fn read_u16<R: Read>(r: &mut R) -> eyre::Result<u16> {
267    let mut b = [0u8; 2];
268    r.read_exact(&mut b)?;
269    Ok(u16::from_le_bytes(b))
270}
271
272fn read_u32<R: Read>(r: &mut R) -> eyre::Result<u32> {
273    let mut b = [0u8; 4];
274    r.read_exact(&mut b)?;
275    Ok(u32::from_le_bytes(b))
276}
277
278fn read_u64<R: Read>(r: &mut R) -> eyre::Result<u64> {
279    let mut b = [0u8; 8];
280    r.read_exact(&mut b)?;
281    Ok(u64::from_le_bytes(b))
282}
283
284fn read_f64<R: Read>(r: &mut R) -> eyre::Result<f64> {
285    let mut b = [0u8; 8];
286    r.read_exact(&mut b)?;
287    Ok(f64::from_le_bytes(b))
288}
289
290fn skip_bytes<R: Read>(r: &mut R, n: u64) -> eyre::Result<()> {
291    let skipped = std::io::copy(&mut r.take(n), &mut std::io::sink())?;
292    if skipped != n {
293        return Err(eyre::eyre!("unexpected EOF skipping {n} header bytes"));
294    }
295    Ok(())
296}
