20/// Largest qubit count any backend can simulate. The widest size bucket is
21/// backed by 2048-bit integers (`U2048` / `[u8; 256]`), so every constructor
22/// rejects `n_qubits > MAX_QUBITS` rather than panicking.
23pub const MAX_QUBITS: usize = 2048;
24
25/// Truncation strategy used by every `PauliSum` / `LossyPauliSum` size bucket.
26/// Coefficient-threshold pruning is always on; the Pauli-weight cap is set per
27/// run from the header (defaults to `usize::MAX` = no cap).
28type PauliSumStrategy = CombinedStrategy<CoefficientThreshold, MaxPauliWeight>;
29
30/// `PauliSum<T>`'s `T` for the lossless backend: `[u8; N]` storage, fx hash,
31/// f64 coefficients, the strategy above.
32type PauliSumConfig<const N: usize> = ByteFxHashF64<N, PauliSumStrategy>;
33
34/// Same as `PauliSumConfig` but with `LossyPauliWord` as the word type, so the
35/// loss-channel methods are dispatchable on the resulting `PauliSum<T>`.
36/// `LossyPauliWord`'s second type parameter (hasher) defaults to
37/// `fxhash::FxBuildHasher`, matching `ByteFxHashF64`'s internal hasher.
38type LossyPauliSumConfig<const N: usize> =
39    ByteFxHashF64<N, PauliSumStrategy, LossyPauliWord<[u8; N]>>;
40
41/// Build a `PauliSumStrategy` value from a `PPVMDeviceInfo`. Pulled out so the
42/// six size-bucket constructors don't each repeat the strategy spelling.
43fn paulisum_strategy(info: &PPVMDeviceInfo) -> PauliSumStrategy {
44    CombinedStrategy(
45        CoefficientThreshold(info.coefficient_threshold),
46        MaxPauliWeight(info.max_pauli_weight.unwrap_or(usize::MAX)),
47    )
48}

50macro_rules! batch_for {
51    ($tab:expr, $method:ident, $addrs:expr) => {
52        for addr in $addrs { $tab.$method(*addr); }
53    };
54    ($tab:expr, $method:ident, $addrs:expr, $($arg:expr),+) => {
55        for addr in $addrs { $tab.$method(*addr, $($arg),+); }
56    };
57}
58
59/// Two-qubit sibling of [`batch_for!`]: drives a method that takes two qubit
60/// addresses (plus optional extra args) over a slice of `(usize, usize)` pairs.
61macro_rules! batch_pairs_for {
62    ($state:expr, $method:ident, $pairs:expr) => {
63        for &(a, b) in $pairs { $state.$method(a, b); }
64    };
65    ($state:expr, $method:ident, $pairs:expr, $($arg:expr),+) => {
66        for &(a, b) in $pairs { $state.$method(a, b, $($arg),+); }
67    };
68}

74#[component(instruction = CircuitInstruction, message = CircuitMessage, effect = CircuitOutcomeEffect)]
75impl<T, I, C> CircuitExecutor<T, I, C>
76where
77    T: Config<Coeff = f64>,
78    <<T as Config>::Storage as BitView>::Store: PrimInt,
79    I: TableauIndex + Send + Sync + std::fmt::Debug,
80    C: SparseVector<Complex64, I> + std::fmt::Debug,
81{
82    fn execute(
83        &mut self,
84        inst: CircuitInstruction,
85        msg: CircuitMessage,
86    ) -> Result<Effects<CircuitOutcomeEffect>> {
87        self.execute_instruction(&inst, &msg)
88    }
89
90    fn execute_instruction(
91        &mut self,
92        inst: &CircuitInstruction,
93        msg: &CircuitMessage,
94    ) -> Result<Effects<CircuitOutcomeEffect>> {
95        use CircuitInstruction::*;
96        use CircuitMessage::*;
97
98        match (inst, msg) {
99            // Single-qubit Clifford
100            (X, &Qubit(addr)) => self.tab.x(addr),
101            (Y, &Qubit(addr)) => self.tab.y(addr),
102            (Z, &Qubit(addr)) => self.tab.z(addr),
103            (H, &Qubit(addr)) => self.tab.h(addr),
104            (S, &Qubit(addr)) => self.tab.s(addr),
105            (SAdj, &Qubit(addr)) => self.tab.s_dag(addr),
106            (SqrtX, &Qubit(addr)) => self.tab.sqrt_x(addr),
107            (SqrtY, &Qubit(addr)) => self.tab.sqrt_y(addr),
108            (SqrtXAdj, &Qubit(addr)) => self.tab.sqrt_x_dag(addr),
109            (SqrtYAdj, &Qubit(addr)) => self.tab.sqrt_y_dag(addr),
110
111            // Controlled gates
112            (CNOT, &TwoQubit(addr0, addr1)) => self.tab.cnot(addr0, addr1),
113            (CZ, &TwoQubit(addr0, addr1)) => self.tab.cz(addr0, addr1),
114
115            // T gate
116            (T, &Qubit(addr)) => self.tab.t(addr),
117            (TAdj, &Qubit(addr)) => self.tab.t_dag(addr),
118
119            // Single-qubit rotations
120            (RX, &QubitAndFloat(addr, angle)) => self.tab.rx(addr, angle),
121            (RY, &QubitAndFloat(addr, angle)) => self.tab.ry(addr, angle),
122            (RZ, &QubitAndFloat(addr, angle)) => self.tab.rz(addr, angle),
123
124            // Two-qubit rotations
125            (RXX, &TwoQubitAndFloat(addr0, addr1, angle)) => self.tab.rxx(addr0, addr1, angle),
126            (RYY, &TwoQubitAndFloat(addr0, addr1, angle)) => self.tab.ryy(addr0, addr1, angle),
127            (RZZ, &TwoQubitAndFloat(addr0, addr1, angle)) => self.tab.rzz(addr0, addr1, angle),
128
129            // U3
130            (U3, &QubitU3(addr, theta, phi, lam)) => self.tab.u3(addr, theta, phi, lam),
131
132            // RXY: rotation about an axis in the x/y plane
133            (R, &QubitAndTwoFloats(addr, axis_angle, theta)) => self.tab.r(addr, axis_angle, theta),
134
135            // Measure & Reset
136            (Measure, &Qubit(addr)) => {
137                let outcome: MeasurementOutcome = self.tab.measure(addr).into();
138                return Ok(Effects::one(CircuitOutcomeEffect::Measurement(
139                    MeasurementEffect {
140                        measurement_results: smallvec::smallvec![outcome],
141                    },
142                )));
143            }
144            (Reset, &Qubit(addr)) => self.tab.reset(addr),
145
146            // Noise
147            (Depolarize, &QubitAndFloat(addr, p)) => self.tab.depolarize1(addr, p),
148            (Depolarize2, &TwoQubitAndFloat(addr0, addr1, p)) => {
149                self.tab.depolarize2(addr0, addr1, p)
150            }
151            (PauliError, QubitAndFloatArr3(addr0, ps)) => self.tab.pauli_error(*addr0, *ps),
152            (TwoQubitPauliError, TwoQubitAndFloatArr15(addr0, addr1, ps)) => {
153                self.tab.two_qubit_pauli_error(*addr0, *addr1, *ps)
154            }
155
156            // Loss
157            (Loss, &QubitAndFloat(addr, p)) => self.tab.loss_channel(addr, p),
158            (CorrelatedLoss, TwoQubitAndFloatArr3(addr0, addr1, ps)) => {
159                self.tab.correlated_loss_channel(*addr0, *addr1, *ps)
160            }
161
162            /* BATCH OPERATIONS START HERE */
163            // Batch: dedicated batch methods
164            (SqrtX, QubitBatch(addrs)) => self.tab.sqrt_x_many(addrs),
165            (SqrtY, QubitBatch(addrs)) => self.tab.sqrt_y_many(addrs),
166            (SqrtXAdj, QubitBatch(addrs)) => self.tab.sqrt_x_dag_many(addrs),
167            (SqrtYAdj, QubitBatch(addrs)) => self.tab.sqrt_y_dag_many(addrs),
168            (H, QubitBatch(addrs)) => self.tab.h_many(addrs),
169            (CZ, TwoQubitBatch(pairs)) => self.tab.cz_many(pairs),
170
171            // TODO: replace things below by actual batched methods once they are available
172            // Batch: single-qubit for loops
173            (X, QubitBatch(addrs)) => self.tab.x_many(addrs),
174            (Y, QubitBatch(addrs)) => self.tab.y_many(addrs),
175            (Z, QubitBatch(addrs)) => self.tab.z_many(addrs),
176            (S, QubitBatch(addrs)) => self.tab.s_many(addrs),
177            (SAdj, QubitBatch(addrs)) => self.tab.s_dag_many(addrs),
178            (T, QubitBatch(addrs)) => self.tab.t_many(addrs),
179            (TAdj, QubitBatch(addrs)) => self.tab.t_dag_many(addrs),
180            (Reset, QubitBatch(addrs)) => self.tab.reset_many(addrs),
181            (RX, QubitBatchAndFloat(addrs, angle)) => self.tab.rx_many(addrs, *angle),
182            (RY, QubitBatchAndFloat(addrs, angle)) => self.tab.ry_many(addrs, *angle),
183            (RZ, QubitBatchAndFloat(addrs, angle)) => self.tab.rz_many(addrs, *angle),
184            (Depolarize, QubitBatchAndFloat(addrs, p)) => {
185                batch_for!(self.tab, depolarize1, addrs, *p)
186            }
187            (Loss, QubitBatchAndFloat(addrs, p)) => batch_for!(self.tab, loss_channel, addrs, *p),
188            (PauliError, QubitBatchAndFloatArr3(addrs, ps)) => {
189                batch_for!(self.tab, pauli_error, addrs, *ps)
190            }
191            (U3, QubitBatchU3(addrs, theta, phi, lam)) => {
192                batch_for!(self.tab, u3, addrs, *theta, *phi, *lam)
193            }
194
195            // Batch: two-qubit for loops
196            (CNOT, TwoQubitBatch(pairs)) => {
197                self.tab.cnot_many(pairs);
198            }
199            (RXX, TwoQubitBatchAndFloat(pairs, angle)) => {
200                self.tab.rxx_many(pairs, *angle);
201            }
202            (RYY, TwoQubitBatchAndFloat(pairs, angle)) => {
203                self.tab.ryy_many(pairs, *angle);
204            }
205            (RZZ, TwoQubitBatchAndFloat(pairs, angle)) => {
206                self.tab.rzz_many(pairs, *angle);
207            }
208            (Depolarize2, TwoQubitBatchAndFloat(pairs, p)) => {
209                for &(a, b) in pairs {
210                    self.tab.depolarize2(a, b, *p);
211                }
212            }
213            (TwoQubitPauliError, TwoQubitBatchAndFloatArr15(pairs, ps)) => {
214                for &(a, b) in pairs {
215                    self.tab.two_qubit_pauli_error(a, b, *ps);
216                }
217            }
218            (CorrelatedLoss, TwoQubitBatchAndFloatArr3(pairs, ps)) => {
219                for &(a, b) in pairs {
220                    self.tab.correlated_loss_channel(a, b, *ps);
221                }
222            }
223
224            // Batch: measure (emits per qubit)
225            (Measure, QubitBatch(addrs)) => {
226                let outcomes = self.tab.measure_many(addrs);
227                return Ok(Effects::one(CircuitOutcomeEffect::Measurement(
228                    MeasurementEffect {
229                        measurement_results: outcomes.iter().map(|&o| o.into()).collect(),
230                    },
231                )));
232            }
233
234            // Truncate is a silent no-op on the Tableau backend — the tableau's
235            // gate methods already prune via the configured coefficient
236            // threshold, so there's nothing extra to do here.
237            (Truncate, None) => {}
238
239            // Trace: parse the resolved pattern and compute Σ_{P matches} ⟨ψ|P|ψ⟩
240            // on the tableau state. Asymmetric with the PauliSum semantics by
241            // design (Decision 9): on the tableau this is a sum of expectations,
242            // not a coefficient filter.
243            (Trace, PauliPatternStr(s)) => {
244                let pat = PauliPattern::parse(s)
245                    .map_err(|e| eyre!("invalid Pauli pattern `{s}`: {e:?}"))?;
246                let value = self.tab.trace(&pat);
247                return Ok(Effects::one(CircuitOutcomeEffect::Trace(TraceEffect {
248                    value,
249                })));
250            }
251
252            // Fallback
253            (inst, msg) => {
254                return Err(eyre!(
255                    "Invalid circuit instruction arguments {:?} for instruction {:?}",
256                    msg,
257                    inst
258                ));
259            }
260        };
261
262        Ok(Effects::None)
263    }
264}

278/// Shared dispatch body for `PauliSumExecutor` and `LossyPauliSumExecutor`.
279/// Every non-loss `CircuitInstruction` lands here. `LossyPauliSumExecutor`
280/// matches `Loss` / `CorrelatedLoss` (single + batched) before invoking this
281/// macro and never reaches the loss-rejection arm below.
282///
283/// `$self` is passed as an `ident` (typically `self`) so the macro's
284/// expansion shares hygiene with the surrounding method's `self` parameter.
285/// `$inst` / `$msg` are passed the same way; `$backend` is the human-readable
286/// backend name baked into error messages.
287macro_rules! dispatch_common_paulisum {
288    ($self:ident, $inst:ident, $msg:ident, $backend:literal) => {{
289        use CircuitInstruction::*;
290        use CircuitMessage::*;
291        match ($inst, $msg) {
292            // Single-qubit Clifford
293            (X, &Qubit(addr)) => $self.state.x(addr),
294            (Y, &Qubit(addr)) => $self.state.y(addr),
295            (Z, &Qubit(addr)) => $self.state.z(addr),
296            (H, &Qubit(addr)) => $self.state.h(addr),
297            (S, &Qubit(addr)) => $self.state.s(addr),
298            (SAdj, &Qubit(addr)) => $self.state.s_dag(addr),
299            (SqrtX, &Qubit(addr)) => $self.state.sqrt_x(addr),
300            (SqrtY, &Qubit(addr)) => $self.state.sqrt_y(addr),
301            (SqrtXAdj, &Qubit(addr)) => $self.state.sqrt_x_dag(addr),
302            (SqrtYAdj, &Qubit(addr)) => $self.state.sqrt_y_dag(addr),
303
304            // Controlled gates
305            (CNOT, &TwoQubit(addr0, addr1)) => $self.state.cnot(addr0, addr1),
306            (CZ, &TwoQubit(addr0, addr1)) => $self.state.cz(addr0, addr1),
307
308            // Single-qubit rotations
309            (RX, &QubitAndFloat(addr, angle)) => $self.state.rx(addr, angle),
310            (RY, &QubitAndFloat(addr, angle)) => $self.state.ry(addr, angle),
311            (RZ, &QubitAndFloat(addr, angle)) => $self.state.rz(addr, angle),
312
313            // Two-qubit rotations
314            (RXX, &TwoQubitAndFloat(addr0, addr1, angle)) => {
315                $self.state.rxx(addr0, addr1, angle)
316            }
317            (RYY, &TwoQubitAndFloat(addr0, addr1, angle)) => {
318                $self.state.ryy(addr0, addr1, angle)
319            }
320            (RZZ, &TwoQubitAndFloat(addr0, addr1, angle)) => {
321                $self.state.rzz(addr0, addr1, angle)
322            }
323
324            // RXY: rotation about an axis in the x/y plane
325            (R, &QubitAndTwoFloats(addr, axis_angle, theta)) => {
326                $self.state.r(addr, axis_angle, theta)
327            }
328
329            // Noise
330            (Depolarize, &QubitAndFloat(addr, p)) => $self.state.depolarize1(addr, p),
331            (Depolarize2, &TwoQubitAndFloat(addr0, addr1, p)) => {
332                $self.state.depolarize2(addr0, addr1, p)
333            }
334            (PauliError, QubitAndFloatArr3(addr0, ps)) => {
335                $self.state.pauli_error(*addr0, *ps)
336            }
337            (TwoQubitPauliError, TwoQubitAndFloatArr15(addr0, addr1, ps)) => {
338                $self.state.two_qubit_pauli_error(*addr0, *addr1, *ps)
339            }
340
341            // Truncate: pruning per the configured strategy.
342            (Truncate, None) => $self.state.truncate(),
343
344            // Batched arms: simple for-loop dispatch (no dedicated batch
345            // methods on PauliSum<T>, unlike GeneralizedTableau).
346            (X, QubitBatch(addrs)) => $self.state.x_many(addrs),
347            (Y, QubitBatch(addrs)) => $self.state.y_many(addrs),
348            (Z, QubitBatch(addrs)) => $self.state.z_many(addrs),
349            (H, QubitBatch(addrs)) => $self.state.h_many(addrs),
350            (S, QubitBatch(addrs)) => $self.state.s_many(addrs),
351            (SAdj, QubitBatch(addrs)) => $self.state.s_dag_many(addrs),
352            (SqrtX, QubitBatch(addrs)) => $self.state.sqrt_x_many(addrs),
353            (SqrtY, QubitBatch(addrs)) => $self.state.sqrt_y_many(addrs),
354            (SqrtXAdj, QubitBatch(addrs)) => $self.state.sqrt_x_dag_many(addrs),
355            (SqrtYAdj, QubitBatch(addrs)) => $self.state.sqrt_y_dag_many(addrs),
356            (RX, QubitBatchAndFloat(addrs, angle)) => {
357                $self.state.rx_many(addrs, *angle)
358            }
359            (RY, QubitBatchAndFloat(addrs, angle)) => {
360                $self.state.ry_many(addrs, *angle)
361            }
362            (RZ, QubitBatchAndFloat(addrs, angle)) => {
363                $self.state.rz_many(addrs, *angle)
364            }
365            (Depolarize, QubitBatchAndFloat(addrs, p)) => {
366                batch_for!($self.state, depolarize1, addrs, *p)
367            }
368            (PauliError, QubitBatchAndFloatArr3(addrs, ps)) => {
369                batch_for!($self.state, pauli_error, addrs, *ps)
370            }
371            (CNOT, TwoQubitBatch(pairs)) => $self.state.cnot_many(pairs),
372            (CZ, TwoQubitBatch(pairs)) => $self.state.cz_many(pairs),
373            (RXX, TwoQubitBatchAndFloat(pairs, angle)) => {
374                $self.state.rxx_many(pairs, *angle)
375            }
376            (RYY, TwoQubitBatchAndFloat(pairs, angle)) => {
377                $self.state.ryy_many(pairs, *angle)
378            }
379            (RZZ, TwoQubitBatchAndFloat(pairs, angle)) => {
380                $self.state.rzz_many(pairs, *angle)
381            }
382            (Depolarize2, TwoQubitBatchAndFloat(pairs, p)) => {
383                batch_pairs_for!($self.state, depolarize2, pairs, *p)
384            }
385            (TwoQubitPauliError, TwoQubitBatchAndFloatArr15(pairs, ps)) => {
386                batch_pairs_for!($self.state, two_qubit_pauli_error, pairs, *ps)
387            }
388
389            // Not supported on either backend (Decision 11 + Gate Support
390            // Matrix). Loss / CorrelatedLoss handling differs by backend
391            // and lives in the caller's impl block, not this macro.
392            (Measure | Reset, _) => {
393                return Err(eyre!("{} is not supported on the {} backend", $inst, $backend));
394            }
395
396            (T, Qubit(addr)) => $self.state.rz(*addr, std::f64::consts::PI / 8.0),
397            (T, QubitBatch(addrs)) => $self.state.rz_many(addrs, std::f64::consts::PI / 8.0),
398            (TAdj, Qubit(addr)) => $self.state.rz(*addr, -std::f64::consts::PI / 8.0),
399            (TAdj, QubitBatch(addrs)) => $self.state.rz_many(addrs, -std::f64::consts::PI / 8.0),
400
401            // U3(θ, φ, λ) = RZ(φ)·RY(θ)·RZ(λ), matching the tableau / Python
402            // `u3` (which applies rz(λ); ry(θ); rz(φ)). PauliSum is the
403            // Heisenberg picture, so the sub-rotations run in reversed order
404            // with the same angles: rz(φ); ry(θ); rz(λ).
405            (U3, &QubitU3(addr, theta, phi, lam)) => {
406                $self.state.rz(addr, phi);
407                $self.state.ry(addr, theta);
408                $self.state.rz(addr, lam);
409            }
410
411            (U3, QubitBatchU3(addrs, theta, phi, lam)) => {
412                $self.state.rz_many(addrs, *phi);
413                $self.state.ry_many(addrs, *theta);
414                $self.state.rz_many(addrs, *lam);
415            }
416
417            // Trace: parse the resolved pattern string and compute the
418            // trace. Per plan Decision 9, parsing happens on every
419            // execution; no module-load caching.
420            (Trace, PauliPatternStr(s)) => {
421                let pat = PauliPattern::parse(s)
422                    .map_err(|e| eyre!("invalid Pauli pattern `{}`: {:?}", s, e))?;
423                let value = $self.state.trace(&pat);
424                return Ok(Effects::one(CircuitOutcomeEffect::Trace(TraceEffect {
425                    value,
426                })));
427            }
428
429            // Fallback (mismatched shapes, etc.)
430            (inst, msg) => {
431                return Err(eyre!(
432                    "Invalid circuit instruction arguments {:?} for instruction {:?} on the {} backend",
433                    msg,
434                    inst,
435                    $backend
436                ));
437            }
438        };
439        Ok(Effects::None)
440    }};
441}

443/// PauliSum-backed executor (Heisenberg picture). Holds a `PauliSum<T>` and
444/// answers the same `CircuitInstruction` vocabulary as `CircuitExecutor`,
445/// but without measurement / reset / loss support.
446pub struct PauliSumExecutor<T: Config<Coeff = f64>> {
447    pub state: PauliSum<T>,
448    /// Snapshot of the seeded observable, restored by `reset`.
449    initial: PauliSum<T>,
450}
451
452#[component(instruction = CircuitInstruction, message = CircuitMessage, effect = CircuitOutcomeEffect)]
453impl<T> PauliSumExecutor<T>
454where
455    T: Config<Coeff = f64>,
456    for<'a> PauliSum<T>: Trace<'a, PauliPattern, Output = f64>,
457{
458    fn execute(
459        &mut self,
460        inst: CircuitInstruction,
461        msg: CircuitMessage,
462    ) -> Result<Effects<CircuitOutcomeEffect>> {
463        self.execute_instruction(&inst, &msg)
464    }
465
466    fn execute_instruction(
467        &mut self,
468        inst: &CircuitInstruction,
469        msg: &CircuitMessage,
470    ) -> Result<Effects<CircuitOutcomeEffect>> {
471        use CircuitInstruction::*;
472
473        if matches!(inst, Loss | CorrelatedLoss) {
474            return Err(eyre!(
475                "{inst} is not supported on the PauliSum backend; use the LossyPauliSum backend instead"
476            ));
477        }
478
479        dispatch_common_paulisum!(self, inst, msg, "PauliSum")
480    }
481}

484where
485    T: Config<Coeff = f64>,
486    PauliSum<T>: Clone,
487{
488    fn reset(&mut self) {
489        self.state = self.initial.clone();
490    }

493/// LossyPauliSum-backed executor. Same dispatch as `PauliSumExecutor` plus
494/// `Loss` / `CorrelatedLoss` channels. The concrete `T` used by the
495/// enclosing `Circuit::LossyPauliSum` variant is a `Config` whose
496/// `PauliWordType` is `LossyPauliWord` (see `LossyPauliSumConfig`).
497pub struct LossyPauliSumExecutor<T: Config<Coeff = f64>> {
498    pub state: PauliSum<T>,
499    /// Snapshot of the seeded observable, restored by `reset`.
500    initial: PauliSum<T>,
501}
502
503#[component(instruction = CircuitInstruction, message = CircuitMessage, effect = CircuitOutcomeEffect)]
504impl<T> LossyPauliSumExecutor<T>
505where
506    T: Config<Coeff = f64>,
507    for<'a> PauliSum<T>: Trace<'a, PauliPattern, Output = f64>,
508{
509    fn execute(
510        &mut self,
511        inst: CircuitInstruction,
512        msg: CircuitMessage,
513    ) -> Result<Effects<CircuitOutcomeEffect>> {
514        self.execute_instruction(&inst, &msg)
515    }
516
517    fn execute_instruction(
518        &mut self,
519        inst: &CircuitInstruction,
520        msg: &CircuitMessage,
521    ) -> Result<Effects<CircuitOutcomeEffect>> {
522        use CircuitInstruction::*;
523        use CircuitMessage::*;
524
525        // Loss / CorrelatedLoss are the only instructions that differ from
526        // PauliSum; handle them here then delegate everything else to the
527        // shared dispatch.
528        match (inst, msg) {
529            (Loss, &QubitAndFloat(addr, p)) => {
530                self.state.loss_channel(addr, p);
531                return Ok(Effects::None);
532            }
533            (CorrelatedLoss, TwoQubitAndFloatArr3(addr0, addr1, ps)) => {
534                self.state.correlated_loss_channel(*addr0, *addr1, *ps);
535                return Ok(Effects::None);
536            }
537            (Loss, QubitBatchAndFloat(addrs, p)) => {
538                batch_for!(self.state, loss_channel, addrs, *p);
539                return Ok(Effects::None);
540            }
541            (CorrelatedLoss, TwoQubitBatchAndFloatArr3(pairs, ps)) => {
542                batch_pairs_for!(self.state, correlated_loss_channel, pairs, *ps);
543                return Ok(Effects::None);
544            }
545            _ => {}
546        }
547
548        dispatch_common_paulisum!(self, inst, msg, "LossyPauliSum")
549    }
550}

553where
554    T: Config<Coeff = f64>,
555    PauliSum<T>: Clone,
556{
557    fn reset(&mut self) {
558        self.state = self.initial.clone();
559    }

668/// PauliSum-backed inner enum (Heisenberg picture). Per Decision 7 of the plan,
669/// the size buckets carry `[u8; N]`-storage `ByteFxHashF64` configs (N = 8, 16,
670/// …, 256) rather than the tableau's `[u64; N]` configs; bucket labels match
671/// the semantic qubit count (`Bits64` = 64 qubits) so the outer enum's dispatch
672/// is uniform across backends.
673pub enum PauliSumCircuit {
674    Bits64(PauliSumExecutor<PauliSumConfig<8>>),
675    Bits128(PauliSumExecutor<PauliSumConfig<16>>),
676    Bits256(PauliSumExecutor<PauliSumConfig<32>>),

682impl PauliSumCircuit {
683    /// Build a PauliSum-backed circuit, seeding the state with every term:
684    /// `for (word, coef) in terms { state += (word, coef); }`. Words must
685    /// already be validated against `info.n_qubits` by the caller.

690                    .n_qubits(info.n_qubits)
691                    .strategy(paulisum_strategy(info))
692                    .build();
693                for (word, coef) in terms {
694                    state += (word.as_str(), *coef);

696                let initial = state.clone();

698            }};
699        }
700        if info.n_qubits <= 64 {
701            build!(Bits64, 8)
702        } else if info.n_qubits <= 128 {
703            build!(Bits128, 16)
704        } else if info.n_qubits <= 256 {
705            build!(Bits256, 32)
706        } else if info.n_qubits <= 512 {
707            build!(Bits512, 64)
708        } else if info.n_qubits <= 1024 {
709            build!(Bits1024, 128)
710        } else if info.n_qubits <= MAX_QUBITS {
711            build!(Bits2048, 256)
712        } else {
713            Err(eyre!(
714                "cannot simulate {} qubits: maximum is {MAX_QUBITS}",
715                info.n_qubits
716            ))
717        }
718    }

721        &mut self,
722        inst: &CircuitInstruction,

731            Self::Bits2048(ex) => ex.execute_instruction(inst, msg),

849/// Outer `Circuit` enum: backend selector. Picks one of the three inner enums

861    /// `coefficient_threshold` from `info`; no observable required.
