4//! Device configuration for the circuit component. Split into its own module
5//! so [`crate::component`] can depend on [`PPVMDeviceInfo`] without pulling in
6//! the composite machine (which is added in a later PR).
7
8pub const PPVM_MAGIC: u32 = 0x5050564D;
9
10/// Which execution backend the circuit runs on. Selected via the
11/// `device circuit.backend` header; defaults to `Tableau` so existing
12/// programs that don't declare a backend keep working.
13#[derive(Debug, Clone, Copy, PartialEq, Eq, Default, vihaco_parser::Parse)]
14pub enum BackendKind {
15    #[default]
16    Tableau,
17    PauliSum,
18    #[token = "lossy_paulisum"]
19    LossyPauliSum,
20}
21
22#[derive(Debug, Clone, PartialEq)]
23pub struct PPVMDeviceInfo {
24    pub magic: u32,
25    pub n_qubits: usize,
26    pub coefficient_threshold: f64,
27    pub backend: BackendKind,
28    pub observable: Option<String>,
29    pub max_pauli_weight: Option<usize>,
30}
31
32impl Default for PPVMDeviceInfo {
33    fn default() -> Self {
34        Self {
35            magic: PPVM_MAGIC,
36            n_qubits: 0,
37            coefficient_threshold: 1e-10,
38            backend: BackendKind::default(),
39            observable: None,
40            max_pauli_weight: None,
41        }
42    }
43}
