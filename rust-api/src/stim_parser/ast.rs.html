172/// Required argument-count rule for a Stim instruction.
173#[derive(Debug, Clone, Copy, PartialEq, Eq)]
174pub enum ArgCount {
175    /// Any arg count is accepted.
176    Any,
177    /// No args allowed. `(…)` parens must be absent.
178    None,
179    /// Exactly `n` args.
180    Exact(usize),
181    /// Either no args or exactly `n` args. Used by Stim measurement
182    /// instructions, where the optional single arg is the readout-flip
183    /// probability.
184    Optional(usize),
185    /// Skip parse-time arg validation. The downstream layer (extended
186    /// dialect) enforces an instruction-specific rule based on tags.
187    Deferred,
188}

203/// Carries a chumsky 0.12 `Rich<char>` error plus a shared `LineMap`
204/// so that `Display` formats `line:col` consistently with the typed
205/// validation variants of [`ParseError`].
206#[derive(Debug, Clone)]
207pub struct SyntaxError {
208    pub(crate) rich: Rich<'static, char>,
209    pub(crate) line_map: Arc<LineMap>,
210}
211
212impl SyntaxError {
213    /// Synthesise a `SyntaxError` from a (line, col) position. Used by
214    /// the validator when it needs to emit a syntax error for an issue
215    /// the grammar could not catch (e.g. an annotation-tolerated target
216    /// that fails `usize` parsing for a non-annotation instruction).
217    pub(crate) fn synth(
218        line: usize,
219        col: usize,
220        message: impl Into<String>,
221        line_map: Arc<LineMap>,
222    ) -> Self {
223        let line_idx = line.saturating_sub(1);
224        let line_start = line_map.starts_at(line_idx).unwrap_or(0);
225        let byte = line_start + col.saturating_sub(1);
226        let span = chumsky::span::SimpleSpan::from(byte..byte);
227        let rich = Rich::<char>::custom(span, message.into());
228        SyntaxError { rich, line_map }
229    }
230
231    /// Construct from a chumsky `Rich` (with any borrow lifetime) and
232    /// a shared `LineMap`. `into_owned` widens the lifetime to `'static`.
233    pub(crate) fn from_rich<'src>(rich: Rich<'src, char>, line_map: Arc<LineMap>) -> Self {
234        SyntaxError {
235            rich: rich.into_owned(),
236            line_map,
237        }
238    }
239}
240
241impl fmt::Display for SyntaxError {
242    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
243        let span = self.rich.span();
244        let (line, col) = self.line_map.line_col(span.start);
245        write!(f, "syntax error at line {line}, col {col}: {}", self.rich)
246    }
247}

249#[derive(Debug, thiserror::Error)]
250pub enum ParseError {
251    #[error("{0}")]
252    Syntax(SyntaxError),
253
254    #[error("unknown instruction '{name}' at line {line}")]
255    UnknownInstruction { name: String, line: usize },
256
257    #[error("'{name}' at line {line} expected {expected} args, got {found}")]
258    ArgCount {
259        name: String,
260        expected: usize,
261        found: usize,
262        line: usize,
263    },
264
265    #[error("'{name}' at line {line} expected target count divisible by {divisor}, got {found}")]
266    TargetCount {
267        name: String,
268        divisor: usize,
269        found: usize,
270        line: usize,
271    },
272}
