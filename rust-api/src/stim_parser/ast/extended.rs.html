4//! Typed AST for Stim with PPVM tag-based extensions promoted to
5//! first-class instruction variants.
6
7use std::sync::Arc;
8
9use crate::ast::shared::{AnnotationOp, Axis, GateOp, MeasureOp, MppOp, NoiseOp, Tag};
10use crate::diagnostics::{LineMap, Span};
11
12#[derive(Debug, Clone, PartialEq)]
13pub enum ExtendedInstruction {
14    // Pass-through families — the SAME structs as the vanilla AST.
15    Gate(GateOp),
16    Noise(NoiseOp),
17    Measure(MeasureOp),
18    Annotation(AnnotationOp),
19    Mpp(MppOp),
20
21    // Promoted sugar.
22    T {
23        targets: Vec<usize>,
24        span: Span,
25    },
26    TDag {
27        targets: Vec<usize>,
28        span: Span,
29    },
30    Rotation {
31        axis: Axis,
32        theta: f64,
33        targets: Vec<usize>,
34        span: Span,
35    },
36    U3 {
37        theta: f64,
38        phi: f64,
39        lambda: f64,
40        targets: Vec<usize>,
41        span: Span,
42    },
43    Loss {
44        p: f64,
45        targets: Vec<usize>,
46        span: Span,
47    },
48    CorrelatedLoss {
49        ps: [f64; 3],
50        targets: Vec<(usize, usize)>,
51        span: Span,
52    },
53    MPad {
54        tags: Vec<Tag>,
55        prob: Option<f64>,
56        bits: Vec<bool>,
57        span: Span,
58    },
59    Repeat {
60        count: u64,
61        body: Vec<ExtendedInstruction>,
62        span: Span,
63    },
64}
65
66#[derive(Debug, Clone)]
67pub struct ExtendedProgram {
68    pub instructions: Vec<ExtendedInstruction>,
69    pub line_map: Arc<LineMap>,
70}
71
72impl PartialEq for ExtendedProgram {
73    fn eq(&self, other: &Self) -> bool {
74        self.instructions == other.instructions
75    }
76}
77
78impl ExtendedProgram {
79    /// Total recorded bits the program produces, accounting for REPEAT
80    /// factors. Pure AST property; backend-agnostic.
81    pub fn measurement_count(&self) -> usize {
82        count_in_slice(&self.instructions, 1)
83    }
84}
85
86fn count_in_slice(instructions: &[ExtendedInstruction], factor: u64) -> usize {
87    let mut total = 0usize;
88    let factor_usize = usize::try_from(factor).unwrap_or(usize::MAX);
89    for instr in instructions {
90        match instr {
91            ExtendedInstruction::Measure(op) => {
92                total = total.saturating_add(op.targets.len().saturating_mul(factor_usize));
93            }
94            ExtendedInstruction::MPad { bits, .. } => {
95                total = total.saturating_add(bits.len().saturating_mul(factor_usize));
96            }
97            ExtendedInstruction::Mpp(op) => {
98                total = total.saturating_add(op.products.len().saturating_mul(factor_usize));
99            }
100            ExtendedInstruction::Repeat { count, body, .. } => {
101                total = total.saturating_add(count_in_slice(body, factor.saturating_mul(*count)));
102            }
103            ExtendedInstruction::Gate(_)
104            | ExtendedInstruction::Noise(_)
105            | ExtendedInstruction::Annotation(_)
106            | ExtendedInstruction::T { .. }
107            | ExtendedInstruction::TDag { .. }
108            | ExtendedInstruction::Rotation { .. }
109            | ExtendedInstruction::U3 { .. }
110            | ExtendedInstruction::Loss { .. }
111            | ExtendedInstruction::CorrelatedLoss { .. } => {}
112        }
113    }
114    total
115}

117#[cfg(test)]
118mod tests {
119    use super::*;
120    use crate::ast::shared::{GateOp, MeasureOp};
121    use crate::diagnostics::{LineMap, Span};
122    use crate::instructions::{GateName, MeasureName};
123    use std::sync::Arc;
124
125    fn span() -> Span {
126        Span::new(0, 1)
127    }
128
129    #[test]
130    fn measurement_count_scales_with_repeat() {
131        let m = ExtendedInstruction::Measure(MeasureOp {
132            name: MeasureName::M,
133            tags: vec![],
134            args: vec![],
135            targets: vec![0, 1],
136            span: span(),
137        });
138        let prog = ExtendedProgram {
139            instructions: vec![ExtendedInstruction::Repeat {
140                count: 3,
141                body: vec![m],
142                span: span(),
143            }],
144            line_map: Arc::new(LineMap::new("")),
145        };
146        assert_eq!(prog.measurement_count(), 6);
147    }
148
149    #[test]
150    fn gate_op_is_shared_with_vanilla() {
151        let _ = ExtendedInstruction::Gate(GateOp {
152            name: GateName::H,
153            tags: vec![],
154            args: vec![],
155            targets: vec![],
156            span: span(),
157        });
158    }
