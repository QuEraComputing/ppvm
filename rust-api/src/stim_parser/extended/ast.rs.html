1//! Typed AST for Stim with PPVM tag-based extensions promoted to
2//! first-class instruction variants.
3
4use crate::ast::{RawInstruction, Tag};
5
6#[derive(Debug, Clone, PartialEq)]
7pub struct ExtendedProgram {
8    pub instructions: Vec<ExtendedInstruction>,
9}
10
11#[derive(Debug, Clone, PartialEq)]
12pub enum ExtendedInstruction {
13    /// Pass-through from vanilla Stim — covers `Gate`, `Noise`, `Measure`,
14    /// `Annotation`. `MPad` and `Repeat` are NOT in `Raw` because their
15    /// bits/body shapes diverge between dialects.
16    Raw(RawInstruction),
17
18    // --- Extended-dialect sugar variants ---
19    T {
20        targets: Vec<usize>,
21        line: usize,
22    },
23    TDag {
24        targets: Vec<usize>,
25        line: usize,
26    },
27    Rotation {
28        axis: Axis,
29        theta: f64,
30        targets: Vec<usize>,
31        line: usize,
32    },
33    U3 {
34        theta: f64,
35        phi: f64,
36        lambda: f64,
37        targets: Vec<usize>,
38        line: usize,
39    },
40    Loss {
41        p: f64,
42        targets: Vec<usize>,
43        line: usize,
44    },
45    CorrelatedLoss {
46        ps: [f64; 3],
47        targets: Vec<(usize, usize)>,
48        line: usize,
49    },
50    /// Extended-dialect MPad with validated bits (each ∈ {0, 1}).
51    MPad {
52        tags: Vec<Tag>,
53        prob: Option<f64>,
54        bits: Vec<bool>,
55        line: usize,
56    },
57    Repeat {
58        count: u64,
59        body: Vec<ExtendedInstruction>,
60        line: usize,
61    },
62}
63
64#[derive(Debug, Clone, Copy, PartialEq, Eq)]
65pub enum Axis {
66    X,
67    Y,
68    Z,
69}
70
71impl ExtendedProgram {
72    /// Total number of recorded bits the program will produce, accounting for
73    /// `REPEAT` factors. Pure AST property; backend-agnostic.
74    pub fn measurement_count(&self) -> usize {
75        count_in_slice(&self.instructions, 1)
76    }
77}
78
79fn count_in_slice(instructions: &[ExtendedInstruction], factor: u64) -> usize {
80    let mut total = 0usize;
81    let factor_usize = usize::try_from(factor).unwrap_or(usize::MAX);
82    for instr in instructions {
83        match instr {
84            ExtendedInstruction::Raw(RawInstruction::Measure { targets, .. }) => {
85                total = total.saturating_add(targets.len().saturating_mul(factor_usize));
86            }
87            ExtendedInstruction::MPad { bits, .. } => {
88                total = total.saturating_add(bits.len().saturating_mul(factor_usize));
89            }
90            ExtendedInstruction::Repeat { count, body, .. } => {
91                total = total.saturating_add(count_in_slice(body, factor.saturating_mul(*count)));
92            }
93            ExtendedInstruction::Raw(_)
94            | ExtendedInstruction::T { .. }
95            | ExtendedInstruction::TDag { .. }
96            | ExtendedInstruction::Rotation { .. }
97            | ExtendedInstruction::U3 { .. }
98            | ExtendedInstruction::Loss { .. }
99            | ExtendedInstruction::CorrelatedLoss { .. } => {}
100        }
101    }
102    total
103}
