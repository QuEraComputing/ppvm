4//! Typed AST for Stim with PPVM tag-based extensions promoted to
5//! first-class instruction variants.
6
7use crate::ast::{AnnotationKind, GateName, MeasureName, NoiseName, RawInstruction, Tag};
8
9#[derive(Debug, Clone, PartialEq)]
10pub struct ExtendedProgram {
11    pub instructions: Vec<ExtendedInstruction>,
12}
13
14/// Subset of [`RawInstruction`] that passes through the extended-dialect
15/// interpreter unchanged. Excludes `MPad` and `Repeat`, which are always
16/// lowered to the typed [`ExtendedInstruction::MPad`] /
17/// [`ExtendedInstruction::Repeat`] variants — so by construction
18/// `ExtendedInstruction::Raw(_)` can never wrap one of them.
19#[derive(Debug, Clone, PartialEq)]
20pub enum RawPassthrough {
21    Gate {
22        name: GateName,
23        tags: Vec<Tag>,
24        args: Vec<f64>,
25        targets: Vec<usize>,
26        line: usize,
27    },
28    Noise {
29        name: NoiseName,
30        tags: Vec<Tag>,
31        args: Vec<f64>,
32        targets: Vec<usize>,
33        line: usize,
34    },
35    Measure {
36        name: MeasureName,
37        tags: Vec<Tag>,
38        args: Vec<f64>,
39        targets: Vec<usize>,
40        line: usize,
41    },
42    Annotation {
43        kind: AnnotationKind,
44        args: Vec<f64>,
45        targets: Vec<usize>,
46        line: usize,
47    },
48}
49
50impl RawPassthrough {
51    /// Lift back into [`RawInstruction`]. Used by the printer to share
52    /// formatting code with the vanilla AST.
53    pub fn into_raw(self) -> RawInstruction {
54        match self {
55            RawPassthrough::Gate {
56                name,
57                tags,
58                args,
59                targets,
60                line,
61            } => RawInstruction::Gate {
62                name,
63                tags,
64                args,
65                targets,
66                line,
67            },
68            RawPassthrough::Noise {
69                name,
70                tags,
71                args,
72                targets,
73                line,
74            } => RawInstruction::Noise {
75                name,
76                tags,
77                args,
78                targets,
79                line,
80            },
81            RawPassthrough::Measure {
82                name,
83                tags,
84                args,
85                targets,
86                line,
87            } => RawInstruction::Measure {
88                name,
89                tags,
90                args,
91                targets,
92                line,
93            },
94            RawPassthrough::Annotation {
95                kind,
96                args,
97                targets,
98                line,
99            } => RawInstruction::Annotation {
100                kind,
101                args,
102                targets,
103                line,
104            },
105        }
106    }
107
108    pub fn line(&self) -> usize {
109        match self {
110            RawPassthrough::Gate { line, .. }
111            | RawPassthrough::Noise { line, .. }
112            | RawPassthrough::Measure { line, .. }
113            | RawPassthrough::Annotation { line, .. } => *line,
114        }
115    }
116}

118#[derive(Debug, Clone, PartialEq)]
119pub enum ExtendedInstruction {
120    /// Pass-through from vanilla Stim — covers `Gate`, `Noise`, `Measure`,
121    /// `Annotation`. `MPad` and `Repeat` get their own typed variants on
122    /// [`ExtendedInstruction`] because their bits/body shapes diverge
123    /// between dialects.
124    Raw(RawPassthrough),
125
126    // --- Extended-dialect sugar variants ---
127    T {
128        targets: Vec<usize>,
129        line: usize,
130    },
131    TDag {
132        targets: Vec<usize>,
133        line: usize,
134    },
135    Rotation {
136        axis: Axis,
137        theta: f64,
138        targets: Vec<usize>,
139        line: usize,
140    },
141    U3 {
142        theta: f64,
143        phi: f64,
144        lambda: f64,
145        targets: Vec<usize>,
146        line: usize,
147    },
148    Loss {
149        p: f64,
150        targets: Vec<usize>,
151        line: usize,
152    },
153    CorrelatedLoss {
154        ps: [f64; 3],
155        targets: Vec<(usize, usize)>,
156        line: usize,
157    },
158    /// Extended-dialect MPad with validated bits (each ∈ {0, 1}).
159    MPad {
160        tags: Vec<Tag>,
161        prob: Option<f64>,
162        bits: Vec<bool>,
163        line: usize,
164    },
165    Repeat {
166        count: u64,
167        body: Vec<ExtendedInstruction>,
168        line: usize,
169    },
170}
171
172#[derive(Debug, Clone, Copy, PartialEq, Eq)]
173pub enum Axis {
174    X,
175    Y,
176    Z,
177}
178
179impl ExtendedProgram {
180    /// Total number of recorded bits the program will produce, accounting for
181    /// `REPEAT` factors. Pure AST property; backend-agnostic.
182    pub fn measurement_count(&self) -> usize {
183        count_in_slice(&self.instructions, 1)
184    }
185}
186
187fn count_in_slice(instructions: &[ExtendedInstruction], factor: u64) -> usize {
188    let mut total = 0usize;
189    let factor_usize = usize::try_from(factor).unwrap_or(usize::MAX);
190    for instr in instructions {
191        match instr {
192            ExtendedInstruction::Raw(RawPassthrough::Measure { targets, .. }) => {
193                total = total.saturating_add(targets.len().saturating_mul(factor_usize));
194            }
195            ExtendedInstruction::MPad { bits, .. } => {
196                total = total.saturating_add(bits.len().saturating_mul(factor_usize));
197            }
198            ExtendedInstruction::Repeat { count, body, .. } => {
199                total = total.saturating_add(count_in_slice(body, factor.saturating_mul(*count)));
200            }
201            ExtendedInstruction::Raw(_)
202            | ExtendedInstruction::T { .. }
203            | ExtendedInstruction::TDag { .. }
204            | ExtendedInstruction::Rotation { .. }
205            | ExtendedInstruction::U3 { .. }
206            | ExtendedInstruction::Loss { .. }
207            | ExtendedInstruction::CorrelatedLoss { .. } => {}
208        }
209    }
210    total
211}
