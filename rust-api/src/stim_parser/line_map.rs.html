4/// Maps byte offsets in source to 1-indexed line/column positions.
5pub struct LineMap {
6    /// `starts[i]` = byte offset of the start of line (i+1).
7    starts: Vec<usize>,
8}
9
10impl std::fmt::Debug for LineMap {
11    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
12        f.debug_struct("LineMap")
13            .field("lines", &self.starts.len())
14            .finish()
15    }
16}
17
18impl LineMap {
19    /// Build a `LineMap` for `src`.
20    pub fn new(src: &str) -> Self {
21        let mut starts = vec![0];
22        for (i, b) in src.bytes().enumerate() {
23            if b == b'\n' {
24                starts.push(i + 1);
25            }
26        }
27        Self { starts }
28    }
29
30    /// 1-indexed line number for a byte offset.
31    pub fn line_of(&self, byte_offset: usize) -> usize {
32        match self.starts.binary_search(&byte_offset) {
33            Ok(i) => i + 1,
34            Err(i) => i, // i is the insertion index; start of line `i` is at starts[i-1].
35        }
36    }
37
38    /// 1-indexed `(line, col)` for a byte offset.
39    pub fn line_col(&self, byte_offset: usize) -> (usize, usize) {
40        let line = self.line_of(byte_offset);
41        let line_start = self.starts[line - 1];
42        let col = byte_offset - line_start + 1;
43        (line, col)
44    }
45
46    /// Byte offset of the start of line `(line_idx + 1)`. `None` for
47    /// out-of-range indices.
48    pub fn starts_at(&self, line_idx: usize) -> Option<usize> {
49        self.starts.get(line_idx).copied()
50    }
51}

53#[cfg(test)]
54mod line_map_tests {
55    use super::LineMap;
56
57    #[test]
58    fn line_col_at_line_start() {
59        let m = LineMap::new("abc\ndef\nghi");
60        assert_eq!(m.line_col(0), (1, 1));
61        assert_eq!(m.line_col(4), (2, 1));
62        assert_eq!(m.line_col(8), (3, 1));
63    }
64
65    #[test]
66    fn line_col_mid_line() {
67        let m = LineMap::new("abc\ndef\nghi");
68        assert_eq!(m.line_col(2), (1, 3));
69        assert_eq!(m.line_col(6), (2, 3));
70    }
71
72    #[test]
73    fn line_col_at_eof() {
74        let m = LineMap::new("abc\ndef");
75        assert_eq!(m.line_col(7), (2, 4));
76    }
77}
