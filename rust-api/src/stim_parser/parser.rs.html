25/// Raw syntactic tree produced by the chumsky grammar before
26/// table-driven validation. `pub(crate)` because it is plumbing
27/// between `grammar.rs` and the validator post-pass; not part of the
28/// public API.
29#[derive(Debug, Clone)]
30pub(crate) enum RawSyntaxNode {
31    Instruction {
32        name: String,
33        tags: Vec<Tag>,
34        args: Vec<f64>,
35        targets: Vec<RawTarget>,
36        span: SimpleSpan<usize>,
37    },
38    Repeat {
39        count: u64,
40        body: Vec<RawSyntaxNode>,
41        span: SimpleSpan<usize>,
42    },
43}

45#[derive(Debug, Clone)]
46pub(crate) struct RawTarget {

48    pub span: SimpleSpan<usize>,
49}

51/// Stack size for the dedicated parsing thread. The chumsky grammar
52/// is built around `recursive(...)`, which descends into REPEAT bodies
53/// via recursive parser calls; on the default thread stack (typically
54/// 2–8 MiB), deeply nested programs overflow somewhere past ~24 levels
55/// in debug builds. Running on an oversized dedicated stack lets us
56/// support thousands of nested REPEATs without rewriting the grammar.
57#[cfg(not(target_arch = "wasm32"))]
58const PARSER_STACK_SIZE: usize = 16 * 1024 * 1024;
59
60/// Run `f` with a large stack for the recursive chumsky grammar. On targets
61/// with OS threads this spawns a dedicated [`PARSER_STACK_SIZE`]-byte thread;
62/// on `wasm32` (no `std::thread`) `f` runs inline on the caller's stack, so
63/// deeply nested REPEAT bodies rely on the wasm stack instead (configurable
64/// via the linker `stack-size` arg). Used by both [`parse`] and
65/// [`parse_extended`] (which also recurses into REPEAT bodies during the
66/// interpret pass).
67pub(crate) fn run_on_parser_stack<R, F>(f: F) -> R
68where
69    R: Send,
70    F: FnOnce() -> R + Send,
71{
72    #[cfg(target_arch = "wasm32")]
73    {
74        f()
75    }
76    #[cfg(not(target_arch = "wasm32"))]
77    {
78        std::thread::scope(|s| {
79            let handle = std::thread::Builder::new()
80                .stack_size(PARSER_STACK_SIZE)
81                .name("stim-parser".to_string())
82                .spawn_scoped(s, f)
83                .expect("failed to spawn parser thread");
84            match handle.join() {
85                Ok(value) => value,
86                Err(payload) => std::panic::resume_unwind(payload),
87            }
88        })
89    }
90}

92/// Parse Stim source into a [`Program`].
93pub fn parse(src: &str) -> Result<Program, ParseError> {
94    run_on_parser_stack(|| parse_impl(src))
95}

97pub(crate) fn parse_impl(src: &str) -> Result<Program, ParseError> {
98    use chumsky::Parser;
99    let line_map = Arc::new(LineMap::new(src));
100    let parse_result = grammar::program_parser().parse(src);
101    let nodes = match parse_result.into_result() {
102        Ok(nodes) => nodes,
103        Err(errors) => {
104            let first = errors.into_iter().next().expect("non-empty error list");
105            return Err(ParseError::Syntax(SyntaxError::from_rich(
106                first,
107                Arc::clone(&line_map),
108            )));
109        }
110    };
111    let instructions = validate_program(nodes, &line_map)?;
112    Ok(Program { instructions })
113}

115/// Walk the raw syntactic tree and emit validated instructions.
116fn validate_program(
117    nodes: Vec<RawSyntaxNode>,
118    line_map: &Arc<LineMap>,
119) -> Result<Vec<RawInstruction>, ParseError> {
120    let mut out = Vec::with_capacity(nodes.len());
121    for node in nodes {
122        out.push(validate_node(node, line_map)?);
123    }
124    Ok(out)
125}
126
127fn validate_node(
128    node: RawSyntaxNode,
129    line_map: &Arc<LineMap>,
130) -> Result<RawInstruction, ParseError> {
131    match node {
132        RawSyntaxNode::Instruction {
133            name,
134            tags,
135            args,
136            targets,
137            span,
138        } => {
139            let line = line_map.line_of(span.start);
140            let entry = lookup(&name).ok_or(ParseError::UnknownInstruction {
141                name: name.clone(),
142                line,
143            })?;
144            let arg_rule = entry.args;
145            let target_rule = entry.targets;
146            let canonical = entry.canonical();
147
148            // MPP carries Pauli-product targets (`X0*Y1*Z2`), not qubit indices,
149            // so it parses on a dedicated path rather than the qubit/record loop.
150            if matches!(entry.kind, EntryKind::Measure(MeasureName::MPP)) {
151                let products = parse_mpp_products(&targets, line_map)?;
152                check_arg_count(arg_rule, args.len(), canonical, line)?;
153                if products.is_empty() {
154                    return Err(ParseError::TargetCount {
155                        name: canonical.to_string(),
156                        divisor: 1,
157                        found: 0,
158                        line,
159                    });
160                }
161                return Ok(RawInstruction::Mpp {
162                    tags,
163                    args,
164                    products,
165                    line,
166                });
167            }
168
169            let is_annotation = matches!(entry.kind, EntryKind::Annotation(_));
170            // Only gates accept measurement-record controls (`rec[-k]`), as in
171            // `CX rec[-1] 1`. For every other instruction a `rec[...]` target is
172            // either tolerated-and-dropped (annotations) or an error.
173            let is_gate = matches!(entry.kind, EntryKind::Gate(_));
174            let mut parsed_targets: Vec<Target> = Vec::with_capacity(targets.len());
175            for t in &targets {
176                if let Ok(n) = t.text.parse::<usize>() {
177                    parsed_targets.push(Target::Qubit(n));
178                    continue;
179                }
180                if is_gate && let Some(k) = parse_rec(&t.text) {
181                    parsed_targets.push(Target::Rec(k));
182                    continue;
183                }
184                if is_annotation {
185                    // Annotations (DETECTOR, OBSERVABLE_INCLUDE, …) tolerate
186                    // non-numeric targets like `rec[-1]` by dropping them.
187                    continue;
188                }
189                let (l, c) = line_map.line_col(t.span.start);
190                return Err(syntax(
191                    l,
192                    c,
193                    format!("invalid target {:?}", t.text),
194                    line_map,
195                ));
196            }
197
198            check_arg_count(arg_rule, args.len(), canonical, line)?;
199
200            let divisor = match target_rule {
201                TargetArity::Any => None,
202                TargetArity::AtLeastOne => Some(1),
203                TargetArity::Pairs => Some(2),
204                TargetArity::Quadruples => Some(4),
205            };
206            if let Some(d) = divisor {
207                let n = parsed_targets.len();
208                if n == 0 || !n.is_multiple_of(d) {
209                    return Err(ParseError::TargetCount {
210                        name: canonical.to_string(),
211                        divisor: d,
212                        found: n,
213                        line,
214                    });
215                }
216            }
217
218            Ok(build_instruction(entry, tags, args, parsed_targets, line))
219        }
220        RawSyntaxNode::Repeat { count, body, span } => {
221            let line = line_map.line_of(span.start);
222            let body = validate_program(body, line_map)?;
223            Ok(RawInstruction::Repeat { count, body, line })
224        }
225    }
226}

228/// Validate an instruction's argument count against its table rule.
229fn check_arg_count(
230    arg_rule: ArgCount,
231    found: usize,
232    canonical: &str,
233    line: usize,
234) -> Result<(), ParseError> {
235    let expected = match arg_rule {
236        ArgCount::Deferred | ArgCount::Any => return Ok(()),
237        ArgCount::None if found != 0 => 0,
238        ArgCount::Exact(n) if found != n => n,
239        ArgCount::Optional(n) if found != 0 && found != n => n,
240        _ => return Ok(()),
241    };
242    Err(ParseError::ArgCount {
243        name: canonical.to_string(),
244        expected,
245        found,
246        line,
247    })
248}

250/// Parse the space-separated Pauli-product targets of an `MPP` instruction,
251/// e.g. `X0*Y1*Z2 Z3*Z4` into two products. Each product is a `*`-joined run
252/// of single-qubit Pauli factors (`<X|Y|Z><qubit>`).
253fn parse_mpp_products(
254    targets: &[RawTarget],
255    line_map: &Arc<LineMap>,
256) -> Result<Vec<Vec<PauliFactor>>, ParseError> {
257    let mut products = Vec::with_capacity(targets.len());
258    for t in targets {
259        let mut factors = Vec::new();
260        for factor in t.text.split('*') {
261            let mut chars = factor.chars();
262            let axis = match chars.next() {
263                Some('X') => PauliAxis::X,
264                Some('Y') => PauliAxis::Y,
265                Some('Z') => PauliAxis::Z,
266                _ => return Err(invalid_mpp_target(t, line_map)),
267            };
268            let Ok(qubit) = chars.as_str().parse::<usize>() else {
269                return Err(invalid_mpp_target(t, line_map));
270            };
271            factors.push(PauliFactor { axis, qubit });
272        }
273        products.push(factors);
274    }
275    Ok(products)
276}

278fn invalid_mpp_target(t: &RawTarget, line_map: &Arc<LineMap>) -> ParseError {
279    let (l, c) = line_map.line_col(t.span.start);
280    syntax(l, c, format!("invalid MPP target {:?}", t.text), line_map)
281}

283/// Parse a measurement-record lookback target `rec[-k]` into its lookback
284/// distance `k >= 1`. Returns `None` for anything that isn't a well-formed,
285/// strictly-negative record reference (so `rec[0]`, `rec[1]`, `rec[]` and
286/// non-`rec` text all fall through to the caller's error handling).
287fn parse_rec(text: &str) -> Option<usize> {
288    let inner = text.strip_prefix("rec[")?.strip_suffix(']')?;
289    let magnitude = inner.strip_prefix('-')?;
290    match magnitude.parse::<usize>() {
291        Ok(k) if k >= 1 => Some(k),
292        _ => None,
293    }
294}

296/// Convert validated targets to bare qubit indices. Only gate targets may be
297/// `rec[...]`; every other instruction's targets were validated as qubits in
298/// [`validate_node`], so this never drops information for them.
299fn qubit_indices(targets: Vec<Target>) -> Vec<usize> {
300    targets
301        .into_iter()
302        .map(|t| {
303            t.as_qubit()
304                .expect("non-gate targets are validated as plain qubits")
305        })
306        .collect()
307}

309fn build_instruction(
310    entry: TableEntry,
311    tags: Vec<Tag>,
312    args: Vec<f64>,
313    targets: Vec<Target>,
314    line: usize,
315) -> RawInstruction {
316    match entry.kind {
317        EntryKind::Gate(name) => RawInstruction::Gate {
318            name,
319            tags,
320            args,
321            targets,
322            line,
323        },
324        EntryKind::Noise(name) => RawInstruction::Noise {
325            name,
326            tags,
327            args,
328            targets: qubit_indices(targets),
329            line,
330        },
331        EntryKind::Measure(name) => RawInstruction::Measure {
332            name,
333            tags,
334            args,
335            targets: qubit_indices(targets),
336            line,
337        },
338        EntryKind::Annotation(kind) => RawInstruction::Annotation {
339            kind,
340            args,
341            targets: qubit_indices(targets),
342            line,
343        },
344        EntryKind::MPad => RawInstruction::MPad {
345            tags,
346            prob: args.into_iter().next(),
347            bits: qubit_indices(targets),
348            line,
349        },
350    }
351}

367        targets: Vec<&str>,
