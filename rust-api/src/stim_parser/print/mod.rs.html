4//! First-class canonical printer for both AST layers. Output is canonical
5//! Stim — 4-space REPEAT indentation, `[tags](args) targets` ordering — and
6//! round-trips: parse → print → parse is a fixpoint.
7
8use std::borrow::Cow;
9use std::fmt;
10
11use crate::ast::shared::{
12    AnnotationOp, Axis, GateOp, MeasureOp, MppOp, NoiseOp, PauliFactor, Tag, TagParam, Target,
13};
14use crate::ast::{ExtendedInstruction, ExtendedProgram, Instruction, Program};
15
16pub struct PrintOptions {
17    pub indent: Cow<'static, str>,
18}
19
20impl Default for PrintOptions {
21    fn default() -> Self {
22        PrintOptions {
23            indent: Cow::Borrowed("    "),
24        }
25    }
26}
27
28pub trait StimPrint {
29    fn print(&self, out: &mut dyn fmt::Write, opts: &PrintOptions, depth: usize) -> fmt::Result;
30
31    fn to_stim(&self) -> String {
32        self.to_stim_with(&PrintOptions::default())
33    }
34
35    fn to_stim_with(&self, opts: &PrintOptions) -> String {
36        let mut s = String::new();
37        let _ = self.print(&mut s, opts, 0);
38        s
39    }
40}
41
42// ---------------------------------------------------------------------------
43// Low-level writers shared by every `StimPrint` impl. They produce canonical
44// Stim byte-for-byte and write to any `&mut dyn fmt::Write`.
45// ---------------------------------------------------------------------------
46
47fn write_indent(out: &mut dyn fmt::Write, opts: &PrintOptions, depth: usize) -> fmt::Result {
48    for _ in 0..depth {
49        out.write_str(&opts.indent)?;
50    }
51    Ok(())
52}
53
54fn write_tags(out: &mut dyn fmt::Write, tags: &[Tag]) -> fmt::Result {
55    if tags.is_empty() {
56        return Ok(());
57    }
58    out.write_str("[")?;
59    for (i, tag) in tags.iter().enumerate() {
60        if i > 0 {
61            out.write_str(", ")?;
62        }
63        out.write_str(&tag.name)?;
64        if !tag.params.is_empty() {
65            out.write_str("(")?;
66            for (j, p) in tag.params.iter().enumerate() {
67                if j > 0 {
68                    out.write_str(", ")?;
69                }
70                match p {
71                    TagParam::Positional(v) => write!(out, "{}", FloatLit(*v))?,
72                    TagParam::Named { key, value } => {
73                        write!(out, "{key}={}", FloatLit(*value))?;
74                    }
75                }
76            }
77            out.write_str(")")?;
78        }
79    }
80    out.write_str("]")
81}
82
83fn write_args(out: &mut dyn fmt::Write, args: &[f64]) -> fmt::Result {
84    if args.is_empty() {
85        return Ok(());
86    }
87    out.write_str("(")?;
88    for (i, a) in args.iter().enumerate() {
89        if i > 0 {
90            out.write_str(", ")?;
91        }
92        write!(out, "{}", FloatLit(*a))?;
93    }
94    out.write_str(")")
95}
96
97fn write_usize_targets(out: &mut dyn fmt::Write, targets: &[usize]) -> fmt::Result {
98    for t in targets {
99        write!(out, " {t}")?;
100    }
101    Ok(())
102}
103
104/// Print gate targets, rendering measurement-record controls as `rec[-k]`
105/// so the output round-trips back through the parser.
106fn write_targets(out: &mut dyn fmt::Write, targets: &[Target]) -> fmt::Result {
107    for t in targets {
108        match t {
109            Target::Qubit(q) => write!(out, " {q}")?,
110            Target::Rec(k) => write!(out, " rec[-{k}]")?,
111        }
112    }
113    Ok(())
114}
115
116/// Print a `REPEAT count { … }` block, recursively printing the body one
117/// indent level deeper and closing the brace at the block's own depth. The
118/// caller is responsible for the trailing newline after the closing brace.
119fn write_repeat_block<T: StimPrint>(
120    out: &mut dyn fmt::Write,
121    opts: &PrintOptions,
122    depth: usize,
123    count: u64,
124    body: &[T],
125) -> fmt::Result {
126    writeln!(out, "REPEAT {count} {{")?;
127    for instr in body {
128        instr.print(out, opts, depth + 1)?;
129    }
130    write_indent(out, opts, depth)?;
131    out.write_str("}")
132}
133
134/// Print `MPP` products as space-separated, `*`-joined Pauli factors
135/// (`X0*Y1*Z2`) so the output round-trips back through the parser.
136fn write_mpp_products(out: &mut dyn fmt::Write, products: &[Vec<PauliFactor>]) -> fmt::Result {
137    for product in products {
138        out.write_str(" ")?;
139        for (i, factor) in product.iter().enumerate() {
140            if i > 0 {
141                out.write_str("*")?;
142            }
143            write!(out, "{}{}", factor.axis.as_char(), factor.qubit)?;
144        }
145    }
146    Ok(())
147}

149/// f64 formatter that always emits a decimal point. The grammar's
150/// `signed_float` accepts bare integers too (`42` parses), so this is
151/// purely a readability choice — printing `1.0` instead of `1` keeps the
152/// canonical output looking like floating-point everywhere args are
153/// expected.
154struct FloatLit(f64);
155
156impl fmt::Display for FloatLit {
157    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
158        let s = format!("{}", self.0);
159        if s.contains('.') || s.contains('e') || s.contains('E') || s.contains("inf") || s == "NaN"
160        {
161            f.write_str(&s)
162        } else {
163            write!(f, "{s}.0")
164        }
165    }
166}

220impl StimPrint for Instruction {
221    fn print(&self, out: &mut dyn fmt::Write, opts: &PrintOptions, depth: usize) -> fmt::Result {
222        write_indent(out, opts, depth)?;
223        match self {
224            Instruction::Gate(op) => op.print(out, opts, depth)?,
225            Instruction::Noise(op) => op.print(out, opts, depth)?,
226            Instruction::Measure(op) => op.print(out, opts, depth)?,
227            Instruction::Annotation(op) => op.print(out, opts, depth)?,
228            Instruction::Mpp(op) => op.print(out, opts, depth)?,
229            Instruction::MPad {
230                tags, prob, bits, ..
231            } => {
232                out.write_str("MPAD")?;
233                write_tags(out, tags)?;
234                if let Some(p) = prob {
235                    write!(out, "({})", FloatLit(*p))?;
236                }
237                write_usize_targets(out, bits)?;
238            }
239            Instruction::Repeat { count, body, .. } => {
240                write_repeat_block(out, opts, depth, *count, body)?;
241            }
242        }
243        writeln!(out)
244    }
245}

266impl StimPrint for ExtendedInstruction {
267    fn print(&self, out: &mut dyn fmt::Write, opts: &PrintOptions, depth: usize) -> fmt::Result {
268        write_indent(out, opts, depth)?;
269        match self {
270            ExtendedInstruction::Gate(op) => op.print(out, opts, depth)?,
271            ExtendedInstruction::Noise(op) => op.print(out, opts, depth)?,
272            ExtendedInstruction::Measure(op) => op.print(out, opts, depth)?,
273            ExtendedInstruction::Annotation(op) => op.print(out, opts, depth)?,
274            ExtendedInstruction::Mpp(op) => op.print(out, opts, depth)?,
275            ExtendedInstruction::T { targets, .. } => {
276                out.write_str("S[T]")?;
277                write_usize_targets(out, targets)?;
278            }
279            ExtendedInstruction::TDag { targets, .. } => {
280                out.write_str("S_DAG[T]")?;
281                write_usize_targets(out, targets)?;
282            }
283            ExtendedInstruction::Rotation {
284                axis,
285                theta,
286                targets,
287                ..
288            } => {
289                let axis_tag = match axis {
290                    Axis::X => "R_X",
291                    Axis::Y => "R_Y",
292                    Axis::Z => "R_Z",
293                };
294                write!(out, "I[{}(theta={})]", axis_tag, FloatLit(*theta))?;
295                write_usize_targets(out, targets)?;
296            }
297            ExtendedInstruction::U3 {
298                theta,
299                phi,
300                lambda,
301                targets,
302                ..
303            } => {
304                write!(
305                    out,
306                    "I[U3(theta={}, phi={}, lambda={})]",
307                    FloatLit(*theta),
308                    FloatLit(*phi),
309                    FloatLit(*lambda),
310                )?;
311                write_usize_targets(out, targets)?;
312            }
313            ExtendedInstruction::Loss { p, targets, .. } => {
314                write!(out, "I_ERROR[loss]({})", FloatLit(*p))?;
315                write_usize_targets(out, targets)?;
316            }
317            ExtendedInstruction::CorrelatedLoss { ps, targets, .. } => {
318                write!(
319                    out,
320                    "I_ERROR[correlated_loss]({}, {}, {})",
321                    FloatLit(ps[0]),
322                    FloatLit(ps[1]),
323                    FloatLit(ps[2]),
324                )?;
325                for &(a, b) in targets {
326                    write!(out, " {a} {b}")?;
327                }
328            }
329            ExtendedInstruction::MPad {
330                tags, prob, bits, ..
331            } => {
332                out.write_str("MPAD")?;
333                write_tags(out, tags)?;
334                if let Some(p) = prob {
335                    write!(out, "({})", FloatLit(*p))?;
336                }
337                for &bit in bits {
338                    write!(out, " {}", u8::from(bit))?;
339                }
340            }
341            ExtendedInstruction::Repeat { count, body, .. } => {
342                write_repeat_block(out, opts, depth, *count, body)?;
343            }
344        }
345        writeln!(out)
346    }
347}

368#[cfg(test)]
369mod tests {
370    use crate::print::StimPrint;
371    use crate::{parse, parse_extended};
372
373    #[test]
374    fn vanilla_printed_form_is_canonical_shape() {
375        let src = "H 0  # trail\nCX  0   1\nDEPOLARIZE1(0.05) 0 1\nREPEAT 2 { X 0 }\n";
376        let ast = parse(src).unwrap();
377        let expected = "H 0\nCX 0 1\nDEPOLARIZE1(0.05) 0 1\nREPEAT 2 {\n    X 0\n}\n";
378        assert_eq!(ast.to_stim(), expected);
379        assert_eq!(format!("{ast}"), expected);
380    }
381
382    #[test]
383    fn extended_printed_form_lowers_sugar_into_canonical_stim() {
384        let src = "S[T] 0\nI[R_X(theta=0.25)] 1\nI_ERROR[loss](0.01) 2\n";
385        let ast = parse_extended(src).unwrap();
386        let expected = "S[T] 0\nI[R_X(theta=0.25)] 1\nI_ERROR[loss](0.01) 2\n";
387        assert_eq!(ast.to_stim(), expected);
388    }
389
390    #[test]
391    fn rec_and_mpp_targets_round_trip() {
392        // rec[-k] feed-forward control and MPP Pauli products print canonically.
393        let ast = parse("CX rec[-1] 0\nMPP X0*Y1*Z2\n").unwrap();
394        assert_eq!(ast.to_stim(), "CX rec[-1] 0\nMPP X0*Y1*Z2\n");
395    }
