4//! Chumsky 0.12 grammar for Stim source.
5//!
6//! Reads top-to-bottom: whitespace/comments -> numbers -> pi-expressions ->
7//! identifiers -> tags -> args -> targets -> instruction line -> REPEAT block ->
8//! program. Pure syntax; no table lookups.
9
10use chumsky::error::Rich;
11use chumsky::extra;
12use chumsky::prelude::*;
13
14type Extra<'src> = extra::Err<Rich<'src, char>>;
15
16/// `# ...` comment, stopping before `\n` if present.
17fn line_comment<'src>() -> impl Parser<'src, &'src str, (), Extra<'src>> + Clone {
18    just('#')
19        .ignore_then(any().filter(|c: &char| *c != '\n').repeated())
20        .ignored()
21}
22
23/// Pad: zero or more whitespace characters or `# ...` comments.
24/// Includes newlines. Used between instructions.
25pub(crate) fn pad<'src>() -> impl Parser<'src, &'src str, (), Extra<'src>> + Clone {
26    let ws = any().filter(|c: &char| c.is_whitespace()).ignored();
27    choice((line_comment(), ws)).repeated().ignored()
28}
29
30/// Inline pad: spaces/tabs/CR only. Excludes comments and `\n`.
31pub(crate) fn inline_pad<'src>() -> impl Parser<'src, &'src str, (), Extra<'src>> + Clone {
32    let ws = any()
33        .filter(|c: &char| matches!(*c, ' ' | '\t' | '\r'))
34        .ignored();
35    ws.repeated().ignored()
36}
37
38/// At least one inline whitespace character. Used before each target.
39pub(crate) fn inline_ws1<'src>() -> impl Parser<'src, &'src str, (), Extra<'src>> + Clone {
40    any()
41        .filter(|c: &char| matches!(*c, ' ' | '\t' | '\r'))
42        .repeated()
43        .at_least(1)
44        .ignored()
45}
46
47/// Optional trailing spaces/tabs plus an optional line comment.
48pub(crate) fn trailing_pad<'src>() -> impl Parser<'src, &'src str, (), Extra<'src>> + Clone {
49    inline_pad().then(line_comment().or_not()).ignored()
50}

52/// Identifier: `[A-Za-z_][A-Za-z0-9_]*`. Returns owned `String`.
53pub(crate) fn ident<'src>() -> impl Parser<'src, &'src str, String, Extra<'src>> + Clone {
54    any()
55        .filter(|c: &char| c.is_ascii_alphabetic() || *c == '_')
56        .then(
57            any()
58                .filter(|c: &char| c.is_ascii_alphanumeric() || *c == '_')
59                .repeated(),
60        )
61        .to_slice()
62        .map(|s: &str| s.to_string())
63}

65/// Signed float: common decimal/scientific forms accepted by `f64`
66/// parsing, excluding non-finite names such as `NaN`/`inf`.
67pub(crate) fn signed_float<'src>() -> impl Parser<'src, &'src str, f64, Extra<'src>> + Clone {
68    let digits = any()
69        .filter(|c: &char| c.is_ascii_digit())
70        .repeated()
71        .at_least(1);
72    let decimal = choice((
73        digits.then(just('.')).then(digits.or_not()).ignored(),
74        just('.').then(digits).ignored(),
75    ));
76    let exp_part = one_of("eE").then(one_of("+-").or_not()).then(digits);
77    one_of("+-")
78        .or_not()
79        .then(choice((decimal, digits.ignored())))
80        .then(exp_part.or_not())
81        .to_slice()
82        .map(|s: &str| s.parse::<f64>().expect("validated by combinator shape"))
83}

85/// Pi-expression, paired with whether `pi` actually appeared in the source:
86/// `pi` -> `(PI, true)`, `<num>[*]pi` -> `(num*PI, true)`, `<num>` -> `(num, false)`.
87/// The optional `*` may be omitted (`0.5pi`) or spaced (`0.5 * pi`).
88/// The flag lets rotation/U3 tags enforce the half-turn `<n>*pi` convention.
89pub(crate) fn pi_expr_flagged<'src>()
90-> impl Parser<'src, &'src str, (f64, bool), Extra<'src>> + Clone {
91    let pi_kw = just("pi").to((std::f64::consts::PI, true));
92    let pi_suffix = inline_pad()
93        .ignore_then(just('*').then(inline_pad()).or_not())
94        .then(just("pi"));
95    let num_then_pi = signed_float().then(pi_suffix.or_not()).map(|(n, suffix)| {
96        if suffix.is_some() {
97            (n * std::f64::consts::PI, true)
98        } else {
99            (n, false)
100        }
101    });
102    choice((pi_kw, num_then_pi))
103}
104
105/// Pi-expression: `pi`, `<num>*pi`, or plain number. Evaluates to f64.
106pub(crate) fn pi_expr<'src>() -> impl Parser<'src, &'src str, f64, Extra<'src>> + Clone {
107    pi_expr_flagged().map(|(value, _)| value)
108}

112/// `<ident>=<pi_expr>` (Named) or `<pi_expr>` (Positional).
113pub(crate) fn tag_param<'src>() -> impl Parser<'src, &'src str, TagParam, Extra<'src>> + Clone {
114    let named = ident()
115        .then_ignore(inline_pad())
116        .then_ignore(just('='))
117        .then_ignore(inline_pad())
118        .then(pi_expr_flagged())
119        .map(|(key, (value, had_pi))| TagParam::Named { key, value, had_pi });
120    let positional = pi_expr().map(TagParam::Positional);
121    choice((named, positional))
122}
123
124/// Tag: `<ident>` or `<ident>(<tag_param>, ...)`.
125pub(crate) fn tag<'src>() -> impl Parser<'src, &'src str, Tag, Extra<'src>> + Clone {
126    let params = tag_param()
127        .separated_by(inline_pad().then(just(',')).then(inline_pad()))
128        .allow_trailing()
129        .collect::<Vec<_>>()
130        .delimited_by(just('(').then(inline_pad()), inline_pad().then(just(')')));
131    ident().then(params.or_not()).map(|(name, params)| Tag {
132        name,
133        params: params.unwrap_or_default(),
134    })
135}
136
137/// `[tag, tag, ...]`.
138pub(crate) fn tags_block<'src>() -> impl Parser<'src, &'src str, Vec<Tag>, Extra<'src>> + Clone {
139    tag()
140        .separated_by(inline_pad().then(just(',')).then(inline_pad()))
141        .allow_trailing()
142        .collect::<Vec<_>>()
143        .delimited_by(just('[').then(inline_pad()), inline_pad().then(just(']')))
144}
145
146/// `(pi_expr, pi_expr, ...)`.
147pub(crate) fn args_block<'src>() -> impl Parser<'src, &'src str, Vec<f64>, Extra<'src>> + Clone {
148    pi_expr()
149        .separated_by(inline_pad().then(just(',')).then(inline_pad()))
150        .allow_trailing()
151        .collect::<Vec<_>>()
152        .delimited_by(just('(').then(inline_pad()), inline_pad().then(just(')')))
153}

155use crate::syntax::raw::{RawSyntaxNode, RawTarget};
156use chumsky::span::SimpleSpan;

158/// One non-whitespace, non-`#`, non-`{`/`}` lexeme. Captures the span
159/// so the validator can derive a line number for invalid-target errors.
160pub(crate) fn target_lexeme<'src>() -> impl Parser<'src, &'src str, RawTarget, Extra<'src>> + Clone
161{
162    any()
163        .filter(|c: &char| !c.is_whitespace() && *c != '#' && *c != '{' && *c != '}')
164        .repeated()
165        .at_least(1)
166        .to_slice()
167        .map_with(|s: &str, e| RawTarget {
168            text: s.to_string(),
169            span: e.span(),
170        })
171}

173/// `<ident> [<tags>]? (<args>)?`. Returns name, tags, args, and the
174/// span of the identifier (used for line-number reporting).
175pub(crate) fn instruction_head<'src>()
176-> impl Parser<'src, &'src str, (String, Vec<Tag>, Vec<f64>, SimpleSpan<usize>), Extra<'src>> + Clone
177{
178    ident()
179        .map_with(|name, e| (name, e.span()))
180        .then(tags_block().or_not())
181        .then(args_block().or_not())
182        .map(|(((name, span), tags), args)| {
183            (
184                name,
185                tags.unwrap_or_default(),
186                args.unwrap_or_default(),
187                span,
188            )
189        })
190}

192/// End of an instruction: optional trailing spaces/comment followed by
193/// newline, EOF, or a `}` that belongs to the enclosing REPEAT parser.
194pub(crate) fn line_end<'src>() -> impl Parser<'src, &'src str, (), Extra<'src>> + Clone {
195    trailing_pad()
196        .then(choice((
197            just('\n').ignored(),
198            just('}').rewind().ignored(),
199            end(),
200        )))
201        .ignored()
202}
203
204/// Instruction line: head + space-separated raw targets, terminated by
205/// a newline / EOF / `}`. Newline is consumed; `}` is only peeked so
206/// the enclosing REPEAT parser can consume it.
207pub(crate) fn instruction_line<'src>()
208-> impl Parser<'src, &'src str, RawSyntaxNode, Extra<'src>> + Clone {
209    instruction_head()
210        .then(
211            inline_ws1()
212                .ignore_then(target_lexeme())
213                .repeated()
214                .collect::<Vec<RawTarget>>(),
215        )
216        .map(
217            |((name, tags, args, span), targets)| RawSyntaxNode::Instruction {
218                name,
219                tags,
220                args,
221                targets,
222                span,
223            },
224        )
225        .then_ignore(line_end())
226}

228/// `REPEAT <count> { <body> }`. `body` is the parser for a list of
229/// nodes (instructions and nested REPEATs).
230fn repeat_block<'src>(
231    body: impl Parser<'src, &'src str, Vec<RawSyntaxNode>, Extra<'src>> + Clone + 'src,
232) -> impl Parser<'src, &'src str, RawSyntaxNode, Extra<'src>> + Clone + 'src {
233    let digits = any()
234        .filter(|c: &char| c.is_ascii_digit())
235        .repeated()
236        .at_least(1)
237        .to_slice()
238        .try_map(|s: &str, span| {
239            s.parse::<u64>()
240                .map_err(|_| Rich::custom(span, format!("invalid REPEAT count {s:?}")))
241        });
242    just("REPEAT")
243        .map_with(|_, e| e.span())
244        .then_ignore(inline_ws1())
245        .then(digits)
246        .then_ignore(inline_pad())
247        .then_ignore(just('{'))
248        .then_ignore(pad())
249        .then(body)
250        .then_ignore(pad())
251        .then_ignore(just('}'))
252        .map(|((span, count), body)| RawSyntaxNode::Repeat { count, body, span })
253}
254
255/// Top-level program parser. Recursively defines the body shared by
256/// the program and REPEAT blocks.
257pub(crate) fn program_parser<'src>() -> impl Parser<'src, &'src str, Vec<RawSyntaxNode>, Extra<'src>>
258{
259    recursive(|body| {
260        let item = choice((repeat_block(body.clone()), instruction_line()));
261        pad()
262            .ignore_then(item)
263            .repeated()
264            .collect::<Vec<RawSyntaxNode>>()
265            .then_ignore(pad())
266    })
267    .then_ignore(end())
268}
