1//! Skill code block: symbolic Pauli propagation.
2//!
3//! Mirrors the §7 "ppvm-sym — symbolic propagation" example in
4//! `skills/ppvm-usage/SKILL.md`. We propagate a Pauli observable
5//! through a small parametric circuit with `Term`-valued coefficients,
6//! then evaluate the resulting expression at a concrete pair of
7//! angles to check that the API surface still matches what the skill
8//! tells agents to write.
9
10use ppvm_runtime::prelude::*;
11use ppvm_sym::Term;
12
13fn main() {
14    let mut sum = PauliSum::<ppvm_runtime::config::fxhash::Byte<2, Term>>::builder()
15        .n_qubits(2)
16        .build();
17    sum += ("ZZ", Term::from(1.0));
18
19    sum.rz(0, Term::var(0));
20    sum.ry(0, Term::var(1));
21    sum.cnot(0, 1);
22    sum.rx(1, Term::var(1));
23
24    let pat: PauliPattern = "Z?*".into();
25    let trace = sum.trace(&pat);
26
27    // Substitute concrete angles. The skill claims this finishing step
28    // works — assert it does and that the value is finite.
29    let value = trace.eval(&[0.3_f64, 0.5_f64]).expect("trace evaluation");
30    assert!(value.is_finite(), "non-finite trace value: {value}");
31    println!("symbolic_trace_finite=true value={value:.6}");
32}
