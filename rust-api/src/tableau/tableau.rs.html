1//! Skill code block: generalized stabilizer tableau in Rust.
2//!
3//! Mirrors the §5 "ppvm-tableau" / "Generalized stabilizer tableau"
4//! example in `skills/ppvm-usage/SKILL.md`. CI compiles + runs this,
5//! so a signature change to `GeneralizedTableau::new` or to the gate
6//! trait surface will break the build instead of silently leaving the
7//! skill stale.
8
9use ppvm_runtime::prelude::*;
10use ppvm_tableau::prelude::*;
11
12fn main() {
13    // GeneralizedTableau::new takes (n_qubits, coefficient_threshold).
14    // The third generic parameter (sparse-vector backing) defaults to
15    // Vec<(Complex64, IndexType)>, so we leave it implicit.
16    let mut tab: GeneralizedTableau<ppvm_runtime::config::indexmap::ByteFxHashF64<1>, usize> =
17        GeneralizedTableau::new(2, 1e-10);
18
19    tab.h(0);
20    tab.cnot(0, 1);
21
22    let r0 = tab.measure(0);
23    let r1 = tab.measure(1);
24
25    // GHZ: outcomes are perfectly correlated on each shot.
26    assert_eq!(r0, r1, "GHZ correlation broken: {r0:?} vs {r1:?}");
27    println!("q0={r0:?} q1={r1:?} correlated=true");
28}
