9#[derive(Debug, Clone, Copy, PartialEq, Eq, Instruction, Parse)]
10pub enum CircuitInstruction {
11    // NOTE: longer tokens need to go first
12    TwoQubitPauliError, // needs to go before T
13    Truncate,           // needs to go before T
14    Trace,              // needs to go before T
15
16    // Single-Qubit Clifford gates
17    X,
18    Y,
19    Z,
20    H,
21
22    #[token = "sqrt_x_adj"]
23    SqrtXAdj,
24
25    #[token = "sqrt_x"]
26    SqrtX,
27
28    #[token = "sqrt_y_adj"]
29    SqrtYAdj,
30
31    #[token = "sqrt_y"]
32    SqrtY,
33
34    #[token = "s_adj"]
35    SAdj,
36    S,
37
38    // Controlled gates
39    CNOT,
40    CZ,
41
42    // T gate
43    TAdj,
44    T,
45
46    // Two-qubit rotations
47    RXX,
48    RYY,
49    RZZ,
50
51    // Single-qubit rotations
52    RX,
53    RY,
54    RZ,
55
56    // U3
57    U3,
58
59    // Measurement & Reset
60    Measure,
61    Reset,
62

64    R,
65
66    // Loss
67    Loss,
68    CorrelatedLoss,
69
70    // Noise
71    PauliError,
72    Depolarize2,
73    Depolarize,
74}

131#[derive(Debug, Clone, PartialEq, Message)]
132pub enum CircuitMessage {
133    None,                                           // Truncate (no operand)
134    Qubit(usize),                                   // X, Y, Z, ...
135    QubitAndFloat(usize, f64),                      // RX, depolarize, ...
136    QubitAndTwoFloats(usize, f64, f64),             // R
137    TwoQubit(usize, usize),                         // CX, CZ
138    TwoQubitAndFloat(usize, usize, f64),            // RXX, ...
139    QubitU3(usize, f64, f64, f64),                  // U3
140    QubitAndFloatArr3(usize, [f64; 3]),             // PauliError
141    TwoQubitAndFloatArr3(usize, usize, [f64; 3]),   // Correlated loss
142    TwoQubitAndFloatArr15(usize, usize, [f64; 15]), // TwoQubitPauliError
143
144    PauliPatternStr(String), // Trace (resolved Pauli-pattern source)
145
146    // batched instructions
147    QubitBatch(SmallVec<[usize; 8]>),              // X, Y, Z, ...
148    QubitBatchAndFloat(SmallVec<[usize; 8]>, f64), // RX, depolarize, ...
149    TwoQubitBatch(SmallVec<[(usize, usize); 8]>),  // CX, CZ
150    TwoQubitBatchAndFloat(SmallVec<[(usize, usize); 8]>, f64), // RXX, ...
151    QubitBatchU3(SmallVec<[usize; 8]>, f64, f64, f64), // U3
152    QubitBatchAndFloatArr3(SmallVec<[usize; 8]>, [f64; 3]), // PauliError
153    TwoQubitBatchAndFloatArr3(SmallVec<[(usize, usize); 8]>, [f64; 3]), // Correlated loss
154    TwoQubitBatchAndFloatArr15(SmallVec<[(usize, usize); 8]>, [f64; 15]), // TwoQubitPauliError
155}

210    fn parse(src: &str) -> CircuitInstruction {
211        CircuitInstruction::parser()
212            .parse(src)
213            .into_result()
214            .unwrap_or_else(|e| panic!("parse of `{src}` failed: {e:?}"))
215    }

235    // ─── Parse: prefix-sensitive disambiguation ───────────────────────────
236    //
237    // These pairs share a prefix, so the declaration order in the enum is
238    // load-bearing: the longer token must win. These tests pin that contract.
239
240    #[test]
241    fn parses_t_family_without_prefix_collision() {
242        // `t` is a prefix of `tadj`, `trace`, `truncate`, and `twoqubitpaulierror`.
243        assert_eq!(parse("t"), T);
244        assert_eq!(parse("tadj"), TAdj);
245        assert_eq!(parse("trace"), Trace);
246        assert_eq!(parse("truncate"), Truncate);
247        assert_eq!(parse("twoqubitpaulierror"), TwoQubitPauliError);
248    }
249
250    #[test]
251    fn parses_s_family_without_prefix_collision() {
252        // `s` is a prefix of `s_adj`, `sqrt_x`, `sqrt_y`, etc.
253        assert_eq!(parse("s"), S);
254        assert_eq!(parse("s_adj"), SAdj);
255        assert_eq!(parse("sqrt_x"), SqrtX);
256        assert_eq!(parse("sqrt_x_adj"), SqrtXAdj);
257        assert_eq!(parse("sqrt_y"), SqrtY);
258        assert_eq!(parse("sqrt_y_adj"), SqrtYAdj);
259    }

288    #[test]
289    fn opcodes_are_unit_width() {
290        // All variants are field-less, so each encodes to a single byte.
291        assert_eq!(CircuitInstruction::width(), 1);
292    }

294    #[test]
295    fn opcodes_are_unique() {
296        let mut seen = std::collections::HashSet::new();
297        for inst in ALL {
298            assert!(
299                seen.insert(inst.opcode()),
300                "duplicate opcode {} for {inst:?}",
301                inst.opcode()
302            );
303        }
304        assert_eq!(seen.len(), ALL.len());
305    }
306
307    #[test]
308    fn opcodes_match_declaration_order() {
309        // Opcodes default to the variant index, which is the on-disk contract
310        // for bytecode. Reordering variants silently breaks old bytecode, so
311        // pin the assignment here.
312        for (index, inst) in ALL.iter().enumerate() {
313            assert_eq!(inst.opcode() as usize, index, "{inst:?}");
314        }
315    }
316
317    #[test]
318    fn write_then_read_round_trips_every_variant() {
319        for inst in ALL {
320            let mut buf = Vec::new();
321            inst.write_bytes(&mut buf).unwrap();
322            assert_eq!(buf, [inst.opcode()], "{inst:?} should encode to one byte");
323
324            let mut cursor = std::io::Cursor::new(buf);
325            let back = CircuitInstruction::from_bytes(&mut cursor).unwrap();
326            assert_eq!(back, *inst);
327        }
328    }

333        let err = CircuitInstruction::from_bytes(&mut cursor).unwrap_err();
334        assert!(err.to_string().contains("invalid opcode"), "err: {err}");
